32/// FastAppendAction is a transaction action for fast append data files to the table.
33pub struct FastAppendAction {
34    check_duplicate: bool,
35    // below are properties used to create SnapshotProducer when commit
36    commit_uuid: Option<Uuid>,
37    key_metadata: Option<Vec<u8>>,
38    snapshot_properties: HashMap<String, String>,
39    added_data_files: Vec<DataFile>,
40}
41
42impl FastAppendAction {
43    pub(crate) fn new() -> Self {
44        Self {
45            check_duplicate: true,
46            commit_uuid: None,
47            key_metadata: None,
48            snapshot_properties: HashMap::default(),
49            added_data_files: vec![],
50        }
51    }
52
53    /// Set whether to check duplicate files
54    pub fn with_check_duplicate(mut self, v: bool) -> Self {
55        self.check_duplicate = v;
56        self
57    }
58
59    /// Add data files to the snapshot.
60    pub fn add_data_files(mut self, data_files: impl IntoIterator<Item = DataFile>) -> Self {
61        self.added_data_files.extend(data_files);
62        self
63    }
64
65    /// Set commit UUID for the snapshot.
66    pub fn set_commit_uuid(mut self, commit_uuid: Uuid) -> Self {
67        self.commit_uuid = Some(commit_uuid);
68        self
69    }
70
71    /// Set key metadata for manifest files.
72    pub fn set_key_metadata(mut self, key_metadata: Vec<u8>) -> Self {
73        self.key_metadata = Some(key_metadata);
74        self
75    }
76
77    /// Set snapshot summary properties.
78    pub fn set_snapshot_properties(mut self, snapshot_properties: HashMap<String, String>) -> Self {
79        self.snapshot_properties = snapshot_properties;
80        self
81    }
82}

84#[async_trait]
85impl TransactionAction for FastAppendAction {
86    async fn commit(self: Arc<Self>, table: &Table) -> Result<ActionCommit> {
87        let snapshot_producer = SnapshotProducer::new(
88            table,
89            self.commit_uuid.unwrap_or_else(Uuid::now_v7),
90            self.key_metadata.clone(),
91            self.snapshot_properties.clone(),
92            self.added_data_files.clone(),
93        );
94
95        // validate added files
96        snapshot_producer.validate_added_data_files()?;
97
98        // Checks duplicate files
99        if self.check_duplicate {
100            snapshot_producer.validate_duplicate_files().await?;
101        }
102
103        snapshot_producer
104            .commit(FastAppendOperation, DefaultManifestProcess)
105            .await
106    }
107}
108
109struct FastAppendOperation;
110
111impl SnapshotProduceOperation for FastAppendOperation {
112    fn operation(&self) -> Operation {
113        Operation::Append
114    }
115
116    async fn delete_entries(
117        &self,
118        _snapshot_produce: &SnapshotProducer<'_>,
119    ) -> Result<Vec<ManifestEntry>> {
120        Ok(vec![])
121    }
122
123    async fn existing_manifest(
124        &self,
125        snapshot_produce: &SnapshotProducer<'_>,
126    ) -> Result<Vec<ManifestFile>> {
127        let Some(snapshot) = snapshot_produce.table.metadata().current_snapshot() else {
128            return Ok(vec![]);
129        };
130
131        let manifest_list = snapshot_produce
132            .table
133            .manifest_list_reader(snapshot)
134            .load()
135            .await?;
136
137        Ok(manifest_list
138            .entries()
139            .iter()
140            .filter(|entry| {
141                // Keep delete-only manifests too: they record which files were removed and
142                // must persist across snapshots until `expire_snapshots` cleans them up.
143                // Dropping them lets the removed files reappear as live data (see #2148).
144                entry.has_added_files() || entry.has_existing_files() || entry.has_deleted_files()
145            })
146            .cloned()
147            .collect())
148    }
149}

179    /// Builds a table whose current snapshot's manifest list contains a data manifest
180    /// followed by a delete-only manifest (one entry with `ManifestStatus::Deleted`,
181    /// so `deleted_files_count > 0` while `added_files_count == existing_files_count == 0`).
182    ///
183    /// Returns the table plus the `manifest_path` of the delete-only manifest so callers
184    /// can assert whether a subsequent append carries it forward.
185    async fn make_table_with_delete_only_manifest() -> (Table, TempDir, String) {
186        let tmp_dir = TempDir::new().unwrap();
187        let table_location = tmp_dir.path().join("table1");
188        let manifest_list_location = table_location.join("metadata/manifests_list_1.avro");
189        let table_metadata_location = table_location.join("metadata/v1.json");
190
191        let file_io = FileIO::new_with_fs();
192
193        let template = fs::read_to_string(format!(
194            "{}/testdata/example_table_metadata_v2.json",
195            env!("CARGO_MANIFEST_DIR")
196        ))
197        .unwrap();
198        // The template has two snapshots; point the current one at our manifest list.
199        let metadata_json = render_template(&template, context! {
200            table_location => &table_location,
201            manifest_list_1_location => &manifest_list_location,
202            manifest_list_2_location => &manifest_list_location,
203            table_metadata_1_location => &table_metadata_location,
204        });
205        let table_metadata = serde_json::from_str::<TableMetadata>(&metadata_json).unwrap();
206
207        let table = Table::builder()
208            .metadata(table_metadata)
209            .identifier(TableIdent::from_strs(["db", "table1"]).unwrap())
210            .file_io(file_io)
211            .metadata_location(table_metadata_location.to_str().unwrap())
212            .runtime(test_runtime())
213            .build()
214            .unwrap();
215
216        let current_snapshot = table.metadata().current_snapshot().unwrap();
217        let schema = current_snapshot.schema(table.metadata()).unwrap();
218        let partition_spec = table.metadata().default_partition_spec();
219
220        let next_manifest_file = |location: &str| {
221            table
222                .file_io()
223                .new_output(format!(
224                    "{}/metadata/manifest_{}.avro",
225                    location,
226                    Uuid::new_v4()
227                ))
228                .unwrap()
229        };
230        let table_location_str = table_location.to_str().unwrap().to_string();
231
232        // Data manifest: one Added data file.
233        let mut data_writer = ManifestWriterBuilder::new(
234            next_manifest_file(&table_location_str),
235            Some(current_snapshot.snapshot_id()),
236            None,
237            schema.clone(),
238            partition_spec.as_ref().clone(),
239        )
240        .build_v2_data();
241        data_writer
242            .add_entry(
243                ManifestEntry::builder()
244                    .status(ManifestStatus::Added)
245                    .data_file(
246                        DataFileBuilder::default()
247                            .partition_spec_id(0)
248                            .content(DataContentType::Data)
249                            .file_path(format!("{table_location_str}/data.parquet"))
250                            .file_format(DataFileFormat::Parquet)
251                            .file_size_in_bytes(100)
252                            .record_count(1)
253                            .partition(Struct::from_iter([Some(Literal::long(100))]))
254                            .build()
255                            .unwrap(),
256                    )
257                    .build(),
258            )
259            .unwrap();
260        let data_manifest = data_writer.write_manifest_file().await.unwrap();
261
262        // Delete-only manifest: a single Deleted entry, nothing added or existing.
263        let mut delete_writer = ManifestWriterBuilder::new(
264            next_manifest_file(&table_location_str),
265            Some(current_snapshot.snapshot_id()),
266            None,
267            schema.clone(),
268            partition_spec.as_ref().clone(),
269        )
270        .build_v2_data();
271        delete_writer
272            .add_delete_entry(
273                ManifestEntry::builder()
274                    .status(ManifestStatus::Deleted)
275                    .sequence_number(0)
276                    .file_sequence_number(0)
277                    .data_file(
278                        DataFileBuilder::default()
279                            .partition_spec_id(0)
280                            .content(DataContentType::Data)
281                            .file_path(format!("{table_location_str}/removed.parquet"))
282                            .file_format(DataFileFormat::Parquet)
283                            .file_size_in_bytes(100)
284                            .record_count(1)
285                            .partition(Struct::from_iter([Some(Literal::long(100))]))
286                            .build()
287                            .unwrap(),
288                    )
289                    .build(),
290            )
291            .unwrap();
292        let delete_manifest = delete_writer.write_manifest_file().await.unwrap();
293        let delete_manifest_path = delete_manifest.manifest_path.clone();
294
295        // Sanity: the delete manifest really is delete-only.
296        assert!(delete_manifest.has_deleted_files());
297        assert!(!delete_manifest.has_added_files());
298        assert!(!delete_manifest.has_existing_files());
299
300        let mut manifest_list_write = ManifestListWriter::v2(
301            table
302                .file_io()
303                .new_output(current_snapshot.manifest_list())
304                .unwrap()
305                .writer()
306                .await
307                .unwrap(),
308            current_snapshot.snapshot_id(),
309            current_snapshot.parent_snapshot_id(),
310            current_snapshot.sequence_number(),
311        );
312        manifest_list_write
313            .add_manifests(vec![data_manifest, delete_manifest].into_iter())
314            .unwrap();
315        manifest_list_write.close().await.unwrap();
316
317        (table, tmp_dir, delete_manifest_path)
318    }

320    /// Regression test for #2148: a `fast_append` must carry delete-only manifests
321    /// forward into the new snapshot. Dropping them lets the files they mark as
322    /// removed reappear as live data on the next append.
323    #[tokio::test]
324    async fn test_fast_append_preserves_delete_only_manifest() {
325        let (table, _tmp_dir, delete_manifest_path) = make_table_with_delete_only_manifest().await;
326
327        // Append a new data file via the public transaction API.
328        let new_file = DataFileBuilder::default()
329            .content(DataContentType::Data)
330            .file_path(format!("{}/appended.parquet", table.metadata().location()))
331            .file_format(DataFileFormat::Parquet)
332            .file_size_in_bytes(100)
333            .record_count(1)
334            .partition_spec_id(table.metadata().default_partition_spec_id())
335            .partition(Struct::from_iter([Some(Literal::long(100))]))
336            .build()
337            .unwrap();
338
339        let tx = Transaction::new(&table);
340        let action = tx.fast_append().add_data_files(vec![new_file]);
341        let mut action_commit = Arc::new(action).commit(&table).await.unwrap();
342        let updates = action_commit.take_updates();
343
344        let new_snapshot: SnapshotRef = if let TableUpdate::AddSnapshot { snapshot } = &updates[0] {
345            SnapshotRef::new(snapshot.clone())
346        } else {
347            unreachable!("first update of a fast append should be AddSnapshot")
348        };
349
350        let manifest_list = table
351            .manifest_list_reader(&new_snapshot)
352            .load()
353            .await
354            .unwrap();
355
356        assert!(
357            manifest_list
358                .entries()
359                .iter()
360                .any(|m| m.manifest_path == delete_manifest_path),
361            "delete-only manifest {delete_manifest_path} was dropped from the new snapshot's \
362             manifest list; the files it removed would reappear as live data"
363        );
364    }

469    #[tokio::test]
470    async fn test_fast_append() {
471        let table = make_v2_minimal_table();
472        let tx = Transaction::new(&table);
473        let action = tx.fast_append();
474
475        let data_file = DataFileBuilder::default()
476            .content(DataContentType::Data)
477            .file_path("test/3.parquet".to_string())
478            .file_format(DataFileFormat::Parquet)
479            .file_size_in_bytes(100)
480            .record_count(1)
481            .partition_spec_id(table.metadata().default_partition_spec_id())
482            .partition(Struct::from_iter([Some(Literal::long(300))]))
483            .build()
484            .unwrap();
485
486        let action = action.add_data_files(vec![data_file.clone()]);
487        let mut action_commit = Arc::new(action).commit(&table).await.unwrap();
488        let updates = action_commit.take_updates();
489        let requirements = action_commit.take_requirements();
490
491        // check updates and requirements
492        assert!(
493            matches!((&updates[0],&updates[1]), (TableUpdate::AddSnapshot { snapshot },TableUpdate::SetSnapshotRef { reference,ref_name }) if snapshot.snapshot_id() == reference.snapshot_id && ref_name == MAIN_BRANCH)
494        );
495        assert_eq!(
496            vec![
497                TableRequirement::UuidMatch {
498                    uuid: table.metadata().uuid()
499                },
500                TableRequirement::RefSnapshotIdMatch {
501                    r#ref: MAIN_BRANCH.to_string(),
502                    snapshot_id: table.metadata().current_snapshot_id
503                }
504            ],
505            requirements
506        );
507
508        // check manifest list
509        let new_snapshot: SnapshotRef = if let TableUpdate::AddSnapshot { snapshot } = &updates[0] {
510            SnapshotRef::new(snapshot.clone())
511        } else {
512            unreachable!()
513        };
514        let manifest_list = table
515            .manifest_list_reader(&new_snapshot)
516            .load()
517            .await
518            .unwrap();
519        assert_eq!(1, manifest_list.entries().len());
520        assert_eq!(
521            manifest_list.entries()[0].sequence_number,
522            new_snapshot.sequence_number()
523        );
524
525        // check manifest
526        let manifest = manifest_list.entries()[0]
527            .load_manifest(table.file_io())
528            .await
529            .unwrap();
530        assert_eq!(1, manifest.entries().len());
531        assert_eq!(
532            new_snapshot.sequence_number(),
533            manifest.entries()[0]
534                .sequence_number()
535                .expect("Inherit sequence number by load manifest")
536        );
537
538        assert_eq!(
539            new_snapshot.snapshot_id(),
540            manifest.entries()[0].snapshot_id().unwrap()
541        );
542        assert_eq!(data_file, *manifest.entries()[0].data_file());
543    }
