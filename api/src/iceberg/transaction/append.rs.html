32/// FastAppendAction is a transaction action for fast append data files to the table.
33pub struct FastAppendAction {
34    check_duplicate: bool,
35    // below are properties used to create SnapshotProducer when commit
36    commit_uuid: Option<Uuid>,
37    snapshot_properties: HashMap<String, String>,
38    added_data_files: Vec<DataFile>,
39}
40
41impl FastAppendAction {
42    pub(crate) fn new() -> Self {
43        Self {
44            check_duplicate: true,
45            commit_uuid: None,
46            snapshot_properties: HashMap::default(),
47            added_data_files: vec![],
48        }
49    }
50
51    /// Set whether to check duplicate files
52    pub fn with_check_duplicate(mut self, v: bool) -> Self {
53        self.check_duplicate = v;
54        self
55    }
56
57    /// Add data files to the snapshot.
58    pub fn add_data_files(mut self, data_files: impl IntoIterator<Item = DataFile>) -> Self {
59        self.added_data_files.extend(data_files);
60        self
61    }
62
63    /// Set commit UUID for the snapshot.
64    pub fn set_commit_uuid(mut self, commit_uuid: Uuid) -> Self {
65        self.commit_uuid = Some(commit_uuid);
66        self
67    }
68
69    /// Set snapshot summary properties.
70    pub fn set_snapshot_properties(mut self, snapshot_properties: HashMap<String, String>) -> Self {
71        self.snapshot_properties = snapshot_properties;
72        self
73    }
74}

76#[async_trait]
77impl TransactionAction for FastAppendAction {
78    async fn commit(self: Arc<Self>, table: &Table) -> Result<ActionCommit> {
79        let snapshot_producer = SnapshotProducer::new(
80            table,
81            self.commit_uuid.unwrap_or_else(Uuid::now_v7),
82            self.snapshot_properties.clone(),
83            self.added_data_files.clone(),
84        );
85
86        // validate added files
87        snapshot_producer.validate_added_data_files()?;
88
89        // Checks duplicate files
90        if self.check_duplicate {
91            snapshot_producer.validate_duplicate_files().await?;
92        }
93
94        snapshot_producer
95            .commit(FastAppendOperation, DefaultManifestProcess)
96            .await
97    }
98}

100struct FastAppendOperation;
101
102impl SnapshotProduceOperation for FastAppendOperation {
103    fn operation(&self) -> Operation {
104        Operation::Append
105    }
106
107    async fn delete_entries(
108        &self,
109        _snapshot_produce: &SnapshotProducer<'_>,
110    ) -> Result<Vec<ManifestEntry>> {
111        Ok(vec![])
112    }
113
114    async fn existing_manifest(
115        &self,
116        snapshot_produce: &SnapshotProducer<'_>,
117    ) -> Result<Vec<ManifestFile>> {
118        let Some(snapshot) = snapshot_produce.table.metadata().current_snapshot() else {
119            return Ok(vec![]);
120        };
121
122        let manifest_list = snapshot_produce
123            .table
124            .manifest_list_reader(snapshot)
125            .load()
126            .await?;
127
128        Ok(manifest_list
129            .entries()
130            .iter()
131            .filter(|entry| {
132                // Keep delete-only manifests too: they record which files were removed and
133                // must persist across snapshots until `expire_snapshots` cleans them up.
134                // Dropping them lets the removed files reappear as live data (see #2148).
135                entry.has_added_files() || entry.has_existing_files() || entry.has_deleted_files()
136            })
137            .cloned()
138            .collect())
139    }
140}

172    /// Builds a table whose current snapshot's manifest list contains a data manifest
173    /// followed by a delete-only manifest (one entry with `ManifestStatus::Deleted`,
174    /// so `deleted_files_count > 0` while `added_files_count == existing_files_count == 0`).
175    ///
176    /// Returns the table plus the `manifest_path` of the delete-only manifest so callers
177    /// can assert whether a subsequent append carries it forward.
178    async fn make_table_with_delete_only_manifest() -> (Table, TempDir, String) {
179        let tmp_dir = TempDir::new().unwrap();
180        let table_location = tmp_dir.path().join("table1");
181        let manifest_list_location = table_location.join("metadata/manifests_list_1.avro");
182        let table_metadata_location = table_location.join("metadata/v1.json");
183
184        let file_io = FileIO::new_with_fs();
185
186        let template = fs::read_to_string(format!(
187            "{}/testdata/example_table_metadata_v2.json",
188            env!("CARGO_MANIFEST_DIR")
189        ))
190        .unwrap();
191        // The template has two snapshots; point the current one at our manifest list.
192        let metadata_json = render_template(&template, context! {
193            table_location => &table_location,
194            manifest_list_1_location => &manifest_list_location,
195            manifest_list_2_location => &manifest_list_location,
196            table_metadata_1_location => &table_metadata_location,
197        });
198        let table_metadata = serde_json::from_str::<TableMetadata>(&metadata_json).unwrap();
199
200        let table = Table::builder()
201            .metadata(table_metadata)
202            .identifier(TableIdent::from_strs(["db", "table1"]).unwrap())
203            .file_io(file_io)
204            .metadata_location(table_metadata_location.to_str().unwrap())
205            .runtime(test_runtime())
206            .build()
207            .unwrap();
208
209        let current_snapshot = table.metadata().current_snapshot().unwrap();
210        let schema = current_snapshot.schema(table.metadata()).unwrap();
211        let partition_spec = table.metadata().default_partition_spec();
212
213        let next_manifest_file = |location: &str| {
214            table
215                .file_io()
216                .new_output(format!(
217                    "{}/metadata/manifest_{}.avro",
218                    location,
219                    Uuid::new_v4()
220                ))
221                .unwrap()
222        };
223        let table_location_str = table_location.to_str().unwrap().to_string();
224
225        // Data manifest: one Added data file.
226        let mut data_writer = ManifestWriterBuilder::new(
227            next_manifest_file(&table_location_str),
228            Some(current_snapshot.snapshot_id()),
229            schema.clone(),
230            partition_spec.as_ref().clone(),
231        )
232        .build_v2_data();
233        data_writer
234            .add_entry(
235                ManifestEntry::builder()
236                    .status(ManifestStatus::Added)
237                    .data_file(
238                        DataFileBuilder::default()
239                            .partition_spec_id(0)
240                            .content(DataContentType::Data)
241                            .file_path(format!("{table_location_str}/data.parquet"))
242                            .file_format(DataFileFormat::Parquet)
243                            .file_size_in_bytes(100)
244                            .record_count(1)
245                            .partition(Struct::from_iter([Some(Literal::long(100))]))
246                            .build()
247                            .unwrap(),
248                    )
249                    .build(),
250            )
251            .unwrap();
252        let data_manifest = data_writer.write_manifest_file().await.unwrap();
253
254        // Delete-only manifest: a single Deleted entry, nothing added or existing.
255        let mut delete_writer = ManifestWriterBuilder::new(
256            next_manifest_file(&table_location_str),
257            Some(current_snapshot.snapshot_id()),
258            schema.clone(),
259            partition_spec.as_ref().clone(),
260        )
261        .build_v2_data();
262        delete_writer
263            .add_delete_entry(
264                ManifestEntry::builder()
265                    .status(ManifestStatus::Deleted)
266                    .sequence_number(0)
267                    .file_sequence_number(0)
268                    .data_file(
269                        DataFileBuilder::default()
270                            .partition_spec_id(0)
271                            .content(DataContentType::Data)
272                            .file_path(format!("{table_location_str}/removed.parquet"))
273                            .file_format(DataFileFormat::Parquet)
274                            .file_size_in_bytes(100)
275                            .record_count(1)
276                            .partition(Struct::from_iter([Some(Literal::long(100))]))
277                            .build()
278                            .unwrap(),
279                    )
280                    .build(),
281            )
282            .unwrap();
283        let delete_manifest = delete_writer.write_manifest_file().await.unwrap();
284        let delete_manifest_path = delete_manifest.manifest_path.clone();
285
286        // Sanity: the delete manifest really is delete-only.
287        assert!(delete_manifest.has_deleted_files());
288        assert!(!delete_manifest.has_added_files());
289        assert!(!delete_manifest.has_existing_files());
290
291        let mut manifest_list_writer = ManifestListWriter::v2(
292            table
293                .file_io()
294                .new_output(current_snapshot.manifest_list())
295                .unwrap()
296                .writer()
297                .await
298                .unwrap(),
299            current_snapshot.snapshot_id(),
300            current_snapshot.parent_snapshot_id(),
301            current_snapshot.sequence_number(),
302        );
303        manifest_list_writer
304            .add_manifests(vec![data_manifest, delete_manifest].into_iter())
305            .unwrap();
306        manifest_list_writer.close().await.unwrap();
307
308        (table, tmp_dir, delete_manifest_path)
309    }

315    async fn test_fast_append_preserves_delete_only_manifest() {
316        let (table, _tmp_dir, delete_manifest_path) = make_table_with_delete_only_manifest().await;
317
318        // Append a new data file via the public transaction API.
319        let new_file = DataFileBuilder::default()
320            .content(DataContentType::Data)
321            .file_path(format!("{}/appended.parquet", table.metadata().location()))
322            .file_format(DataFileFormat::Parquet)
323            .file_size_in_bytes(100)
324            .record_count(1)
325            .partition_spec_id(table.metadata().default_partition_spec_id())
326            .partition(Struct::from_iter([Some(Literal::long(100))]))
327            .build()
328            .unwrap();
329

331        let action = tx.fast_append().add_data_files(vec![new_file]);
332        let mut action_commit = Arc::new(action).commit(&table).await.unwrap();
333        let updates = action_commit.take_updates();
334
335        let new_snapshot: SnapshotRef = if let TableUpdate::AddSnapshot { snapshot } = &updates[0] {
336            SnapshotRef::new(snapshot.clone())
337        } else {
338            unreachable!("first update of a fast append should be AddSnapshot")
339        };
340

352            "delete-only manifest {delete_manifest_path} was dropped from the new snapshot's \

517    async fn test_snapshot_properties_cannot_override_computed_metrics() {
518        // A user-supplied snapshot property must not shadow a computed metric key
519        // such as `added-data-files`. Matching iceberg-java, the computed value
520        // wins, so the summary reflects the real count and a bad value can neither
521        // corrupt the summary nor panic total computation (see #2184-adjacent fix).
522        let table = make_v2_minimal_table();
523        let tx = Transaction::new(&table);
524
525        let mut snapshot_properties = HashMap::new();
526        // Both a benign-but-wrong value and a non-integer value collide with
527        // computed metric keys; neither should reach the final summary.
528        snapshot_properties.insert("added-data-files".to_string(), "9999".to_string());
529        snapshot_properties.insert("added-records".to_string(), "not-a-number".to_string());
530
531        let data_file = DataFileBuilder::default()
532            .content(DataContentType::Data)
533            .file_path("test/1.parquet".to_string())
534            .file_format(DataFileFormat::Parquet)
535            .file_size_in_bytes(100)
536            .record_count(1)
537            .partition_spec_id(table.metadata().default_partition_spec_id())
538            .partition(Struct::from_iter([Some(Literal::long(300))]))
539            .build()
540            .unwrap();
541
542        let action = tx
543            .fast_append()
544            .set_snapshot_properties(snapshot_properties)
545            .add_data_files(vec![data_file]);
546        // Must not panic during total computation.
547        let mut action_commit = Arc::new(action).commit(&table).await.unwrap();
548        let updates = action_commit.take_updates();
549
550        let new_snapshot = if let TableUpdate::AddSnapshot { snapshot } = &updates[0] {
551            snapshot
552        } else {
553            unreachable!()
554        };
555        let props = &new_snapshot.summary().additional_properties;
556
557        // Computed metric wins over the user's colliding values.
558        assert_eq!(
559            props.get("added-data-files").unwrap(),
560            "1",
561            "computed added-data-files must override the user-supplied value"
562        );
563        assert_eq!(
564            props.get("added-records").unwrap(),
565            "1",
566            "computed added-records must override the user-supplied non-integer value"
567        );
568    }

625        let table = make_v2_minimal_table();
626        let tx = Transaction::new(&table);
627        let action = tx.fast_append();
628
629        let data_file = DataFileBuilder::default()

641        let mut action_commit = Arc::new(action).commit(&table).await.unwrap();
642        let updates = action_commit.take_updates();
643        let requirements = action_commit.take_requirements();
644
645        // check updates and requirements
646        assert!(
647            matches!((&updates[0],&updates[1]), (TableUpdate::AddSnapshot { snapshot },TableUpdate::SetSnapshotRef { reference,ref_name }) if snapshot.snapshot_id() == reference.snapshot_id && ref_name == MAIN_BRANCH)
648        );
649        assert_eq!(
650            vec![
651                TableRequirement::UuidMatch {
652                    uuid: table.metadata().uuid()
653                },
654                TableRequirement::RefSnapshotIdMatch {
655                    r#ref: MAIN_BRANCH.to_string(),
656                    snapshot_id: table.metadata().current_snapshot_id
657                }
658            ],
659            requirements
660        );
661
662        // check manifest list
663        let new_snapshot: SnapshotRef = if let TableUpdate::AddSnapshot { snapshot } = &updates[0] {
664            SnapshotRef::new(snapshot.clone())
665        } else {
666            unreachable!()
667        };
668        let manifest_list = table
669            .manifest_list_reader(&new_snapshot)
670            .load()
671            .await
672            .unwrap();
673        assert_eq!(1, manifest_list.entries().len());
674        assert_eq!(
675            manifest_list.entries()[0].sequence_number,
676            new_snapshot.sequence_number()
677        );
678
679        // check manifest
680        let manifest = manifest_list.entries()[0]
681            .load_manifest(table.file_io())
682            .await
683            .unwrap();
684        assert_eq!(1, manifest.entries().len());
685        assert_eq!(
686            new_snapshot.sequence_number(),
687            manifest.entries()[0]
688                .sequence_number()
689                .expect("Inherit sequence number by load manifest")
690        );
691
692        assert_eq!(
693            new_snapshot.snapshot_id(),
694            manifest.entries()[0].snapshot_id().unwrap()
695        );
696        assert_eq!(data_file, *manifest.entries()[0].data_file());
697    }
