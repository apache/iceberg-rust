30/// Builder to create table scan.
31pub struct TableBuilder {
32    file_io: Option<FileIO>,
33    metadata_location: Option<String>,
34    metadata: Option<TableMetadataRef>,
35    identifier: Option<TableIdent>,
36    readonly: bool,
37    disable_cache: bool,
38    cache_size_bytes: Option<u64>,
39}
40
41impl TableBuilder {
42    pub(crate) fn new() -> Self {
43        Self {
44            file_io: None,
45            metadata_location: None,
46            metadata: None,
47            identifier: None,
48            readonly: false,
49            disable_cache: false,
50            cache_size_bytes: None,
51        }
52    }
53
54    /// required - sets the necessary FileIO to use for the table
55    pub fn file_io(mut self, file_io: FileIO) -> Self {
56        self.file_io = Some(file_io);
57        self
58    }
59
60    /// optional - sets the tables metadata location
61    pub fn metadata_location<T: Into<String>>(mut self, metadata_location: T) -> Self {
62        self.metadata_location = Some(metadata_location.into());
63        self
64    }
65
66    /// required - passes in the TableMetadata to use for the Table
67    pub fn metadata<T: Into<TableMetadataRef>>(mut self, metadata: T) -> Self {
68        self.metadata = Some(metadata.into());
69        self
70    }
71
72    /// required - passes in the TableIdent to use for the Table
73    pub fn identifier(mut self, identifier: TableIdent) -> Self {
74        self.identifier = Some(identifier);
75        self
76    }
77
78    /// specifies if the Table is readonly or not (default not)
79    pub fn readonly(mut self, readonly: bool) -> Self {
80        self.readonly = readonly;
81        self
82    }
83
84    /// specifies if the Table's metadata cache will be disabled,
85    /// so that reads of Manifests and ManifestLists will never
86    /// get cached.
87    pub fn disable_cache(mut self) -> Self {
88        self.disable_cache = true;
89        self
90    }
91
92    /// optionally set a non-default metadata cache size
93    pub fn cache_size_bytes(mut self, cache_size_bytes: u64) -> Self {
94        self.cache_size_bytes = Some(cache_size_bytes);
95        self
96    }
97
98    /// build the Table
99    pub fn build(self) -> Result<Table> {
100        let Self {
101            file_io,
102            metadata_location,
103            metadata,
104            identifier,
105            readonly,
106            disable_cache,
107            cache_size_bytes,
108        } = self;
109
110        let Some(file_io) = file_io else {
111            return Err(Error::new(
112                ErrorKind::DataInvalid,
113                "FileIO must be provided with TableBuilder.file_io()",
114            ));
115        };
116
117        let Some(metadata) = metadata else {
118            return Err(Error::new(
119                ErrorKind::DataInvalid,
120                "TableMetadataRef must be provided with TableBuilder.metadata()",
121            ));
122        };
123
124        let Some(identifier) = identifier else {
125            return Err(Error::new(
126                ErrorKind::DataInvalid,
127                "TableIdent must be provided with TableBuilder.identifier()",
128            ));
129        };
130
131        let object_cache = if disable_cache {
132            Arc::new(ObjectCache::with_disabled_cache(file_io.clone()))
133        } else if let Some(cache_size_bytes) = cache_size_bytes {
134            Arc::new(ObjectCache::new_with_capacity(
135                file_io.clone(),
136                cache_size_bytes,
137            ))
138        } else {
139            Arc::new(ObjectCache::new(file_io.clone()))
140        };
141
142        Ok(Table {
143            file_io,
144            metadata_location,
145            metadata,
146            identifier,
147            readonly,
148            object_cache,
149        })
150    }
151}

153/// Table represents a table in the catalog.
154#[derive(Debug, Clone)]
155pub struct Table {
156    file_io: FileIO,
157    metadata_location: Option<String>,
158    metadata: TableMetadataRef,
159    identifier: TableIdent,
160    readonly: bool,
161    object_cache: Arc<ObjectCache>,
162}
163
164impl Table {
165    pub(crate) fn with_metadata(&mut self, metadata: TableMetadataRef) {
166        self.metadata = metadata;
167    }
168
169    /// Returns a TableBuilder to build a table
170    pub fn builder() -> TableBuilder {
171        TableBuilder::new()
172    }
173
174    /// Returns table identifier.
175    pub fn identifier(&self) -> &TableIdent {
176        &self.identifier
177    }
178    /// Returns current metadata.
179    pub fn metadata(&self) -> &TableMetadata {
180        &self.metadata
181    }
182
183    /// Returns current metadata ref.
184    pub fn metadata_ref(&self) -> TableMetadataRef {
185        self.metadata.clone()
186    }
187
188    /// Returns current metadata location.
189    pub fn metadata_location(&self) -> Option<&str> {
190        self.metadata_location.as_deref()
191    }
192
193    /// Returns file io used in this table.
194    pub fn file_io(&self) -> &FileIO {
195        &self.file_io
196    }
197
198    /// Returns this table's object cache
199    pub(crate) fn object_cache(&self) -> Arc<ObjectCache> {
200        self.object_cache.clone()
201    }
202
203    /// Creates a table scan.
204    pub fn scan(&self) -> TableScanBuilder<'_> {
205        TableScanBuilder::new(self)
206    }
207
208    /// Creates a metadata table which provides table-like APIs for inspecting metadata.
209    /// See [`MetadataTable`] for more details.
210    pub fn inspect(&self) -> MetadataTable<'_> {
211        MetadataTable::new(self)
212    }
213
214    /// Returns the flag indicating whether the `Table` is readonly or not
215    pub fn readonly(&self) -> bool {
216        self.readonly
217    }
218
219    /// Create a reader for the table.
220    pub fn reader_builder(&self) -> ArrowReaderBuilder {
221        ArrowReaderBuilder::new(self.file_io.clone())
222    }
223}

225/// `StaticTable` is a read-only table struct that can be created from a metadata file or from `TableMetaData` without a catalog.
226/// It can only be used to read metadata and for table scan.
227/// # Examples
228///
229/// ```rust, no_run
230/// # use iceberg::io::FileIO;
231/// # use iceberg::table::StaticTable;
232/// # use iceberg::TableIdent;
233/// # async fn example() {
234/// let metadata_file_location = "s3://bucket_name/path/to/metadata.json";
235/// let file_io = FileIO::from_path(&metadata_file_location)
236///     .unwrap()
237///     .build()
238///     .unwrap();
239/// let static_identifier = TableIdent::from_strs(["static_ns", "static_table"]).unwrap();
240/// let static_table =
241///     StaticTable::from_metadata_file(&metadata_file_location, static_identifier, file_io)
242///         .await
243///         .unwrap();
244/// let snapshot_id = static_table
245///     .metadata()
246///     .current_snapshot()
247///     .unwrap()
248///     .snapshot_id();
249/// # }
250/// ```
251#[derive(Debug, Clone)]
252pub struct StaticTable(Table);
253
254impl StaticTable {
255    /// Creates a static table from a given `TableMetadata` and `FileIO`
256    pub async fn from_metadata(
257        metadata: TableMetadata,
258        table_ident: TableIdent,
259        file_io: FileIO,
260    ) -> Result<Self> {
261        let table = Table::builder()
262            .metadata(metadata)
263            .identifier(table_ident)
264            .file_io(file_io.clone())
265            .readonly(true)
266            .build();
267
268        Ok(Self(table?))
269    }
270    /// Creates a static table directly from metadata file and `FileIO`
271    pub async fn from_metadata_file(
272        metadata_file_path: &str,
273        table_ident: TableIdent,
274        file_io: FileIO,
275    ) -> Result<Self> {
276        let metadata_file = file_io.new_input(metadata_file_path)?;
277        let metadata_file_content = metadata_file.read().await?;
278        let table_metadata = serde_json::from_slice::<TableMetadata>(&metadata_file_content)?;
279        Self::from_metadata(table_metadata, table_ident, file_io).await
280    }
281
282    /// Create a TableScanBuilder for the static table.
283    pub fn scan(&self) -> TableScanBuilder<'_> {
284        self.0.scan()
285    }
286
287    /// Get TableMetadataRef for the static table
288    pub fn metadata(&self) -> TableMetadataRef {
289        self.0.metadata_ref()
290    }
291
292    /// Consumes the `StaticTable` and return it as a `Table`
293    /// Please use this method carefully as the Table it returns remains detached from a catalog
294    /// and can't be used to perform modifications on the table.
295    pub fn into_table(self) -> Table {
296        self.0
297    }
298
299    /// Create a reader for the table.
300    pub fn reader_builder(&self) -> ArrowReaderBuilder {
301        ArrowReaderBuilder::new(self.0.file_io.clone())
302    }
303}

309    #[tokio::test]
310    async fn test_static_table_from_file() {
311        let metadata_file_name = "TableMetadataV2Valid.json";
312        let metadata_file_path = format!(
313            "{}/testdata/table_metadata/{}",
314            env!("CARGO_MANIFEST_DIR"),
315            metadata_file_name
316        );
317        let file_io = FileIO::from_path(&metadata_file_path)
318            .unwrap()
319            .build()
320            .unwrap();
321        let static_identifier = TableIdent::from_strs(["static_ns", "static_table"]).unwrap();
322        let static_table =
323            StaticTable::from_metadata_file(&metadata_file_path, static_identifier, file_io)
324                .await
325                .unwrap();
326        let snapshot_id = static_table
327            .metadata()
328            .current_snapshot()
329            .unwrap()
330            .snapshot_id();
331        assert_eq!(
332            snapshot_id, 3055729675574597004,
333            "snapshot id from metadata don't match"
334        );
335    }
