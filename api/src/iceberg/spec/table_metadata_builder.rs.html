36/// Manipulating table metadata.
37///
38/// For this builder the order of called functions matters. Functions are applied in-order.
39/// All operations applied to the `TableMetadata` are tracked in `changes` as  a chronologically
40/// ordered vec of `TableUpdate`.
41/// If an operation does not lead to a change of the `TableMetadata`, the corresponding update
42/// is omitted from `changes`.
43///
44/// Unlike a typical builder pattern, the order of function calls matters.
45/// Some basic rules:
46/// - `add_schema` must be called before `set_current_schema`.
47/// - If a new partition spec and schema are added, the schema should be added first.
48#[derive(Debug, Clone)]
49pub struct TableMetadataBuilder {
50    metadata: TableMetadata,
51    changes: Vec<TableUpdate>,
52    last_added_schema_id: Option<i32>,
53    last_added_spec_id: Option<i32>,
54    last_added_order_id: Option<i64>,
55    // None if this is a new table (from_metadata) method not used
56    previous_history_entry: Option<MetadataLog>,
57    last_updated_ms: Option<i64>,
58}

75    /// Create a `TableMetadata` object from scratch.
76    ///
77    /// This method re-assign ids of fields in the schema, schema.id, sort_order.id and
78    /// spec.id. It should only be used to create new table metadata from scratch.
79    pub fn new(
80        schema: Schema,
81        spec: impl Into<UnboundPartitionSpec>,
82        sort_order: SortOrder,
83        location: String,
84        format_version: FormatVersion,
85        properties: HashMap<String, String>,
86    ) -> Result<Self> {
87        // Re-assign field_ids, schema.id, sort_order.id and spec.id for a new table.
88        let (fresh_schema, fresh_spec, fresh_sort_order) =
89            Self::reassign_ids(schema, spec.into(), sort_order)?;
90        let schema_id = fresh_schema.schema_id();
91
92        let builder = Self {
93            metadata: TableMetadata {
94                format_version,
95                table_uuid: Uuid::now_v7(),
96                location: "".to_string(), // Overwritten immediately by set_location
97                last_sequence_number: 0,
98                last_updated_ms: 0,    // Overwritten by build() if not set before
99                last_column_id: -1,    // Overwritten immediately by add_current_schema
100                current_schema_id: -1, // Overwritten immediately by add_current_schema
101                schemas: HashMap::new(),
102                partition_specs: HashMap::new(),
103                default_spec: Arc::new(
104                    // The spec id (-1) is just a proxy value and can be any negative number.
105                    // 0 would lead to wrong changes in the builder if the provided spec by the user is
106                    // also unpartitioned.
107                    // The `default_spec` value is always replaced at the end of this method by he `add_default_partition_spec`
108                    // method.
109                    PartitionSpec::unpartition_spec().with_spec_id(-1),
110                ), // Overwritten immediately by add_default_partition_spec
111                default_partition_type: StructType::new(vec![]),
112                last_partition_id: UNPARTITIONED_LAST_ASSIGNED_ID,
113                properties: HashMap::new(),
114                current_snapshot_id: None,
115                snapshots: HashMap::new(),
116                snapshot_log: vec![],
117                sort_orders: HashMap::new(),
118                metadata_log: vec![],
119                default_sort_order_id: -1, // Overwritten immediately by add_default_sort_order
120                refs: HashMap::default(),
121                statistics: HashMap::new(),
122                partition_statistics: HashMap::new(),
123                encryption_keys: HashMap::new(),
124            },
125            last_updated_ms: None,
126            changes: vec![],
127            last_added_schema_id: Some(schema_id),
128            last_added_spec_id: None,
129            last_added_order_id: None,
130            previous_history_entry: None,
131        };
132
133        builder
134            .set_location(location)
135            .add_current_schema(fresh_schema)?
136            .add_default_partition_spec(fresh_spec.into_unbound())?
137            .add_default_sort_order(fresh_sort_order)?
138            .set_properties(properties)
139    }

207    /// Upgrade `FormatVersion`. Downgrades are not allowed.
208    ///
209    /// # Errors
210    /// - Cannot downgrade to older format versions.
211    pub fn upgrade_format_version(mut self, format_version: FormatVersion) -> Result<Self> {
212        if format_version < self.metadata.format_version {
213            return Err(Error::new(
214                ErrorKind::DataInvalid,
215                format!(
216                    "Cannot downgrade FormatVersion from {} to {}",
217                    self.metadata.format_version, format_version
218                ),
219            ));
220        }
221
222        if format_version != self.metadata.format_version {
223            match format_version {
224                FormatVersion::V1 => {
225                    // No changes needed for V1
226                }
227                FormatVersion::V2 => {
228                    self.metadata.format_version = format_version;
229                    self.changes
230                        .push(TableUpdate::UpgradeFormatVersion { format_version });
231                }
232            }
233        }
234
235        Ok(self)
236    }

238    /// Set properties. If a property already exists, it will be overwritten.
239    ///
240    /// If a reserved property is set, the corresponding action is performed and the property is not persisted.
241    /// Currently the following reserved properties are supported:
242    /// * format-version: Set the format version of the table.
243    ///
244    /// # Errors
245    /// - If properties contains a reserved property
246    pub fn set_properties(mut self, properties: HashMap<String, String>) -> Result<Self> {
247        // List of specified properties that are RESERVED and should not be persisted.
248        let reserved_properties = properties
249            .keys()
250            .filter(|key| RESERVED_PROPERTIES.contains(&key.as_str()))
251            .map(ToString::to_string)
252            .collect::<Vec<_>>();
253
254        if !reserved_properties.is_empty() {
255            return Err(Error::new(
256                ErrorKind::DataInvalid,
257                format!(
258                    "Table properties should not contain reserved properties, but got: [{}]",
259                    reserved_properties.join(", ")
260                ),
261            ));
262        }
263
264        if properties.is_empty() {
265            return Ok(self);
266        }
267
268        self.metadata.properties.extend(properties.clone());
269        self.changes.push(TableUpdate::SetProperties {
270            updates: properties,
271        });
272
273        Ok(self)
274    }

276    /// Remove properties from the table metadata.
277    /// Does nothing if the key is not present.
278    ///
279    /// # Errors
280    /// - If properties to remove contains a reserved property
281    pub fn remove_properties(mut self, properties: &[String]) -> Result<Self> {
282        // remove duplicates
283        let properties = properties.iter().cloned().collect::<HashSet<_>>();
284
285        // disallow removal of reserved properties
286        let reserved_properties = properties
287            .iter()
288            .filter(|key| RESERVED_PROPERTIES.contains(&key.as_str()))
289            .map(ToString::to_string)
290            .collect::<Vec<_>>();
291
292        if !reserved_properties.is_empty() {
293            return Err(Error::new(
294                ErrorKind::DataInvalid,
295                format!(
296                    "Table properties to remove contain reserved properties: [{}]",
297                    reserved_properties.join(", ")
298                ),
299            ));
300        }
301
302        for property in &properties {
303            self.metadata.properties.remove(property);
304        }
305
306        if !properties.is_empty() {
307            self.changes.push(TableUpdate::RemoveProperties {
308                removals: properties.into_iter().collect(),
309            });
310        }
311
312        Ok(self)
313    }

328    /// Add a snapshot to the table metadata.
329    ///
330    /// # Errors
331    /// - Snapshot id already exists.
332    /// - For format version > 1: the sequence number of the snapshot is lower than the highest sequence number specified so far.
333    pub fn add_snapshot(mut self, snapshot: Snapshot) -> Result<Self> {
334        if self
335            .metadata
336            .snapshots
337            .contains_key(&snapshot.snapshot_id())
338        {
339            return Err(Error::new(
340                ErrorKind::DataInvalid,
341                format!("Snapshot already exists for: '{}'", snapshot.snapshot_id()),
342            ));
343        }
344
345        if self.metadata.format_version != FormatVersion::V1
346            && snapshot.sequence_number() <= self.metadata.last_sequence_number
347            && snapshot.parent_snapshot_id().is_some()
348        {
349            return Err(Error::new(
350                ErrorKind::DataInvalid,
351                format!(
352                    "Cannot add snapshot with sequence number {} older than last sequence number {}",
353                    snapshot.sequence_number(),
354                    self.metadata.last_sequence_number
355                ),
356            ));
357        }
358
359        if let Some(last) = self.metadata.snapshot_log.last() {
360            // commits can happen concurrently from different machines.
361            // A tolerance helps us avoid failure for small clock skew
362            if snapshot.timestamp_ms() - last.timestamp_ms < -ONE_MINUTE_MS {
363                return Err(Error::new(
364                    ErrorKind::DataInvalid,
365                    format!(
366                        "Invalid snapshot timestamp {}: before last snapshot timestamp {}",
367                        snapshot.timestamp_ms(),
368                        last.timestamp_ms
369                    ),
370                ));
371            }
372        }
373
374        let max_last_updated = self
375            .last_updated_ms
376            .unwrap_or_default()
377            .max(self.metadata.last_updated_ms);
378        if snapshot.timestamp_ms() - max_last_updated < -ONE_MINUTE_MS {
379            return Err(Error::new(
380                ErrorKind::DataInvalid,
381                format!(
382                    "Invalid snapshot timestamp {}: before last updated timestamp {}",
383                    snapshot.timestamp_ms(),
384                    max_last_updated
385                ),
386            ));
387        }
388
389        // Mutation happens in next line - must be infallible from here
390        self.changes.push(TableUpdate::AddSnapshot {
391            snapshot: snapshot.clone(),
392        });
393
394        self.last_updated_ms = Some(snapshot.timestamp_ms());
395        self.metadata.last_sequence_number = snapshot.sequence_number();
396        self.metadata
397            .snapshots
398            .insert(snapshot.snapshot_id(), snapshot.into());
399
400        Ok(self)
401    }

403    /// Append a snapshot to the specified branch.
404    /// Retention settings from the `branch` are re-used.
405    ///
406    /// # Errors
407    /// - Any of the preconditions of `self.add_snapshot` are not met.
408    pub fn set_branch_snapshot(self, snapshot: Snapshot, branch: &str) -> Result<Self> {
409        let reference = self.metadata.refs.get(branch).cloned();
410
411        let reference = if let Some(mut reference) = reference {
412            if !reference.is_branch() {
413                return Err(Error::new(
414                    ErrorKind::DataInvalid,
415                    format!("Cannot append snapshot to non-branch reference '{branch}'",),
416                ));
417            }
418
419            reference.snapshot_id = snapshot.snapshot_id();
420            reference
421        } else {
422            SnapshotReference {
423                snapshot_id: snapshot.snapshot_id(),
424                retention: SnapshotRetention::Branch {
425                    min_snapshots_to_keep: None,
426                    max_snapshot_age_ms: None,
427                    max_ref_age_ms: None,
428                },
429            }
430        };
431
432        self.add_snapshot(snapshot)?.set_ref(branch, reference)
433    }

435    /// Remove snapshots by its ids from the table metadata.
436    /// Does nothing if a snapshot id is not present.
437    /// Keeps as changes only the snapshots that were actually removed.
438    pub fn remove_snapshots(mut self, snapshot_ids: &[i64]) -> Self {
439        let mut removed_snapshots = Vec::with_capacity(snapshot_ids.len());
440
441        self.metadata.snapshots.retain(|k, _| {
442            if snapshot_ids.contains(k) {
443                removed_snapshots.push(*k);
444                false
445            } else {
446                true
447            }
448        });
449
450        if !removed_snapshots.is_empty() {
451            self.changes.push(TableUpdate::RemoveSnapshots {
452                snapshot_ids: removed_snapshots,
453            });
454        }
455
456        // Remove refs that are no longer valid
457        self.metadata
458            .refs
459            .retain(|_, v| self.metadata.snapshots.contains_key(&v.snapshot_id));
460
461        self
462    }

464    /// Set a reference to a snapshot.
465    ///
466    /// # Errors
467    /// - The snapshot id is unknown.
468    pub fn set_ref(mut self, ref_name: &str, reference: SnapshotReference) -> Result<Self> {
469        if self
470            .metadata
471            .refs
472            .get(ref_name)
473            .is_some_and(|snap_ref| snap_ref.eq(&reference))
474        {
475            return Ok(self);
476        }
477
478        let Some(snapshot) = self.metadata.snapshots.get(&reference.snapshot_id) else {
479            return Err(Error::new(
480                ErrorKind::DataInvalid,
481                format!(
482                    "Cannot set '{ref_name}' to unknown snapshot: '{}'",
483                    reference.snapshot_id
484                ),
485            ));
486        };
487
488        // Update last_updated_ms to the exact timestamp of the snapshot if it was added in this commit
489        let is_added_snapshot = self.changes.iter().any(|update| {
490            matches!(update, TableUpdate::AddSnapshot { snapshot: snap } if snap.snapshot_id() == snapshot.snapshot_id())
491        });
492        if is_added_snapshot {
493            self.last_updated_ms = Some(snapshot.timestamp_ms());
494        }
495
496        // Current snapshot id is set only for the main branch
497        if ref_name == MAIN_BRANCH {
498            self.metadata.current_snapshot_id = Some(snapshot.snapshot_id());
499            let timestamp_ms = if let Some(last_updated_ms) = self.last_updated_ms {
500                last_updated_ms
501            } else {
502                let last_updated_ms = chrono::Utc::now().timestamp_millis();
503                self.last_updated_ms = Some(last_updated_ms);
504                last_updated_ms
505            };
506
507            self.metadata.snapshot_log.push(SnapshotLog {
508                snapshot_id: snapshot.snapshot_id(),
509                timestamp_ms,
510            });
511        }
512
513        self.changes.push(TableUpdate::SetSnapshotRef {
514            ref_name: ref_name.to_string(),
515            reference: reference.clone(),
516        });
517        self.metadata.refs.insert(ref_name.to_string(), reference);
518
519        Ok(self)
520    }

586    /// Add a schema to the table metadata.
587    ///
588    /// The provided `schema.schema_id` may not be used.
589    ///
590    /// Important: Use this method with caution. The builder does not check
591    /// if the added schema is compatible with the current schema.
592    pub fn add_schema(mut self, schema: Schema) -> Self {
593        let new_schema_id = self.reuse_or_create_new_schema_id(&schema);
594        let schema_found = self.metadata.schemas.contains_key(&new_schema_id);
595
596        if schema_found {
597            if self.last_added_schema_id != Some(new_schema_id) {
598                self.changes.push(TableUpdate::AddSchema {
599                    schema: schema.clone(),
600                });
601                self.last_added_schema_id = Some(new_schema_id);
602            }
603
604            return self;
605        }
606
607        // New schemas might contain only old columns. In this case last_column_id should not be
608        // reduced.
609        self.metadata.last_column_id =
610            std::cmp::max(self.metadata.last_column_id, schema.highest_field_id());
611
612        // Set schema-id
613        let schema = match new_schema_id == schema.schema_id() {
614            true => schema,
615            false => schema.with_schema_id(new_schema_id),
616        };
617
618        self.metadata
619            .schemas
620            .insert(new_schema_id, schema.clone().into());
621
622        self.changes.push(TableUpdate::AddSchema { schema });
623
624        self.last_added_schema_id = Some(new_schema_id);
625
626        self
627    }

629    /// Set the current schema id.
630    ///
631    /// If `schema_id` is -1, the last added schema is set as the current schema.
632    ///
633    /// Errors:
634    /// - provided `schema_id` is -1 but no schema has been added via `add_schema`.
635    /// - No schema with the provided `schema_id` exists.
636    pub fn set_current_schema(mut self, mut schema_id: i32) -> Result<Self> {
637        if schema_id == Self::LAST_ADDED {
638            schema_id = self.last_added_schema_id.ok_or_else(|| {
639                Error::new(
640                    ErrorKind::DataInvalid,
641                    "Cannot set current schema to last added schema: no schema has been added.",
642                )
643            })?;
644        };
645        let schema_id = schema_id; // Make immutable
646
647        if schema_id == self.metadata.current_schema_id {
648            return Ok(self);
649        }
650
651        let _schema = self.metadata.schemas.get(&schema_id).ok_or_else(|| {
652            Error::new(
653                ErrorKind::DataInvalid,
654                format!(
655                    "Cannot set current schema to unknown schema with id: '{}'",
656                    schema_id
657                ),
658            )
659        })?;
660
661        // Old partition specs and sort-orders should be preserved even if they are not compatible with the new schema,
662        // so that older metadata can still be interpreted.
663        // Default partition spec and sort order are checked in the build() method
664        // which allows other default partition specs and sort orders to be set before the build.
665
666        self.metadata.current_schema_id = schema_id;
667
668        if self.last_added_schema_id == Some(schema_id) {
669            self.changes.push(TableUpdate::SetCurrentSchema {
670                schema_id: Self::LAST_ADDED,
671            });
672        } else {
673            self.changes
674                .push(TableUpdate::SetCurrentSchema { schema_id });
675        }
676
677        Ok(self)
678    }

685    /// Add a partition spec to the table metadata.
686    ///
687    /// The spec is bound eagerly to the current schema.
688    /// If a schema is added in the same set of changes, the schema should be added first.
689    ///
690    /// Even if `unbound_spec.spec_id` is provided as `Some`, it may not be used.
691    ///
692    /// # Errors
693    /// - The partition spec cannot be bound to the current schema.
694    /// - The partition spec has non-sequential field ids and the table format version is 1.
695    pub fn add_partition_spec(mut self, unbound_spec: UnboundPartitionSpec) -> Result<Self> {
696        let schema = self.get_current_schema()?.clone();
697        let spec = PartitionSpecBuilder::new_from_unbound(unbound_spec.clone(), schema)?
698            .with_last_assigned_field_id(self.metadata.last_partition_id)
699            .build()?;
700
701        let new_spec_id = self.reuse_or_create_new_spec_id(&spec);
702        let spec_found = self.metadata.partition_specs.contains_key(&new_spec_id);
703        let spec = spec.with_spec_id(new_spec_id);
704        let unbound_spec = unbound_spec.with_spec_id(new_spec_id);
705
706        if spec_found {
707            if self.last_added_spec_id != Some(new_spec_id) {
708                self.changes
709                    .push(TableUpdate::AddSpec { spec: unbound_spec });
710                self.last_added_spec_id = Some(new_spec_id);
711            }
712
713            return Ok(self);
714        }
715
716        if self.metadata.format_version <= FormatVersion::V1 && !spec.has_sequential_ids() {
717            return Err(Error::new(
718                ErrorKind::DataInvalid,
719                "Cannot add partition spec with non-sequential field ids to format version 1 table",
720            ));
721        }
722
723        let highest_field_id = spec
724            .highest_field_id()
725            .unwrap_or(UNPARTITIONED_LAST_ASSIGNED_ID);
726        self.metadata
727            .partition_specs
728            .insert(new_spec_id, Arc::new(spec));
729        self.changes
730            .push(TableUpdate::AddSpec { spec: unbound_spec });
731
732        self.last_added_spec_id = Some(new_spec_id);
733        self.metadata.last_partition_id =
734            std::cmp::max(self.metadata.last_partition_id, highest_field_id);
735
736        Ok(self)
737    }

744    pub fn set_default_partition_spec(mut self, mut spec_id: i32) -> Result<Self> {
745        if spec_id == Self::LAST_ADDED {
746            spec_id = self.last_added_spec_id.ok_or_else(|| {
747                Error::new(
748                    ErrorKind::DataInvalid,
749                    "Cannot set default partition spec to last added spec: no spec has been added.",
750                )
751            })?;
752        }
753
754        if self.metadata.default_spec.spec_id() == spec_id {
755            return Ok(self);
756        }
757
758        if !self.metadata.partition_specs.contains_key(&spec_id) {
759            return Err(Error::new(
760                ErrorKind::DataInvalid,
761                format!("Cannot set default partition spec to unknown spec with id: '{spec_id}'",),

763        }
764
765        let schemaless_spec = self
766            .metadata
767            .partition_specs
768            .get(&spec_id)
769            .ok_or_else(|| {
770                Error::new(
771                    ErrorKind::DataInvalid,
772                    format!(
773                        "Cannot set default partition spec to unknown spec with id: '{spec_id}'",
774                    ),
775                )
776            })?
777            .clone();
778        let spec = Arc::unwrap_or_clone(schemaless_spec);
779        let spec_type = spec.partition_type(self.get_current_schema()?)?;
780        self.metadata.default_spec = Arc::new(spec);

785                spec_id: Self::LAST_ADDED,
786            });
787        } else {

830    /// Add a sort order to the table metadata.
831    ///
832    /// The spec is bound eagerly to the current schema and must be valid for it.
833    /// If a schema is added in the same set of changes, the schema should be added first.
834    ///
835    /// Even if `sort_order.order_id` is provided, it may not be used.
836    ///
837    /// # Errors
838    /// - Sort order id to add already exists.
839    /// - Sort order is incompatible with the current schema.
840    pub fn add_sort_order(mut self, sort_order: SortOrder) -> Result<Self> {
841        let new_order_id = self.reuse_or_create_new_sort_id(&sort_order);
842        let sort_order_found = self.metadata.sort_orders.contains_key(&new_order_id);
843
844        if sort_order_found {
845            if self.last_added_order_id != Some(new_order_id) {
846                self.changes.push(TableUpdate::AddSortOrder {
847                    sort_order: sort_order.clone().with_order_id(new_order_id),
848                });
849                self.last_added_order_id = Some(new_order_id);
850            }
851
852            return Ok(self);
853        }
854
855        let schema = self.get_current_schema()?.clone().as_ref().clone();
856        let sort_order = SortOrder::builder()
857            .with_order_id(new_order_id)
858            .with_fields(sort_order.fields)
859            .build(&schema)
860            .map_err(|e| {
861                Error::new(
862                    ErrorKind::DataInvalid,
863                    format!("Sort order to add is incompatible with current schema: {e}"),
864                )
865                .with_source(e)
866            })?;
867
868        self.last_added_order_id = Some(new_order_id);
869        self.metadata
870            .sort_orders
871            .insert(new_order_id, sort_order.clone().into());
872        self.changes.push(TableUpdate::AddSortOrder { sort_order });
873
874        Ok(self)
875    }

925    /// Build the table metadata.
926    pub fn build(mut self) -> Result<TableMetadataBuildResult> {
927        self.metadata.last_updated_ms = self
928            .last_updated_ms
929            .unwrap_or_else(|| chrono::Utc::now().timestamp_millis());
930
931        // Check compatibility of the current schema to the default partition spec and sort order.
932        // We use the `get_xxx` methods from the builder to avoid using the panicking
933        // `TableMetadata.default_partition_spec` etc. methods.
934        let schema = self.get_current_schema()?.clone();
935        let sort_order = Arc::unwrap_or_clone(self.get_default_sort_order()?);
936
937        self.metadata.default_spec = Arc::new(
938            Arc::unwrap_or_clone(self.metadata.default_spec)
939                .into_unbound()
940                .bind(schema.clone())?,
941        );
942        self.metadata.default_partition_type =
943            self.metadata.default_spec.partition_type(&schema)?;
944        SortOrder::builder()
945            .with_fields(sort_order.fields)
946            .build(&schema)?;
947
948        self.update_snapshot_log()?;
949        self.metadata.try_normalize()?;
950
951        if let Some(hist_entry) = self.previous_history_entry.take() {
952            self.metadata.metadata_log.push(hist_entry);
953        }
954        let expired_metadata_logs = self.expire_metadata_log();
955
956        Ok(TableMetadataBuildResult {
957            metadata: self.metadata,
958            changes: self.changes,
959            expired_metadata_logs,
960        })
961    }

963    fn expire_metadata_log(&mut self) -> Vec<MetadataLog> {
964        let max_size = self
965            .metadata
966            .properties

969            .unwrap_or(PROPERTY_METADATA_PREVIOUS_VERSIONS_MAX_DEFAULT)

982    fn update_snapshot_log(&mut self) -> Result<()> {
983        let intermediate_snapshots = self.get_intermediate_snapshots();
984        let has_removed_snapshots = self
985            .changes
986            .iter()
987            .any(|update| matches!(update, TableUpdate::RemoveSnapshots { .. }));
988
989        if intermediate_snapshots.is_empty() && !has_removed_snapshots {
990            return Ok(());
991        }
992
993        let mut new_snapshot_log = Vec::new();
994        for log_entry in &self.metadata.snapshot_log {
995            let snapshot_id = log_entry.snapshot_id;
996            if self.metadata.snapshots.contains_key(&snapshot_id) {
997                if !intermediate_snapshots.contains(&snapshot_id) {
998                    new_snapshot_log.push(log_entry.clone());
999                }
1000            } else if has_removed_snapshots {
1001                // any invalid entry causes the history before it to be removed. otherwise, there could be
1002                // history gaps that cause time-travel queries to produce incorrect results. for example,
1003                // if history is [(t1, s1), (t2, s2), (t3, s3)] and s2 is removed, the history cannot be
1004                // [(t1, s1), (t3, s3)] because it appears that s3 was current during the time between t2
1005                // and t3 when in fact s2 was the current snapshot.
1006                new_snapshot_log.clear();
1007            }
1008        }
1009
1010        if let Some(current_snapshot_id) = self.metadata.current_snapshot_id {
1011            let last_id = new_snapshot_log.last().map(|entry| entry.snapshot_id);
1012            if last_id != Some(current_snapshot_id) {
1013                return Err(Error::new(
1014                    ErrorKind::DataInvalid,
1015                    "Cannot set invalid snapshot log: latest entry is not the current snapshot",
1016                ));
1017            }
1018        };
1019
1020        self.metadata.snapshot_log = new_snapshot_log;
1021        Ok(())
1022    }

1024    /// Finds intermediate snapshots that have not been committed as the current snapshot.
1025    ///
1026    /// Transactions can create snapshots that are never the current snapshot because several
1027    /// changes are combined by the transaction into one table metadata update. when each
1028    /// intermediate snapshot is added to table metadata, it is added to the snapshot log, assuming
1029    /// that it will be the current snapshot. when there are multiple snapshot updates, the log must
1030    /// be corrected by suppressing the intermediate snapshot entries.
1031    ///     
1032    /// A snapshot is an intermediate snapshot if it was added but is not the current snapshot.
1033    fn get_intermediate_snapshots(&self) -> HashSet<i64> {
1034        let added_snapshot_ids = self
1035            .changes
1036            .iter()
1037            .filter_map(|update| match update {
1038                TableUpdate::AddSnapshot { snapshot } => Some(snapshot.snapshot_id()),
1039                _ => None,
1040            })
1041            .collect::<HashSet<_>>();
1042
1043        self.changes
1044            .iter()
1045            .filter_map(|update| match update {
1046                TableUpdate::SetSnapshotRef {
1047                    ref_name,
1048                    reference,
1049                } => {
1050                    if added_snapshot_ids.contains(&reference.snapshot_id)
1051                        && ref_name == MAIN_BRANCH
1052                        && reference.snapshot_id
1053                            != self
1054                                .metadata
1055                                .current_snapshot_id
1056                                .unwrap_or(i64::from(Self::LAST_ADDED))
1057                    {
1058                        Some(reference.snapshot_id)
1059                    } else {
1060                        None
1061                    }
1062                }
1063                _ => None,
1064            })
1065            .collect()
1066    }

1068    fn reassign_ids(
1069        schema: Schema,
1070        spec: UnboundPartitionSpec,
1071        sort_order: SortOrder,
1072    ) -> Result<(Schema, PartitionSpec, SortOrder)> {
1073        // Re-assign field ids and schema ids for a new table.
1074        let previous_id_to_name = schema.field_id_to_name_map().clone();
1075        let fresh_schema = schema
1076            .into_builder()
1077            .with_schema_id(DEFAULT_SCHEMA_ID)
1078            .with_reassigned_field_ids(FIRST_FIELD_ID)
1079            .build()?;
1080
1081        // Re-build partition spec with new ids
1082        let mut fresh_spec = PartitionSpecBuilder::new(fresh_schema.clone());
1083        for field in spec.fields() {
1084            let source_field_name = previous_id_to_name.get(&field.source_id).ok_or_else(|| {
1085                Error::new(
1086                    ErrorKind::DataInvalid,
1087                    format!(
1088                        "Cannot find source column with id {} for partition column {} in schema.",
1089                        field.source_id, field.name
1090                    ),
1091                )
1092            })?;
1093            fresh_spec =
1094                fresh_spec.add_partition_field(source_field_name, &field.name, field.transform)?;
1095        }
1096        let fresh_spec = fresh_spec.build()?;
1097
1098        // Re-build sort order with new ids
1099        let mut fresh_order = SortOrder::builder();
1100        for mut field in sort_order.fields {
1101            let source_field_name = previous_id_to_name.get(&field.source_id).ok_or_else(|| {
1102                Error::new(
1103                    ErrorKind::DataInvalid,
1104                    format!(
1105                        "Cannot find source column with id {} for sort column in schema.",
1106                        field.source_id
1107                    ),
1108                )
1109            })?;
1110            let new_field_id = fresh_schema
1111                       .field_by_name(source_field_name)
1112                       .ok_or_else(|| {
1113                           Error::new(
1114                               ErrorKind::Unexpected,
1115                               format!(
1116                                   "Cannot find source column with name {} for sort column in re-assigned schema.",
1117                                   source_field_name
1118                               ),
1119                           )
1120                       })?.id;
1121            field.source_id = new_field_id;
1122            fresh_order.with_sort_field(field);
1123        }
1124        let fresh_sort_order = fresh_order.build(&fresh_schema)?;
1125
1126        Ok((fresh_schema, fresh_spec, fresh_sort_order))
1127    }

1235                false

1237                true
1238            }

1242            schema_ids: removed_schemas,
