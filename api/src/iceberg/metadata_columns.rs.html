18//! Metadata columns (virtual/reserved fields) for Iceberg tables.
19//!
20//! This module defines metadata columns that can be requested in projections
21//! but are not stored in data files. Instead, they are computed on-the-fly
22//! during reading. Examples include the _file column (file path) and future
23//! columns like partition values or row numbers.
24
25use std::sync::Arc;
26
27use once_cell::sync::Lazy;
28
29use crate::spec::{NestedField, NestedFieldRef, PrimitiveType, Type};
30use crate::{Error, ErrorKind, Result};
31
32/// Reserved field ID for the file path (_file) column per Iceberg spec
33pub const RESERVED_FIELD_ID_FILE: i32 = i32::MAX - 1;
34
35/// Reserved column name for the file path metadata column
36pub const RESERVED_COL_NAME_FILE: &str = "_file";
37
38/// Documentation for the _file metadata column
39pub const RESERVED_COL_DOC_FILE: &str = "Path of the file in which a row is stored";
40
41/// Lazy-initialized Iceberg field definition for the _file metadata column.
42/// This field represents the file path as a required string field.
43static FILE_FIELD: Lazy<NestedFieldRef> = Lazy::new(|| {
44    Arc::new(
45        NestedField::required(
46            RESERVED_FIELD_ID_FILE,
47            RESERVED_COL_NAME_FILE,
48            Type::Primitive(PrimitiveType::String),
49        )
50        .with_doc(RESERVED_COL_DOC_FILE),
51    )
52});
53
54/// Returns the Iceberg field definition for the _file metadata column.
55///
56/// # Returns
57/// A reference to the _file field definition as an Iceberg NestedField
58pub fn file_field() -> &'static NestedFieldRef {
59    &FILE_FIELD
60}
61
62/// Returns the Iceberg field definition for a metadata field ID.
63///
64/// # Arguments
65/// * `field_id` - The metadata field ID
66///
67/// # Returns
68/// The Iceberg field definition for the metadata column, or an error if not a metadata field
69pub fn get_metadata_field(field_id: i32) -> Result<NestedFieldRef> {
70    match field_id {
71        RESERVED_FIELD_ID_FILE => Ok(Arc::clone(file_field())),
72        _ if is_metadata_field(field_id) => {
73            // Future metadata fields can be added here
74            Err(Error::new(
75                ErrorKind::Unexpected,
76                format!(
77                    "Metadata field ID {field_id} recognized but field definition not implemented"
78                ),
79            ))
80        }
81        _ => Err(Error::new(
82            ErrorKind::Unexpected,
83            format!("Field ID {field_id} is not a metadata field"),
84        )),
85    }
86}
87
88/// Returns the field ID for a metadata column name.
89///
90/// # Arguments
91/// * `column_name` - The metadata column name
92///
93/// # Returns
94/// The field ID of the metadata column, or an error if the column name is not recognized
95pub fn get_metadata_field_id(column_name: &str) -> Result<i32> {
96    match column_name {
97        RESERVED_COL_NAME_FILE => Ok(RESERVED_FIELD_ID_FILE),
98        _ => Err(Error::new(
99            ErrorKind::Unexpected,
100            format!("Unknown/unsupported metadata column name: {column_name}"),
101        )),
102    }
103}
104
105/// Checks if a field ID is a metadata field.
106///
107/// # Arguments
108/// * `field_id` - The field ID to check
109///
110/// # Returns
111/// `true` if the field ID is a (currently supported) metadata field, `false` otherwise
112pub fn is_metadata_field(field_id: i32) -> bool {
113    field_id == RESERVED_FIELD_ID_FILE
114    // Additional metadata fields can be checked here in the future
115}
116
117/// Checks if a column name is a metadata column.
118///
119/// # Arguments
120/// * `column_name` - The column name to check
121///
122/// # Returns
123/// `true` if the column name is a metadata column, `false` otherwise
124pub fn is_metadata_column_name(column_name: &str) -> bool {
125    get_metadata_field_id(column_name).is_ok()
126}
