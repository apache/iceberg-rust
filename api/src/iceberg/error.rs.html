27/// ErrorKind is all kinds of Error of iceberg.
28#[derive(Clone, Copy, Debug, PartialEq, Eq)]
29#[non_exhaustive]
30pub enum ErrorKind {
31    /// The operation was rejected because the system is not in a state required for the operationâ€™s execution.
32    PreconditionFailed,
33
34    /// Iceberg don't know what happened here, and no actions other than
35    /// just returning it back. For example, iceberg returns an internal
36    /// service error.
37    Unexpected,
38
39    /// Iceberg data is invalid.
40    ///
41    /// This error is returned when we try to read a table from iceberg but
42    /// failed to parse it's metadata or data file correctly.
43    ///
44    /// The table could be invalid or corrupted.
45    DataInvalid,
46
47    /// Iceberg namespace already exists at creation.
48    NamespaceAlreadyExists,
49
50    /// Iceberg table already exists at creation.
51    TableAlreadyExists,
52
53    /// Iceberg namespace already exists at creation.
54    NamespaceNotFound,
55
56    /// Iceberg table already exists at creation.
57    TableNotFound,
58
59    /// Iceberg feature is not supported.
60    ///
61    /// This error is returned when given iceberg feature is not supported.
62    FeatureUnsupported,
63}
64
65impl ErrorKind {
66    /// Convert self into static str.
67    pub fn into_static(self) -> &'static str {
68        self.into()
69    }
70}

72impl From<ErrorKind> for &'static str {
73    fn from(v: ErrorKind) -> &'static str {
74        match v {
75            ErrorKind::Unexpected => "Unexpected",
76            ErrorKind::DataInvalid => "DataInvalid",
77            ErrorKind::FeatureUnsupported => "FeatureUnsupported",
78            ErrorKind::TableAlreadyExists => "TableAlreadyExists",
79            ErrorKind::TableNotFound => "TableNotFound",
80            ErrorKind::NamespaceAlreadyExists => "NamespaceAlreadyExists",
81            ErrorKind::NamespaceNotFound => "NamespaceNotFound",
82            ErrorKind::PreconditionFailed => "PreconditionFailed",
83        }
84    }
85}

93/// Error is the error struct returned by all iceberg functions.
94///
95/// ## Display
96///
97/// Error can be displayed in two ways:
98///
99/// - Via `Display`: like `err.to_string()` or `format!("{err}")`
100///
101/// Error will be printed in a single line:
102///
103/// ```shell
104/// Unexpected, context: { path: /path/to/file, called: send_async } => something wrong happened, source: networking error"
105/// ```
106///
107/// - Via `Debug`: like `format!("{err:?}")`
108///
109/// Error will be printed in multi lines with more details and backtraces (if captured):
110///
111/// ```shell
112/// Unexpected => something wrong happened
113///
114/// Context:
115///    path: /path/to/file
116///    called: send_async
117///
118/// Source: networking error
119///
120/// Backtrace:
121///    0: iceberg::error::Error::new
122///              at ./src/error.rs:197:24
123///    1: iceberg::error::tests::generate_error
124///              at ./src/error.rs:241:9
125///    2: iceberg::error::tests::test_error_debug_with_backtrace::{{closure}}
126///              at ./src/error.rs:305:41
127///    ...
128/// ```
129pub struct Error {
130    kind: ErrorKind,
131    message: String,
132
133    context: Vec<(&'static str, String)>,
134
135    source: Option<anyhow::Error>,
136    backtrace: Backtrace,
137}
138
139impl Display for Error {
140    fn fmt(&self, f: &mut Formatter<'_>) -> fmt::Result {
141        write!(f, "{}", self.kind)?;
142
143        if !self.context.is_empty() {
144            write!(f, ", context: {{ ")?;
145            write!(
146                f,
147                "{}",
148                self.context
149                    .iter()
150                    .map(|(k, v)| format!("{k}: {v}"))
151                    .collect::<Vec<_>>()
152                    .join(", ")
153            )?;
154            write!(f, " }}")?;
155        }
156
157        if !self.message.is_empty() {
158            write!(f, " => {}", self.message)?;
159        }
160
161        if let Some(source) = &self.source {
162            write!(f, ", source: {source}")?;
163        }
164
165        Ok(())
166    }
167}
168
169impl Debug for Error {
170    fn fmt(&self, f: &mut Formatter<'_>) -> fmt::Result {
171        // If alternate has been specified, we will print like Debug.
172        if f.alternate() {
173            let mut de = f.debug_struct("Error");
174            de.field("kind", &self.kind);
175            de.field("message", &self.message);
176            de.field("context", &self.context);
177            de.field("source", &self.source);
178            de.field("backtrace", &self.backtrace);
179            return de.finish();
180        }
181
182        write!(f, "{}", self.kind)?;
183        if !self.message.is_empty() {
184            write!(f, " => {}", self.message)?;
185        }
186        writeln!(f)?;
187
188        if !self.context.is_empty() {
189            writeln!(f)?;
190            writeln!(f, "Context:")?;
191            for (k, v) in self.context.iter() {
192                writeln!(f, "   {k}: {v}")?;
193            }
194        }
195        if let Some(source) = &self.source {
196            writeln!(f)?;
197            writeln!(f, "Source: {source:#}")?;
198        }
199
200        if self.backtrace.status() == BacktraceStatus::Captured {
201            writeln!(f)?;
202            writeln!(f, "Backtrace:")?;
203            writeln!(f, "{}", self.backtrace)?;
204        }
205
206        Ok(())
207    }
208}

210impl std::error::Error for Error {
211    fn source(&self) -> Option<&(dyn std::error::Error + 'static)> {
212        self.source.as_ref().map(|v| v.as_ref())
213    }
214}

216impl Error {
217    /// Create a new Error with error kind and message.
218    pub fn new(kind: ErrorKind, message: impl Into<String>) -> Self {
219        Self {
220            kind,
221            message: message.into(),
222            context: Vec::default(),
223
224            source: None,
225            // `Backtrace::capture()` will check if backtrace has been enabled
226            // internally. It's zero cost if backtrace is disabled.
227            backtrace: Backtrace::capture(),
228        }
229    }
230
231    /// Add more context in error.
232    pub fn with_context(mut self, key: &'static str, value: impl Into<String>) -> Self {
233        self.context.push((key, value.into()));
234        self
235    }
236
237    /// Set source for error.
238    ///
239    /// # Notes
240    ///
241    /// If the source has been set, we will raise a panic here.
242    pub fn with_source(mut self, src: impl Into<anyhow::Error>) -> Self {
243        debug_assert!(self.source.is_none(), "the source error has been set");
244
245        self.source = Some(src.into());
246        self
247    }
248
249    /// Set the backtrace for error.
250    ///
251    /// This function is served as testing purpose and not intended to be called
252    /// by users.
253    #[cfg(test)]
254    fn with_backtrace(mut self, backtrace: Backtrace) -> Self {
255        self.backtrace = backtrace;
256        self
257    }
258
259    /// Return error's kind.
260    ///
261    /// Users can use this method to check error's kind and take actions.
262    pub fn kind(&self) -> ErrorKind {
263        self.kind
264    }
265
266    /// Return error's message.
267    #[inline]
268    pub fn message(&self) -> &str {
269        self.message.as_str()
270    }
271}

369/// Converts a timestamp in milliseconds to `DateTime<Utc>`, handling errors.
370///
371/// # Arguments
372///
373/// * `timestamp_ms` - The timestamp in milliseconds to convert.
374///
375/// # Returns
376///
377/// This function returns a `Result<DateTime<Utc>, Error>` which is `Ok` with the `DateTime<Utc>` if the conversion is successful,
378/// or an `Err` with an appropriate error if the timestamp is ambiguous or invalid.
379pub(crate) fn timestamp_ms_to_utc(timestamp_ms: i64) -> Result<DateTime<Utc>> {
380    match Utc.timestamp_millis_opt(timestamp_ms) {
381        chrono::LocalResult::Single(t) => Ok(t),
382        chrono::LocalResult::Ambiguous(_, _) => Err(Error::new(
383            ErrorKind::Unexpected,
384            "Ambiguous timestamp with two possible results",
385        )),
386        chrono::LocalResult::None => Err(Error::new(ErrorKind::DataInvalid, "Invalid timestamp")),
387    }
388    .map_err(|e| e.with_context("timestamp value", timestamp_ms.to_string()))
389}

391/// Helper macro to check arguments.
392///
393///
394/// Example:
395///
396/// Following example check `a > 0`, otherwise returns an error.
397/// ```ignore
398/// use iceberg::check;
399/// ensure_data_valid!(a > 0, "{} is not positive.", a);
400/// ```
401#[macro_export]
402macro_rules! ensure_data_valid {
403    ($cond: expr, $fmt: literal, $($arg:tt)*) => {
404        if !$cond {
405            return Err($crate::error::Error::new($crate::error::ErrorKind::DataInvalid, format!($fmt, $($arg)*)))
406        }
407    };
408}
