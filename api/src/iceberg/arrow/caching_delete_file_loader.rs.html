103    /// Initiates loading of all deletes for all the specified tasks
104    ///
105    /// Returned future completes once all positional deletes and delete vectors
106    /// have loaded. EQ deletes are not waited for in this method but the returned
107    /// DeleteFilter will await their loading when queried for them.
108    ///
109    ///  * Create a single stream of all delete file tasks irrespective of type,
110    ///    so that we can respect the combined concurrency limit
111    ///  * We then process each in two phases: load and parse.
112    ///  * for positional deletes the load phase instantiates an ArrowRecordBatchStream to
113    ///    stream the file contents out
114    ///  * for eq deletes, we first check if the EQ delete is already loaded or being loaded by
115    ///    another concurrently processing data file scan task. If it is, we skip it.
116    ///    If not, the DeleteFilter is updated to contain a notifier to prevent other data file
117    ///    tasks from starting to load the same equality delete file. We spawn a task to load
118    ///    the EQ delete's record batch stream, convert it to a predicate, update the delete filter,
119    ///    and notify any task that was waiting for it.
120    ///  * When this gets updated to add support for delete vectors, the load phase will return
121    ///    a PuffinReader for them.
122    ///  * The parse phase parses each record batch stream according to its associated data type.
123    ///    The result of this is a map of data file paths to delete vectors for the positional
124    ///    delete tasks (and in future for the delete vector tasks). For equality delete
125    ///    file tasks, this results in an unbound Predicate.
126    ///  * The unbound Predicates resulting from equality deletes are sent to their associated oneshot
127    ///    channel to store them in the right place in the delete file managers state.
128    ///  * The results of all of these futures are awaited on in parallel with the specified
129    ///    level of concurrency and collected into a vec. We then combine all the delete
130    ///    vector maps that resulted from any positional delete or delete vector files into a
131    ///    single map and persist it in the state.
132    ///
133    ///
134    ///  Conceptually, the data flow is like this:
135    /// ```none
136    ///                                          FileScanTaskDeleteFile
137    ///                                                     |
138    ///                                             Skip Started EQ Deletes
139    ///                                                     |
140    ///                                                     |
141    ///                                       [load recordbatch stream / puffin]
142    ///                                             DeleteFileContext
143    ///                                                     |
144    ///                                                     |
145    ///                       +-----------------------------+--------------------------+
146    ///                     Pos Del           Del Vec (Not yet Implemented)         EQ Del
147    ///                       |                             |                          |
148    ///              [parse pos del stream]         [parse del vec puffin]       [parse eq del]
149    ///          HashMap<String, RoaringTreeMap> HashMap<String, RoaringTreeMap>   (Predicate, Sender)
150    ///                       |                             |                          |
151    ///                       |                             |                 [persist to state]
152    ///                       |                             |                          ()
153    ///                       |                             |                          |
154    ///                       +-----------------------------+--------------------------+
155    ///                                                     |
156    ///                                             [buffer unordered]
157    ///                                                     |
158    ///                                            [combine del vectors]
159    ///                                        HashMap<String, RoaringTreeMap>
160    ///                                                     |
161    ///                                        [persist del vectors to state]
162    ///                                                    ()
163    ///                                                    |
164    ///                                                    |
165    ///                                                 [join!]
166    /// ```
167    pub(crate) fn load_deletes(
168        &self,
169        delete_file_entries: &[FileScanTaskDeleteFile],
170        schema: SchemaRef,
171    ) -> Receiver<Result<DeleteFilter>> {
172        let (tx, rx) = channel();
173
174        let stream_items = delete_file_entries
175            .iter()
176            .map(|t| {
177                (
178                    t.clone(),
179                    self.basic_delete_file_loader.clone(),
180                    self.delete_filter.clone(),
181                    schema.clone(),
182                )
183            })
184            .collect::<Vec<_>>();
185        let task_stream = futures::stream::iter(stream_items);
186
187        let del_filter = self.delete_filter.clone();
188        let concurrency_limit_data_files = self.concurrency_limit_data_files;
189        let basic_delete_file_loader = self.basic_delete_file_loader.clone();
190        self.runtime.io().spawn(async move {
191            let result = async move {
192                let mut del_filter = del_filter;
193                let basic_delete_file_loader = basic_delete_file_loader.clone();
194
195                let mut results_stream = task_stream
196                    .map(move |(task, file_io, del_filter, schema)| {
197                        let basic_delete_file_loader = basic_delete_file_loader.clone();
198                        async move {
199                            Self::load_file_for_task(
200                                &task,
201                                basic_delete_file_loader.clone(),
202                                del_filter,
203                                schema,
204                            )
205                            .await
206                        }
207                    })
208                    .map(move |ctx| {
209                        Ok(async { Self::parse_file_content_for_task(ctx.await?).await })
210                    })
211                    .try_buffer_unordered(concurrency_limit_data_files);
212
213                while let Some(item) = results_stream.next().await {
214                    let item = item?;
215                    if let ParsedDeleteFileContext::DelVecs { file_path, results } = item {
216                        for (data_file_path, delete_vector) in results.into_iter() {
217                            del_filter.upsert_delete_vector(data_file_path, delete_vector);
218                        }
219                        // Mark the positional delete file as fully loaded so waiters can proceed
220                        del_filter.finish_pos_del_load(&file_path);
221                    }
222                }
223
224                Ok(del_filter)
225            }
226            .await;
227
228            let _ = tx.send(result);
229        });
230
231        rx
232    }

234    async fn load_file_for_task(
235        task: &FileScanTaskDeleteFile,
236        basic_delete_file_loader: BasicDeleteFileLoader,
237        del_filter: DeleteFilter,
238        schema: SchemaRef,
239    ) -> Result<DeleteFileContext> {
240        match task.file_type {
241            DataContentType::PositionDeletes => {
242                match del_filter.try_start_pos_del_load(&task.file_path) {
243                    PosDelLoadAction::AlreadyLoaded => Ok(DeleteFileContext::ExistingPosDel),
244                    PosDelLoadAction::WaitFor(notify) => {
245                        // Positional deletes are accessed synchronously by ArrowReader.
246                        // We must wait here to ensure the data is ready before returning,
247                        // otherwise ArrowReader might get an empty/partial result.
248                        notify.notified().await;
249                        Ok(DeleteFileContext::ExistingPosDel)
250                    }
251                    PosDelLoadAction::Load => Ok(DeleteFileContext::PosDels {
252                        file_path: task.file_path.clone(),
253                        stream: basic_delete_file_loader
254                            .parquet_to_batch_stream(
255                                &task.file_path,
256                                task.file_size_in_bytes,
257                                task.key_metadata.as_deref(),
258                            )
259                            .await?,
260                    }),
261                }
262            }
263
264            DataContentType::EqualityDeletes => {
265                let Some(notify) = del_filter.try_start_eq_del_load(&task.file_path) else {
266                    return Ok(DeleteFileContext::ExistingEqDel);
267                };
268
269                let (sender, receiver) = channel();
270                del_filter.insert_equality_delete(&task.file_path, receiver);
271
272                // Per the Iceberg spec, evolve schema for equality deletes but only for the
273                // equality_ids columns, not all table columns.
274                let equality_ids_vec = task.equality_ids.clone().unwrap();
275                let evolved_stream = BasicDeleteFileLoader::evolve_schema(
276                    basic_delete_file_loader
277                        .parquet_to_batch_stream(
278                            &task.file_path,
279                            task.file_size_in_bytes,
280                            task.key_metadata.as_deref(),
281                        )
282                        .await?,
283                    schema,
284                    &equality_ids_vec,
285                )
286                .await?;
287
288                Ok(DeleteFileContext::FreshEqDel {
289                    batch_stream: evolved_stream,
290                    sender,
291                    equality_ids: HashSet::from_iter(equality_ids_vec),
292                })
293            }
294
295            DataContentType::Data => Err(Error::new(
296                ErrorKind::Unexpected,
297                "tasks with files of type Data not expected here",
298            )),
299        }
300    }
301
302    async fn parse_file_content_for_task(
303        ctx: DeleteFileContext,
304    ) -> Result<ParsedDeleteFileContext> {
305        match ctx {
306            DeleteFileContext::ExistingEqDel => Ok(ParsedDeleteFileContext::EqDel),
307            DeleteFileContext::ExistingPosDel => Ok(ParsedDeleteFileContext::ExistingPosDel),
308            DeleteFileContext::PosDels { file_path, stream } => {
309                let del_vecs = Self::parse_positional_deletes_record_batch_stream(stream).await?;
310                Ok(ParsedDeleteFileContext::DelVecs {
311                    file_path,
312                    results: del_vecs,
313                })
314            }
315            DeleteFileContext::FreshEqDel {
316                sender,
317                batch_stream,
318                equality_ids,
319            } => {
320                let predicate =
321                    Self::parse_equality_deletes_record_batch_stream(batch_stream, equality_ids)
322                        .await?;
323
324                sender
325                    .send(predicate)
326                    .map_err(|err| {
327                        Error::new(
328                            ErrorKind::Unexpected,
329                            "Could not send eq delete predicate to state",
330                        )
331                    })
332                    .map(|_| ParsedDeleteFileContext::EqDel)
333            }
334        }
335    }

337    /// Parses a record batch stream coming from positional delete files
338    ///
339    /// Returns a map of data file path to a delete vector
340    async fn parse_positional_deletes_record_batch_stream(
341        mut stream: ArrowRecordBatchStream,
342    ) -> Result<HashMap<String, DeleteVector>> {
343        let mut result: HashMap<String, DeleteVector> = HashMap::default();
344
345        while let Some(batch) = stream.next().await {
346            let batch = batch?;
347            let schema = batch.schema();
348            let columns = batch.columns();
349
350            let Some(file_paths) = columns[0].as_any().downcast_ref::<StringArray>() else {
351                return Err(Error::new(
352                    ErrorKind::DataInvalid,
353                    "Could not downcast file paths array to StringArray",
354                ));
355            };
356            let Some(positions) = columns[1].as_any().downcast_ref::<Int64Array>() else {
357                return Err(Error::new(
358                    ErrorKind::DataInvalid,
359                    "Could not downcast positions array to Int64Array",
360                ));
361            };
362
363            for (file_path, pos) in file_paths.iter().zip(positions.iter()) {
364                let (Some(file_path), Some(pos)) = (file_path, pos) else {
365                    return Err(Error::new(
366                        ErrorKind::DataInvalid,
367                        "null values in delete file",
368                    ));
369                };
370                if pos < 0 {
371                    return Err(Error::new(
372                        ErrorKind::DataInvalid,
373                        format!("negative position in delete file {file_path}: {pos}"),
374                    ));
375                }
376
377                result
378                    .entry(file_path.to_string())
379                    .or_default()
380                    .insert(pos as u64);
381            }
382        }
383
384        Ok(result)
385    }

387    async fn parse_equality_deletes_record_batch_stream(
388        mut stream: ArrowRecordBatchStream,
389        equality_ids: HashSet<i32>,
390    ) -> Result<Predicate> {
391        let mut row_predicates = Vec::new();
392        let mut batch_schema_iceberg: Option<Schema> = None;
393        let accessor = EqDelRecordBatchPartnerAccessor;
394
395        while let Some(record_batch) = stream.next().await {
396            let record_batch = record_batch?;
397
398            if record_batch.num_columns() == 0 {
399                return Ok(AlwaysTrue);
400            }
401
402            let schema = match &batch_schema_iceberg {
403                Some(schema) => schema,
404                None => {
405                    let schema = arrow_schema_to_schema(record_batch.schema().as_ref())?;
406                    batch_schema_iceberg = Some(schema);
407                    batch_schema_iceberg.as_ref().unwrap()
408                }
409            };
410
411            let root_array: ArrayRef = Arc::new(StructArray::from(record_batch));
412
413            let mut processor = EqDelColumnProcessor::new(&equality_ids);
414            visit_schema_with_partner(schema, &root_array, &mut processor, &accessor)?;
415
416            let mut datum_columns_with_names = processor.finish()?;
417            if datum_columns_with_names.is_empty() {
418                continue;
419            }
420
421            // Iceberg spec (Equality Delete Files): a null data value never equals a non-null
422            // delete value, so a row with a null equality column must be kept. Build the keep
423            // predicate as `col IS NULL OR col != v` (`col IS NOT NULL` for a null delete value);
424            // a bare `col != v` drops nulls.
425            #[allow(clippy::len_zero)]
426            while datum_columns_with_names[0].0.len() > 0 {
427                let mut row_keep_predicate = Predicate::AlwaysFalse;
428                for &mut (ref mut column, ref field_name) in &mut datum_columns_with_names {
429                    if let Some(item) = column.next() {
430                        let reference = Reference::new(field_name.clone());
431                        let cell_keep_predicate = if let Some(datum) = item? {
432                            reference
433                                .clone()
434                                .is_null()
435                                .or(reference.not_equal_to(datum.clone()))
436                        } else {
437                            reference.is_not_null()
438                        };
439                        row_keep_predicate = row_keep_predicate.or(cell_keep_predicate);
440                    }
441                }
442                row_predicates.push(row_keep_predicate);
443            }
444        }
445
446        // All row predicates are combined to a single predicate by creating a balanced binary tree.
447        // Using a simple fold would result in a deeply nested predicate that can cause a stack overflow.
448        while row_predicates.len() > 1 {
449            let mut next_level = Vec::with_capacity(row_predicates.len().div_ceil(2));
450            let mut iter = row_predicates.into_iter();
451            while let Some(p1) = iter.next() {
452                if let Some(p2) = iter.next() {
453                    next_level.push(p1.and(p2));
454                } else {
455                    next_level.push(p1);
456                }
457            }
458            row_predicates = next_level;
459        }
460
461        match row_predicates.pop() {
462            Some(p) => Ok(p),
463            None => Ok(AlwaysTrue),
464        }
465    }
466}

468struct EqDelColumnProcessor<'a> {
469    equality_ids: &'a HashSet<i32>,
470    collected_columns: Vec<(ArrayRef, String, Type)>,
471}
472
473impl<'a> EqDelColumnProcessor<'a> {
474    fn new(equality_ids: &'a HashSet<i32>) -> Self {
475        Self {
476            equality_ids,
477            collected_columns: Vec::with_capacity(equality_ids.len()),
478        }
479    }
480
481    #[allow(clippy::type_complexity)]
482    fn finish(
483        self,
484    ) -> Result<
485        Vec<(
486            Box<dyn ExactSizeIterator<Item = Result<Option<Datum>>>>,
487            String,
488        )>,
489    > {
490        self.collected_columns
491            .into_iter()
492            .map(|(array, field_name, field_type)| {
493                let primitive_type = field_type
494                    .as_primitive_type()
495                    .ok_or_else(|| {
496                        Error::new(ErrorKind::Unexpected, "field is not a primitive type")
497                    })?
498                    .clone();
499
500                let lit_vec = arrow_primitive_to_literal(&array, &field_type)?;
501                let datum_iterator: Box<dyn ExactSizeIterator<Item = Result<Option<Datum>>>> =
502                    Box::new(lit_vec.into_iter().map(move |c| {
503                        c.map(|literal| {
504                            literal
505                                .as_primitive_literal()
506                                .map(|primitive_literal| {
507                                    Datum::new(primitive_type.clone(), primitive_literal)
508                                })
509                                .ok_or(Error::new(
510                                    ErrorKind::Unexpected,
511                                    "failed to convert to primitive literal",
512                                ))
513                        })
514                        .transpose()
515                    }));
516
517                Ok((datum_iterator, field_name))
518            })
519            .collect::<Result<Vec<_>>>()
520    }
521}

581        &self,
582        struct_partner: &'a ArrayRef,
583        field: &NestedField,
584    ) -> Result<&'a ArrayRef> {
585        let Some(struct_array) = struct_partner.as_any().downcast_ref::<StructArray>() else {
586            return Err(Error::new(
587                ErrorKind::Unexpected,
588                "Expected struct array for field extraction",
589            ));
590        };
591
592        // Find the field by name within the struct
593        for (i, field_def) in struct_array.fields().iter().enumerate() {
594            if field_def.name() == &field.name {
595                return Ok(struct_array.column(i));
596            }
597        }
598
599        Err(Error::new(
600            ErrorKind::Unexpected,
601            format!("Field {} not found in parent struct", field.name),
602        ))
603    }
