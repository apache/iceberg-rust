79    pub(crate) fn new(file_io: FileIO, concurrency_limit_data_files: usize) -> Self {
80        CachingDeleteFileLoader {
81            basic_delete_file_loader: BasicDeleteFileLoader::new(file_io),
82            concurrency_limit_data_files,
83            delete_filter: DeleteFilter::default(),
84        }

87    /// Initiates loading of all deletes for all the specified tasks
88    ///
89    /// Returned future completes once all positional deletes and delete vectors
90    /// have loaded. EQ deletes are not waited for in this method but the returned
91    /// DeleteFilter will await their loading when queried for them.
92    ///
93    ///  * Create a single stream of all delete file tasks irrespective of type,
94    ///    so that we can respect the combined concurrency limit
95    ///  * We then process each in two phases: load and parse.
96    ///  * for positional deletes the load phase instantiates an ArrowRecordBatchStream to
97    ///    stream the file contents out
98    ///  * for eq deletes, we first check if the EQ delete is already loaded or being loaded by
99    ///    another concurrently processing data file scan task. If it is, we skip it.
100    ///    If not, the DeleteFilter is updated to contain a notifier to prevent other data file
101    ///    tasks from starting to load the same equality delete file. We spawn a task to load
102    ///    the EQ delete's record batch stream, convert it to a predicate, update the delete filter,
103    ///    and notify any task that was waiting for it.
104    ///  * When this gets updated to add support for delete vectors, the load phase will return
105    ///    a PuffinReader for them.
106    ///  * The parse phase parses each record batch stream according to its associated data type.
107    ///    The result of this is a map of data file paths to delete vectors for the positional
108    ///    delete tasks (and in future for the delete vector tasks). For equality delete
109    ///    file tasks, this results in an unbound Predicate.
110    ///  * The unbound Predicates resulting from equality deletes are sent to their associated oneshot
111    ///    channel to store them in the right place in the delete file managers state.
112    ///  * The results of all of these futures are awaited on in parallel with the specified
113    ///    level of concurrency and collected into a vec. We then combine all the delete
114    ///    vector maps that resulted from any positional delete or delete vector files into a
115    ///    single map and persist it in the state.
116    ///
117    ///
118    ///  Conceptually, the data flow is like this:
119    /// ```none
120    ///                                          FileScanTaskDeleteFile
121    ///                                                     |
122    ///                                             Skip Started EQ Deletes
123    ///                                                     |
124    ///                                                     |
125    ///                                       [load recordbatch stream / puffin]
126    ///                                             DeleteFileContext
127    ///                                                     |
128    ///                                                     |
129    ///                       +-----------------------------+--------------------------+
130    ///                     Pos Del           Del Vec (Not yet Implemented)         EQ Del
131    ///                       |                             |                          |
132    ///              [parse pos del stream]         [parse del vec puffin]       [parse eq del]
133    ///          HashMap<String, RoaringTreeMap> HashMap<String, RoaringTreeMap>   (Predicate, Sender)
134    ///                       |                             |                          |
135    ///                       |                             |                 [persist to state]
136    ///                       |                             |                          ()
137    ///                       |                             |                          |
138    ///                       +-----------------------------+--------------------------+
139    ///                                                     |
140    ///                                             [buffer unordered]
141    ///                                                     |
142    ///                                            [combine del vectors]
143    ///                                        HashMap<String, RoaringTreeMap>
144    ///                                                     |
145    ///                                        [persist del vectors to state]
146    ///                                                    ()
147    ///                                                    |
148    ///                                                    |
149    ///                                                 [join!]
150    /// ```
151    pub(crate) fn load_deletes(
152        &self,
153        delete_file_entries: &[FileScanTaskDeleteFile],
154        schema: SchemaRef,
155    ) -> Receiver<Result<DeleteFilter>> {
156        let (tx, rx) = channel();
157
158        let stream_items = delete_file_entries
159            .iter()
160            .map(|t| {
161                (
162                    t.clone(),
163                    self.basic_delete_file_loader.clone(),
164                    self.delete_filter.clone(),
165                    schema.clone(),
166                )
167            })
168            .collect::<Vec<_>>();
169        let task_stream = futures::stream::iter(stream_items);
170
171        let del_filter = self.delete_filter.clone();
172        let concurrency_limit_data_files = self.concurrency_limit_data_files;
173        let basic_delete_file_loader = self.basic_delete_file_loader.clone();
174        crate::runtime::spawn(async move {
175            let result = async move {
176                let mut del_filter = del_filter;
177                let basic_delete_file_loader = basic_delete_file_loader.clone();
178
179                let mut results_stream = task_stream
180                    .map(move |(task, file_io, del_filter, schema)| {
181                        let basic_delete_file_loader = basic_delete_file_loader.clone();
182                        async move {
183                            Self::load_file_for_task(
184                                &task,
185                                basic_delete_file_loader.clone(),
186                                del_filter,
187                                schema,
188                            )
189                            .await
190                        }
191                    })
192                    .map(move |ctx| {
193                        Ok(async { Self::parse_file_content_for_task(ctx.await?).await })
194                    })
195                    .try_buffer_unordered(concurrency_limit_data_files);
196
197                while let Some(item) = results_stream.next().await {
198                    let item = item?;
199                    if let ParsedDeleteFileContext::DelVecs { file_path, results } = item {
200                        for (data_file_path, delete_vector) in results.into_iter() {
201                            del_filter.upsert_delete_vector(data_file_path, delete_vector);
202                        }
203                        // Mark the positional delete file as fully loaded so waiters can proceed
204                        del_filter.finish_pos_del_load(&file_path);
205                    }
206                }
207
208                Ok(del_filter)
209            }
210            .await;
211
212            let _ = tx.send(result);
213        });
214
215        rx
216    }

218    async fn load_file_for_task(
219        task: &FileScanTaskDeleteFile,
220        basic_delete_file_loader: BasicDeleteFileLoader,
221        del_filter: DeleteFilter,
222        schema: SchemaRef,
223    ) -> Result<DeleteFileContext> {
224        match task.file_type {
225            DataContentType::PositionDeletes => {
226                match del_filter.try_start_pos_del_load(&task.file_path) {
227                    PosDelLoadAction::AlreadyLoaded => Ok(DeleteFileContext::ExistingPosDel),
228                    PosDelLoadAction::WaitFor(notify) => {
229                        // Positional deletes are accessed synchronously by ArrowReader.
230                        // We must wait here to ensure the data is ready before returning,
231                        // otherwise ArrowReader might get an empty/partial result.
232                        notify.notified().await;
233                        Ok(DeleteFileContext::ExistingPosDel)
234                    }
235                    PosDelLoadAction::Load => Ok(DeleteFileContext::PosDels {
236                        file_path: task.file_path.clone(),
237                        stream: basic_delete_file_loader
238                            .parquet_to_batch_stream(&task.file_path)
239                            .await?,
240                    }),
241                }
242            }
243
244            DataContentType::EqualityDeletes => {
245                let Some(notify) = del_filter.try_start_eq_del_load(&task.file_path) else {
246                    return Ok(DeleteFileContext::ExistingEqDel);
247                };
248
249                let (sender, receiver) = channel();
250                del_filter.insert_equality_delete(&task.file_path, receiver);
251
252                // Per the Iceberg spec, evolve schema for equality deletes but only for the
253                // equality_ids columns, not all table columns.
254                let equality_ids_vec = task.equality_ids.clone().unwrap();
255                let evolved_stream = BasicDeleteFileLoader::evolve_schema(
256                    basic_delete_file_loader
257                        .parquet_to_batch_stream(&task.file_path)
258                        .await?,
259                    schema,
260                    &equality_ids_vec,
261                )
262                .await?;
263
264                Ok(DeleteFileContext::FreshEqDel {
265                    batch_stream: evolved_stream,
266                    sender,
267                    equality_ids: HashSet::from_iter(equality_ids_vec),
268                })
269            }
270
271            DataContentType::Data => Err(Error::new(
272                ErrorKind::Unexpected,
273                "tasks with files of type Data not expected here",
274            )),
275        }
276    }
277
278    async fn parse_file_content_for_task(
279        ctx: DeleteFileContext,
280    ) -> Result<ParsedDeleteFileContext> {
281        match ctx {
282            DeleteFileContext::ExistingEqDel => Ok(ParsedDeleteFileContext::EqDel),
283            DeleteFileContext::ExistingPosDel => Ok(ParsedDeleteFileContext::ExistingPosDel),
284            DeleteFileContext::PosDels { file_path, stream } => {
285                let del_vecs = Self::parse_positional_deletes_record_batch_stream(stream).await?;
286                Ok(ParsedDeleteFileContext::DelVecs {
287                    file_path,
288                    results: del_vecs,
289                })
290            }
291            DeleteFileContext::FreshEqDel {
292                sender,
293                batch_stream,
294                equality_ids,
295            } => {
296                let predicate =
297                    Self::parse_equality_deletes_record_batch_stream(batch_stream, equality_ids)
298                        .await?;
299
300                sender
301                    .send(predicate)
302                    .map_err(|err| {
303                        Error::new(
304                            ErrorKind::Unexpected,
305                            "Could not send eq delete predicate to state",
306                        )
307                    })
308                    .map(|_| ParsedDeleteFileContext::EqDel)
309            }
310        }
311    }

313    /// Parses a record batch stream coming from positional delete files
314    ///
315    /// Returns a map of data file path to a delete vector
316    async fn parse_positional_deletes_record_batch_stream(
317        mut stream: ArrowRecordBatchStream,
318    ) -> Result<HashMap<String, DeleteVector>> {
319        let mut result: HashMap<String, DeleteVector> = HashMap::default();
320
321        while let Some(batch) = stream.next().await {
322            let batch = batch?;
323            let schema = batch.schema();
324            let columns = batch.columns();
325
326            let Some(file_paths) = columns[0].as_any().downcast_ref::<StringArray>() else {
327                return Err(Error::new(
328                    ErrorKind::DataInvalid,
329                    "Could not downcast file paths array to StringArray",
330                ));
331            };
332            let Some(positions) = columns[1].as_any().downcast_ref::<Int64Array>() else {
333                return Err(Error::new(
334                    ErrorKind::DataInvalid,
335                    "Could not downcast positions array to Int64Array",
336                ));
337            };
338
339            for (file_path, pos) in file_paths.iter().zip(positions.iter()) {
340                let (Some(file_path), Some(pos)) = (file_path, pos) else {
341                    return Err(Error::new(
342                        ErrorKind::DataInvalid,
343                        "null values in delete file",
344                    ));
345                };
346
347                result
348                    .entry(file_path.to_string())
349                    .or_default()
350                    .insert(pos as u64);
351            }
352        }
353
354        Ok(result)
355    }

357    async fn parse_equality_deletes_record_batch_stream(
358        mut stream: ArrowRecordBatchStream,
359        equality_ids: HashSet<i32>,
360    ) -> Result<Predicate> {
361        let mut row_predicates = Vec::new();
362        let mut batch_schema_iceberg: Option<Schema> = None;
363        let accessor = EqDelRecordBatchPartnerAccessor;
364
365        while let Some(record_batch) = stream.next().await {
366            let record_batch = record_batch?;
367
368            if record_batch.num_columns() == 0 {
369                return Ok(AlwaysTrue);
370            }
371
372            let schema = match &batch_schema_iceberg {
373                Some(schema) => schema,
374                None => {
375                    let schema = arrow_schema_to_schema(record_batch.schema().as_ref())?;
376                    batch_schema_iceberg = Some(schema);
377                    batch_schema_iceberg.as_ref().unwrap()
378                }
379            };
380
381            let root_array: ArrayRef = Arc::new(StructArray::from(record_batch));
382
383            let mut processor = EqDelColumnProcessor::new(&equality_ids);
384            visit_schema_with_partner(schema, &root_array, &mut processor, &accessor)?;
385
386            let mut datum_columns_with_names = processor.finish()?;
387            if datum_columns_with_names.is_empty() {
388                continue;
389            }
390
391            // Process the collected columns in lockstep
392            #[allow(clippy::len_zero)]
393            while datum_columns_with_names[0].0.len() > 0 {
394                let mut row_predicate = AlwaysTrue;
395                for &mut (ref mut column, ref field_name) in &mut datum_columns_with_names {
396                    if let Some(item) = column.next() {
397                        let cell_predicate = if let Some(datum) = item? {
398                            Reference::new(field_name.clone()).equal_to(datum.clone())
399                        } else {
400                            Reference::new(field_name.clone()).is_null()
401                        };
402                        row_predicate = row_predicate.and(cell_predicate)
403                    }
404                }
405                row_predicates.push(row_predicate.not().rewrite_not());
406            }
407        }
408
409        // All row predicates are combined to a single predicate by creating a balanced binary tree.
410        // Using a simple fold would result in a deeply nested predicate that can cause a stack overflow.
411        while row_predicates.len() > 1 {
412            let mut next_level = Vec::with_capacity(row_predicates.len().div_ceil(2));
413            let mut iter = row_predicates.into_iter();
414            while let Some(p1) = iter.next() {
415                if let Some(p2) = iter.next() {
416                    next_level.push(p1.and(p2));
417                } else {
418                    next_level.push(p1);
419                }
420            }
421            row_predicates = next_level;
422        }
423
424        match row_predicates.pop() {
425            Some(p) => Ok(p),
426            None => Ok(AlwaysTrue),
427        }
428    }

431struct EqDelColumnProcessor<'a> {
432    equality_ids: &'a HashSet<i32>,
433    collected_columns: Vec<(ArrayRef, String, Type)>,
434}
435
436impl<'a> EqDelColumnProcessor<'a> {
437    fn new(equality_ids: &'a HashSet<i32>) -> Self {
438        Self {
439            equality_ids,
440            collected_columns: Vec::with_capacity(equality_ids.len()),
441        }
442    }
443
444    #[allow(clippy::type_complexity)]
445    fn finish(
446        self,
447    ) -> Result<
448        Vec<(
449            Box<dyn ExactSizeIterator<Item = Result<Option<Datum>>>>,
450            String,
451        )>,
452    > {
453        self.collected_columns
454            .into_iter()
455            .map(|(array, field_name, field_type)| {
456                let primitive_type = field_type
457                    .as_primitive_type()
458                    .ok_or_else(|| {
459                        Error::new(ErrorKind::Unexpected, "field is not a primitive type")
460                    })?
461                    .clone();
462
463                let lit_vec = arrow_primitive_to_literal(&array, &field_type)?;
464                let datum_iterator: Box<dyn ExactSizeIterator<Item = Result<Option<Datum>>>> =
465                    Box::new(lit_vec.into_iter().map(move |c| {
466                        c.map(|literal| {
467                            literal
468                                .as_primitive_literal()
469                                .map(|primitive_literal| {
470                                    Datum::new(primitive_type.clone(), primitive_literal)
471                                })
472                                .ok_or(Error::new(
473                                    ErrorKind::Unexpected,
474                                    "failed to convert to primitive literal",
475                                ))
476                        })
477                        .transpose()
478                    }));
479
480                Ok((datum_iterator, field_name))
481            })
482            .collect::<Result<Vec<_>>>()
483    }
484}

534impl PartnerAccessor<ArrayRef> for EqDelRecordBatchPartnerAccessor {
535    fn struct_partner<'a>(&self, schema_partner: &'a ArrayRef) -> Result<&'a ArrayRef> {
536        Ok(schema_partner)
537    }
538
539    fn field_partner<'a>(
540        &self,
541        struct_partner: &'a ArrayRef,
542        field: &NestedField,
543    ) -> Result<&'a ArrayRef> {
544        let Some(struct_array) = struct_partner.as_any().downcast_ref::<StructArray>() else {
545            return Err(Error::new(
546                ErrorKind::Unexpected,
547                "Expected struct array for field extraction",
548            ));
549        };
550
551        // Find the field by name within the struct
552        for (i, field_def) in struct_array.fields().iter().enumerate() {
553            if field_def.name() == &field.name {
554                return Ok(struct_array.column(i));
555            }
556        }
557
558        Err(Error::new(
559            ErrorKind::Unexpected,
560            format!("Field {} not found in parent struct", field.name),
561        ))
562    }
563
564    fn list_element_partner<'a>(&self, _list_partner: &'a ArrayRef) -> Result<&'a ArrayRef> {
565        Err(Error::new(
566            ErrorKind::FeatureUnsupported,
567            "List columns are unsupported in equality deletes",
568        ))
569    }
570
571    fn map_key_partner<'a>(&self, _map_partner: &'a ArrayRef) -> Result<&'a ArrayRef> {
572        Err(Error::new(
573            ErrorKind::FeatureUnsupported,
574            "Map columns are unsupported in equality deletes",
575        ))
576    }
577
578    fn map_value_partner<'a>(&self, _map_partner: &'a ArrayRef) -> Result<&'a ArrayRef> {
579        Err(Error::new(
580            ErrorKind::FeatureUnsupported,
581            "Map columns are unsupported in equality deletes",
582        ))
583    }

752    /// Verifies that evolve_schema on partial-schema equality deletes works correctly
753    /// when only equality_ids columns are evolved, not all table columns.
754    ///
755    /// Per the [Iceberg spec](https://iceberg.apache.org/spec/#equality-delete-files),
756    /// equality delete files can contain only a subset of columns.
757    #[tokio::test]
758    async fn test_partial_schema_equality_deletes_evolve_succeeds() {
759        let tmp_dir = TempDir::new().unwrap();
760        let table_location = tmp_dir.path().as_os_str().to_str().unwrap();
761
762        // Create table schema with REQUIRED fields
763        let table_schema = Arc::new(
764            Schema::builder()
765                .with_schema_id(1)
766                .with_fields(vec![
767                    crate::spec::NestedField::required(
768                        1,
769                        "id",
770                        crate::spec::Type::Primitive(crate::spec::PrimitiveType::Int),
771                    )
772                    .into(),
773                    crate::spec::NestedField::required(
774                        2,
775                        "data",
776                        crate::spec::Type::Primitive(crate::spec::PrimitiveType::String),
777                    )
778                    .into(),
779                ])
780                .build()
781                .unwrap(),
782        );
783
784        // Write equality delete file with PARTIAL schema (only 'data' column)
785        let delete_file_path = {
786            let data_vals = vec!["a", "d", "g"];
787            let data_col = Arc::new(StringArray::from(data_vals)) as ArrayRef;
788
789            let delete_schema = Arc::new(arrow_schema::Schema::new(vec![simple_field(
790                "data",
791                DataType::Utf8,
792                false,
793                "2", // field ID
794            )]));
795
796            let delete_batch = RecordBatch::try_new(delete_schema.clone(), vec![data_col]).unwrap();
797
798            let path = format!("{}/partial-eq-deletes.parquet", &table_location);
799            let file = File::create(&path).unwrap();
800            let props = WriterProperties::builder()
801                .set_compression(Compression::SNAPPY)
802                .build();
803            let mut writer =
804                ArrowWriter::try_new(file, delete_batch.schema(), Some(props)).unwrap();
805            writer.write(&delete_batch).expect("Writing batch");
806            writer.close().unwrap();
807            path
808        };
809
810        let file_io = FileIO::from_path(table_location).unwrap().build().unwrap();
811        let basic_delete_file_loader = BasicDeleteFileLoader::new(file_io.clone());
812
813        let batch_stream = basic_delete_file_loader
814            .parquet_to_batch_stream(&delete_file_path)
815            .await
816            .unwrap();
817
818        // Only evolve the equality_ids columns (field 2), not all table columns
819        let equality_ids = vec![2];
820        let evolved_stream =
821            BasicDeleteFileLoader::evolve_schema(batch_stream, table_schema, &equality_ids)
822                .await
823                .unwrap();
824
825        let result = evolved_stream.try_collect::<Vec<_>>().await;
826
827        assert!(
828            result.is_ok(),
829            "Expected success when evolving only equality_ids columns, got error: {:?}",
830            result.err()
831        );
832
833        let batches = result.unwrap();
834        assert_eq!(batches.len(), 1);
835
836        let batch = &batches[0];
837        assert_eq!(batch.num_rows(), 3);
838        assert_eq!(batch.num_columns(), 1); // Only 'data' column
839
840        // Verify the actual values are preserved after schema evolution
841        let data_col = batch.column(0).as_string::<i32>();
842        assert_eq!(data_col.value(0), "a");
843        assert_eq!(data_col.value(1), "d");
844        assert_eq!(data_col.value(2), "g");
845    }
