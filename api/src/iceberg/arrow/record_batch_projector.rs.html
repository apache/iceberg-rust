27/// Help to project specific field from `RecordBatch`` according to the fields id.
28#[derive(Clone, Debug)]
29pub(crate) struct RecordBatchProjector {
30    // A vector of vectors, where each inner vector represents the index path to access a specific field in a nested structure.
31    // E.g. [[0], [1, 2]] means the first field is accessed directly from the first column,
32    // while the second field is accessed from the second column and then from its third subcolumn (second column must be a struct column).
33    field_indices: Vec<Vec<usize>>,
34    // The schema reference after projection. This schema is derived from the original schema based on the given field IDs.
35    projected_schema: SchemaRef,
36}
37
38impl RecordBatchProjector {
39    /// Init ArrowFieldProjector
40    ///
41    /// This function will iterate through the field and fetch the field from the original schema according to the field ids.
42    /// The function to fetch the field id from the field is provided by `field_id_fetch_func`, return None if the field need to be skipped.
43    /// This function will iterate through the nested fields if the field is a struct, `searchable_field_func` can be used to control whether
44    /// iterate into the nested fields.
45    pub(crate) fn new<F1, F2>(
46        original_schema: SchemaRef,
47        field_ids: &[i32],
48        field_id_fetch_func: F1,
49        searchable_field_func: F2,
50    ) -> Result<Self>
51    where
52        F1: Fn(&Field) -> Result<Option<i64>>,
53        F2: Fn(&Field) -> bool,
54    {
55        let mut field_indices = Vec::with_capacity(field_ids.len());
56        let mut fields = Vec::with_capacity(field_ids.len());
57        for &id in field_ids {
58            let mut field_index = vec![];
59            let field = Self::fetch_field_index(
60                original_schema.fields(),
61                &mut field_index,
62                id as i64,
63                &field_id_fetch_func,
64                &searchable_field_func,
65            )?
66            .ok_or_else(|| {
67                Error::new(ErrorKind::Unexpected, "Field not found")
68                    .with_context("field_id", id.to_string())
69            })?;
70            fields.push(field.clone());
71            field_indices.push(field_index);
72        }
73        let delete_arrow_schema = Arc::new(Schema::new(fields));
74        Ok(Self {
75            field_indices,
76            projected_schema: delete_arrow_schema,
77        })
78    }
79
80    fn fetch_field_index<F1, F2>(
81        fields: &Fields,
82        index_vec: &mut Vec<usize>,
83        target_field_id: i64,
84        field_id_fetch_func: &F1,
85        searchable_field_func: &F2,
86    ) -> Result<Option<FieldRef>>
87    where
88        F1: Fn(&Field) -> Result<Option<i64>>,
89        F2: Fn(&Field) -> bool,
90    {
91        for (pos, field) in fields.iter().enumerate() {
92            let id = field_id_fetch_func(field)?;
93            if let Some(id) = id {
94                if target_field_id == id {
95                    index_vec.push(pos);
96                    return Ok(Some(field.clone()));
97                }
98            }
99            if let DataType::Struct(inner) = field.data_type() {
100                if searchable_field_func(field) {
101                    if let Some(res) = Self::fetch_field_index(
102                        inner,
103                        index_vec,
104                        target_field_id,
105                        field_id_fetch_func,
106                        searchable_field_func,
107                    )? {
108                        index_vec.push(pos);
109                        return Ok(Some(res));
110                    }
111                }
112            }
113        }
114        Ok(None)
115    }
116
117    /// Return the reference of projected schema
118    pub(crate) fn projected_schema_ref(&self) -> &SchemaRef {
119        &self.projected_schema
120    }
121
122    /// Do projection with record batch
123    pub(crate) fn project_batch(&self, batch: RecordBatch) -> Result<RecordBatch> {
124        RecordBatch::try_new(
125            self.projected_schema.clone(),
126            self.project_column(batch.columns())?,
127        )
128        .map_err(|err| Error::new(ErrorKind::DataInvalid, format!("{err}")))
129    }
130
131    /// Do projection with columns
132    pub(crate) fn project_column(&self, batch: &[ArrayRef]) -> Result<Vec<ArrayRef>> {
133        self.field_indices
134            .iter()
135            .map(|index_vec| Self::get_column_by_field_index(batch, index_vec))
136            .collect::<Result<Vec<_>>>()
137    }
138
139    fn get_column_by_field_index(batch: &[ArrayRef], field_index: &[usize]) -> Result<ArrayRef> {
140        let mut rev_iterator = field_index.iter().rev();
141        let mut array = batch[*rev_iterator.next().unwrap()].clone();
142        let mut null_buffer = array.logical_nulls();
143        for idx in rev_iterator {
144            array = array
145                .as_any()
146                .downcast_ref::<StructArray>()
147                .ok_or(Error::new(
148                    ErrorKind::Unexpected,
149                    "Cannot convert Array to StructArray",
150                ))?
151                .column(*idx)
152                .clone();
153            null_buffer = NullBuffer::union(null_buffer.as_ref(), array.logical_nulls().as_ref());
154        }
155        Ok(make_array(
156            array.to_data().into_builder().nulls(null_buffer).build()?,
157        ))
158    }
159}

161#[cfg(test)]
162mod test {
163    use std::sync::Arc;
164
165    use arrow_array::{ArrayRef, Int32Array, RecordBatch, StringArray, StructArray};
166    use arrow_schema::{DataType, Field, Fields, Schema};
167
168    use crate::arrow::record_batch_projector::RecordBatchProjector;
169    use crate::{Error, ErrorKind};
170
171    #[test]
172    fn test_record_batch_projector_nested_level() {
173        let inner_fields = vec![
174            Field::new("inner_field1", DataType::Int32, false),
175            Field::new("inner_field2", DataType::Utf8, false),
176        ];
177        let fields = vec![
178            Field::new("field1", DataType::Int32, false),
179            Field::new(
180                "field2",
181                DataType::Struct(Fields::from(inner_fields.clone())),
182                false,
183            ),
184        ];
185        let schema = Arc::new(Schema::new(fields));
186
187        let field_id_fetch_func = |field: &Field| match field.name().as_str() {
188            "field1" => Ok(Some(1)),
189            "field2" => Ok(Some(2)),
190            "inner_field1" => Ok(Some(3)),
191            "inner_field2" => Ok(Some(4)),
192            _ => Err(Error::new(ErrorKind::Unexpected, "Field id not found")),
193        };
194        let projector =
195            RecordBatchProjector::new(schema.clone(), &[1, 3], field_id_fetch_func, |_| true)
196                .unwrap();
197
198        assert_eq!(projector.field_indices.len(), 2);
199        assert_eq!(projector.field_indices[0], vec![0]);
200        assert_eq!(projector.field_indices[1], vec![0, 1]);
201
202        let int_array = Arc::new(Int32Array::from(vec![1, 2, 3])) as ArrayRef;
203        let inner_int_array = Arc::new(Int32Array::from(vec![4, 5, 6])) as ArrayRef;
204        let inner_string_array = Arc::new(StringArray::from(vec!["x", "y", "z"])) as ArrayRef;
205        let struct_array = Arc::new(StructArray::from(vec![
206            (
207                Arc::new(inner_fields[0].clone()),
208                inner_int_array as ArrayRef,
209            ),
210            (
211                Arc::new(inner_fields[1].clone()),
212                inner_string_array as ArrayRef,
213            ),
214        ])) as ArrayRef;
215        let batch = RecordBatch::try_new(schema, vec![int_array, struct_array]).unwrap();
216
217        let projected_batch = projector.project_batch(batch).unwrap();
218        assert_eq!(projected_batch.num_columns(), 2);
219        let projected_int_array = projected_batch
220            .column(0)
221            .as_any()
222            .downcast_ref::<Int32Array>()
223            .unwrap();
224        let projected_inner_int_array = projected_batch
225            .column(1)
226            .as_any()
227            .downcast_ref::<Int32Array>()
228            .unwrap();
229
230        assert_eq!(projected_int_array.values(), &[1, 2, 3]);
231        assert_eq!(projected_inner_int_array.values(), &[4, 5, 6]);
232    }
