31#[derive(Debug)]
32enum EqDelState {
33    Loading(Arc<Notify>),
34    Loaded(Predicate),
35}
36
37#[derive(Debug, Default)]
38struct DeleteFileFilterState {
39    delete_vectors: HashMap<String, Arc<Mutex<DeleteVector>>>,
40    equality_deletes: HashMap<String, EqDelState>,
41}
42
43#[derive(Clone, Debug, Default)]
44pub(crate) struct DeleteFilter {
45    state: Arc<RwLock<DeleteFileFilterState>>,
46}
47
48impl DeleteFilter {
49    /// Retrieve a delete vector for the data file associated with a given file scan task
50    pub(crate) fn get_delete_vector(
51        &self,
52        file_scan_task: &FileScanTask,
53    ) -> Option<Arc<Mutex<DeleteVector>>> {
54        self.get_delete_vector_for_path(file_scan_task.data_file_path())
55    }
56
57    /// Retrieve a delete vector for a data file
58    pub(crate) fn get_delete_vector_for_path(
59        &self,
60        delete_file_path: &str,
61    ) -> Option<Arc<Mutex<DeleteVector>>> {
62        self.state
63            .read()
64            .ok()
65            .and_then(|st| st.delete_vectors.get(delete_file_path).cloned())
66    }
67
68    pub(crate) fn try_start_eq_del_load(&self, file_path: &str) -> Option<Arc<Notify>> {
69        let mut state = self.state.write().unwrap();
70
71        if !state.equality_deletes.contains_key(file_path) {
72            return None;
73        }
74
75        let notifier = Arc::new(Notify::new());
76        state
77            .equality_deletes
78            .insert(file_path.to_string(), EqDelState::Loading(notifier.clone()));
79
80        Some(notifier)
81    }
82
83    /// Retrieve the equality delete predicate for a given eq delete file path
84    pub(crate) async fn get_equality_delete_predicate_for_delete_file_path(
85        &self,
86        file_path: &str,
87    ) -> Option<Predicate> {
88        let notifier = {
89            match self.state.read().unwrap().equality_deletes.get(file_path) {
90                None => return None,
91                Some(EqDelState::Loading(notifier)) => notifier.clone(),
92                Some(EqDelState::Loaded(predicate)) => {
93                    return Some(predicate.clone());
94                }
95            }
96        };
97
98        notifier.notified().await;
99
100        match self.state.read().unwrap().equality_deletes.get(file_path) {
101            Some(EqDelState::Loaded(predicate)) => Some(predicate.clone()),
102            _ => unreachable!("Cannot be any other state than loaded"),
103        }
104    }
105
106    /// Builds eq delete predicate for the provided task.
107    pub(crate) async fn build_equality_delete_predicate(
108        &self,
109        file_scan_task: &FileScanTask,
110    ) -> Result<Option<BoundPredicate>> {
111        // * Filter the task's deletes into just the Equality deletes
112        // * Retrieve the unbound predicate for each from self.state.equality_deletes
113        // * Logical-AND them all together to get a single combined `Predicate`
114        // * Bind the predicate to the task's schema to get a `BoundPredicate`
115
116        let mut combined_predicate = AlwaysTrue;
117        for delete in &file_scan_task.deletes {
118            if !is_equality_delete(delete) {
119                continue;
120            }
121
122            let Some(predicate) = self
123                .get_equality_delete_predicate_for_delete_file_path(&delete.file_path)
124                .await
125            else {
126                return Err(Error::new(
127                    ErrorKind::Unexpected,
128                    format!(
129                        "Missing predicate for equality delete file '{}'",
130                        delete.file_path
131                    ),
132                ));
133            };
134
135            combined_predicate = combined_predicate.and(predicate);
136        }
137
138        if combined_predicate == AlwaysTrue {
139            return Ok(None);
140        }
141
142        // TODO: handle case-insensitive case
143        let bound_predicate = combined_predicate.bind(file_scan_task.schema.clone(), false)?;
144        Ok(Some(bound_predicate))
145    }
146
147    pub(crate) fn upsert_delete_vector(
148        &mut self,
149        data_file_path: String,
150        delete_vector: DeleteVector,
151    ) {
152        let mut state = self.state.write().unwrap();
153
154        let Some(entry) = state.delete_vectors.get_mut(&data_file_path) else {
155            state
156                .delete_vectors
157                .insert(data_file_path, Arc::new(Mutex::new(delete_vector)));
158            return;
159        };
160
161        *entry.lock().unwrap() |= delete_vector;
162    }
163
164    pub(crate) fn insert_equality_delete(
165        &self,
166        delete_file_path: &str,
167        eq_del: Receiver<Predicate>,
168    ) {
169        let notify = Arc::new(Notify::new());
170        {
171            let mut state = self.state.write().unwrap();
172            state.equality_deletes.insert(
173                delete_file_path.to_string(),
174                EqDelState::Loading(notify.clone()),
175            );
176        }
177
178        let state = self.state.clone();
179        let delete_file_path = delete_file_path.to_string();
180        crate::runtime::spawn(async move {
181            let eq_del = eq_del.await.unwrap();
182            {
183                let mut state = state.write().unwrap();
184                state
185                    .equality_deletes
186                    .insert(delete_file_path, EqDelState::Loaded(eq_del));
187            }
188            notify.notify_waiters();
189        });
190    }
191}
192
193pub(crate) fn is_equality_delete(f: &FileScanTaskDeleteFile) -> bool {
194    matches!(f.file_type, DataContentType::EqualityDeletes)
195}

222        let tmp_dir = TempDir::new().unwrap();
223        let table_location = tmp_dir.path();
224        let file_io = FileIO::from_path(table_location.as_os_str().to_str().unwrap())
225            .unwrap()
226            .build()

228
229        // Note that with the delete file parsing not yet in place, all we can test here is that
230        // the call to the loader fails with the expected FeatureUnsupportedError.
231        let delete_file_manager = CachingDeleteFileLoader::new(file_io.clone(), 10);
232
233        let file_scan_tasks = setup(table_location);
234
235        let result = delete_file_manager
236            .load_deletes(&file_scan_tasks[0].deletes, file_scan_tasks[0].schema_ref())

238            .unwrap();
239
240        assert!(result.is_err_and(|e| e.kind() == ErrorKind::FeatureUnsupported));
