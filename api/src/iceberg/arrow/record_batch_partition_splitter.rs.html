29/// Column name for the projected partition values struct
30pub const PROJECTED_PARTITION_VALUE_COLUMN: &str = "_partition";
31
32/// The splitter used to split the record batch into multiple record batches by the partition spec.
33/// 1. It will project and transform the input record batch based on the partition spec, get the partitioned record batch.
34/// 2. Split the input record batch into multiple record batches based on the partitioned record batch.
35///
36/// # Partition Value Modes
37///
38/// The splitter supports two modes for obtaining partition values:
39/// - **Computed mode** (`calculator` is `Some`): Computes partition values from source columns using transforms
40/// - **Pre-computed mode** (`calculator` is `None`): Expects a `_partition` column in the input batch
41// # TODO
42// Remove this after partition writer supported.
43#[allow(dead_code)]
44pub struct RecordBatchPartitionSplitter {
45    schema: SchemaRef,
46    partition_spec: PartitionSpecRef,
47    calculator: Option<PartitionValueCalculator>,
48    partition_type: StructType,
49}
50
51// # TODO
52// Remove this after partition writer supported.
53#[allow(dead_code)]
54impl RecordBatchPartitionSplitter {
55    /// Create a new RecordBatchPartitionSplitter.
56    ///
57    /// # Arguments
58    ///
59    /// * `iceberg_schema` - The Iceberg schema reference
60    /// * `partition_spec` - The partition specification reference
61    /// * `calculator` - Optional calculator for computing partition values from source columns.
62    ///   - `Some(calculator)`: Compute partition values from source columns using transforms
63    ///   - `None`: Expect a pre-computed `_partition` column in the input batch
64    ///
65    /// # Returns
66    ///
67    /// Returns a new `RecordBatchPartitionSplitter` instance or an error if initialization fails.
68    pub fn new(
69        iceberg_schema: SchemaRef,
70        partition_spec: PartitionSpecRef,
71        calculator: Option<PartitionValueCalculator>,
72    ) -> Result<Self> {
73        let partition_type = partition_spec.partition_type(&iceberg_schema)?;
74
75        Ok(Self {
76            schema: iceberg_schema,
77            partition_spec,
78            calculator,
79            partition_type,
80        })
81    }
82
83    /// Create a new RecordBatchPartitionSplitter with computed partition values.
84    ///
85    /// This is a convenience method that creates a calculator and initializes the splitter
86    /// to compute partition values from source columns.
87    ///
88    /// # Arguments
89    ///
90    /// * `iceberg_schema` - The Iceberg schema reference
91    /// * `partition_spec` - The partition specification reference
92    ///
93    /// # Returns
94    ///
95    /// Returns a new `RecordBatchPartitionSplitter` instance or an error if initialization fails.
96    pub fn new_with_computed_values(
97        iceberg_schema: SchemaRef,
98        partition_spec: PartitionSpecRef,
99    ) -> Result<Self> {
100        let calculator = PartitionValueCalculator::try_new(&partition_spec, &iceberg_schema)?;
101        Self::new(iceberg_schema, partition_spec, Some(calculator))
102    }
103
104    /// Create a new RecordBatchPartitionSplitter expecting pre-computed partition values.
105    ///
106    /// This is a convenience method that initializes the splitter to expect a `_partition`
107    /// column in the input batches.
108    ///
109    /// # Arguments
110    ///
111    /// * `iceberg_schema` - The Iceberg schema reference
112    /// * `partition_spec` - The partition specification reference
113    ///
114    /// # Returns
115    ///
116    /// Returns a new `RecordBatchPartitionSplitter` instance or an error if initialization fails.
117    pub fn new_with_precomputed_values(
118        iceberg_schema: SchemaRef,
119        partition_spec: PartitionSpecRef,
120    ) -> Result<Self> {
121        Self::new(iceberg_schema, partition_spec, None)
122    }

124    /// Split the record batch into multiple record batches based on the partition spec.
125    pub fn split(&self, batch: &RecordBatch) -> Result<Vec<(PartitionKey, RecordBatch)>> {
126        let partition_structs = if let Some(calculator) = &self.calculator {
127            // Compute partition values from source columns using calculator
128            let partition_array = calculator.calculate(batch)?;
129            let struct_array = arrow_struct_to_literal(&partition_array, &self.partition_type)?;
130
131            struct_array
132                .into_iter()
133                .map(|s| {
134                    if let Some(Literal::Struct(s)) = s {
135                        Ok(s)
136                    } else {
137                        Err(Error::new(
138                            ErrorKind::DataInvalid,
139                            "Partition value is not a struct literal or is null",
140                        ))
141                    }
142                })
143                .collect::<Result<Vec<_>>>()?
144        } else {
145            // Extract partition values from pre-computed partition column
146            let partition_column = batch
147                .column_by_name(PROJECTED_PARTITION_VALUE_COLUMN)
148                .ok_or_else(|| {
149                    Error::new(
150                        ErrorKind::DataInvalid,
151                        format!(
152                            "Partition column '{}' not found in batch",
153                            PROJECTED_PARTITION_VALUE_COLUMN
154                        ),
155                    )
156                })?;
157
158            let partition_struct_array = partition_column
159                .as_any()
160                .downcast_ref::<StructArray>()
161                .ok_or_else(|| {
162                    Error::new(
163                        ErrorKind::DataInvalid,
164                        "Partition column is not a StructArray",
165                    )
166                })?;
167
168            let arrow_struct_array = Arc::new(partition_struct_array.clone()) as ArrayRef;
169            let struct_array = arrow_struct_to_literal(&arrow_struct_array, &self.partition_type)?;
170
171            struct_array
172                .into_iter()
173                .map(|s| {
174                    if let Some(Literal::Struct(s)) = s {
175                        Ok(s)
176                    } else {
177                        Err(Error::new(
178                            ErrorKind::DataInvalid,
179                            "Partition value is not a struct literal or is null",
180                        ))
181                    }
182                })
183                .collect::<Result<Vec<_>>>()?
184        };
185
186        // Group the batch by row value.
187        let mut group_ids = HashMap::new();
188        partition_structs
189            .iter()
190            .enumerate()
191            .for_each(|(row_id, row)| {
192                group_ids.entry(row.clone()).or_insert(vec![]).push(row_id);
193            });
194
195        // Partition the batch with same partition partition_values
196        let mut partition_batches = Vec::with_capacity(group_ids.len());
197        for (row, row_ids) in group_ids.into_iter() {
198            // generate the bool filter array from column_ids
199            let filter_array: BooleanArray = {
200                let mut filter = vec![false; batch.num_rows()];
201                row_ids.into_iter().for_each(|row_id| {
202                    filter[row_id] = true;
203                });
204                filter.into()
205            };
206
207            // Create PartitionKey from the partition struct
208            let partition_key = PartitionKey::new(
209                self.partition_spec.as_ref().clone(),
210                self.schema.clone(),
211                row,
212            );
213
214            // filter the RecordBatch
215            partition_batches.push((partition_key, filter_record_batch(batch, &filter_array)?));
216        }
217
218        Ok(partition_batches)
219    }
220}

237    #[test]
238    fn test_record_batch_partition_split() {
239        let schema = Arc::new(
240            Schema::builder()
241                .with_fields(vec![
242                    NestedField::required(
243                        1,
244                        "id",
245                        Type::Primitive(crate::spec::PrimitiveType::Int),
246                    )
247                    .into(),
248                    NestedField::required(
249                        2,
250                        "name",
251                        Type::Primitive(crate::spec::PrimitiveType::String),
252                    )
253                    .into(),
254                ])
255                .build()
256                .unwrap(),
257        );
258        let partition_spec = Arc::new(
259            PartitionSpecBuilder::new(schema.clone())
260                .with_spec_id(1)
261                .add_unbound_field(UnboundPartitionField {
262                    source_id: 1,
263                    field_id: None,
264                    name: "id_bucket".to_string(),
265                    transform: Transform::Identity,
266                })
267                .unwrap()
268                .build()
269                .unwrap(),
270        );
271        let partition_splitter =
272            RecordBatchPartitionSplitter::new_with_computed_values(schema.clone(), partition_spec)
273                .expect("Failed to create splitter");
274
275        let arrow_schema = Arc::new(schema_to_arrow_schema(&schema).unwrap());
276        let id_array = Int32Array::from(vec![1, 2, 1, 3, 2, 3, 1]);
277        let data_array = StringArray::from(vec!["a", "b", "c", "d", "e", "f", "g"]);
278        let batch = RecordBatch::try_new(arrow_schema.clone(), vec![
279            Arc::new(id_array),
280            Arc::new(data_array),
281        ])
282        .expect("Failed to create RecordBatch");
283
284        let mut partitioned_batches = partition_splitter
285            .split(&batch)
286            .expect("Failed to split RecordBatch");
287        partitioned_batches.sort_by_key(|(partition_key, _)| {
288            if let PrimitiveLiteral::Int(i) = partition_key.data().fields()[0]
289                .as_ref()
290                .unwrap()
291                .as_primitive_literal()
292                .unwrap()
293            {
294                i
295            } else {
296                panic!("The partition value is not a int");
297            }
298        });
299        assert_eq!(partitioned_batches.len(), 3);
300        {
301            // check the first partition
302            let expected_id_array = Int32Array::from(vec![1, 1, 1]);
303            let expected_data_array = StringArray::from(vec!["a", "c", "g"]);
304            let expected_batch = RecordBatch::try_new(arrow_schema.clone(), vec![
305                Arc::new(expected_id_array),
306                Arc::new(expected_data_array),
307            ])
308            .expect("Failed to create expected RecordBatch");
309            assert_eq!(partitioned_batches[0].1, expected_batch);
310        }
311        {
312            // check the second partition
313            let expected_id_array = Int32Array::from(vec![2, 2]);
314            let expected_data_array = StringArray::from(vec!["b", "e"]);
315            let expected_batch = RecordBatch::try_new(arrow_schema.clone(), vec![
316                Arc::new(expected_id_array),
317                Arc::new(expected_data_array),
318            ])
319            .expect("Failed to create expected RecordBatch");
320            assert_eq!(partitioned_batches[1].1, expected_batch);
321        }
322        {
323            // check the third partition
324            let expected_id_array = Int32Array::from(vec![3, 3]);
325            let expected_data_array = StringArray::from(vec!["d", "f"]);
326            let expected_batch = RecordBatch::try_new(arrow_schema.clone(), vec![
327                Arc::new(expected_id_array),
328                Arc::new(expected_data_array),
329            ])
330            .expect("Failed to create expected RecordBatch");
331            assert_eq!(partitioned_batches[2].1, expected_batch);
332        }
333
334        let partition_values = partitioned_batches
335            .iter()
336            .map(|(partition_key, _)| partition_key.data().clone())
337            .collect::<Vec<_>>();
338        // check partition value is struct(1), struct(2), struct(3)
339        assert_eq!(partition_values, vec![
340            Struct::from_iter(vec![Some(Literal::int(1))]),
341            Struct::from_iter(vec![Some(Literal::int(2))]),
342            Struct::from_iter(vec![Some(Literal::int(3))]),
343        ]);
344    }

346    #[test]
347    fn test_record_batch_partition_split_with_partition_column() {
348        use arrow_array::StructArray;
349        use arrow_schema::{Field, Schema as ArrowSchema};
350
351        let schema = Arc::new(
352            Schema::builder()
353                .with_fields(vec![
354                    NestedField::required(
355                        1,
356                        "id",
357                        Type::Primitive(crate::spec::PrimitiveType::Int),
358                    )
359                    .into(),
360                    NestedField::required(
361                        2,
362                        "name",
363                        Type::Primitive(crate::spec::PrimitiveType::String),
364                    )
365                    .into(),
366                ])
367                .build()
368                .unwrap(),
369        );
370        let partition_spec = Arc::new(
371            PartitionSpecBuilder::new(schema.clone())
372                .with_spec_id(1)
373                .add_unbound_field(UnboundPartitionField {
374                    source_id: 1,
375                    field_id: None,
376                    name: "id_bucket".to_string(),
377                    transform: Transform::Identity,
378                })
379                .unwrap()
380                .build()
381                .unwrap(),
382        );
383
384        // Create input schema with _partition column
385        // Note: partition field IDs start from 1000 by default
386        let partition_field = Field::new("id_bucket", DataType::Int32, false).with_metadata(
387            HashMap::from([(PARQUET_FIELD_ID_META_KEY.to_string(), "1000".to_string())]),
388        );
389        let partition_struct_field = Field::new(
390            PROJECTED_PARTITION_VALUE_COLUMN,
391            DataType::Struct(vec![partition_field.clone()].into()),
392            false,
393        );
394
395        let input_schema = Arc::new(ArrowSchema::new(vec![
396            Field::new("id", DataType::Int32, false),
397            Field::new("name", DataType::Utf8, false),
398            partition_struct_field,
399        ]));
400
401        // Create splitter expecting pre-computed partition column
402        let partition_splitter = RecordBatchPartitionSplitter::new_with_precomputed_values(
403            schema.clone(),
404            partition_spec,
405        )
406        .expect("Failed to create splitter");
407
408        // Create test data with pre-computed partition column
409        let id_array = Int32Array::from(vec![1, 2, 1, 3, 2, 3, 1]);
410        let data_array = StringArray::from(vec!["a", "b", "c", "d", "e", "f", "g"]);
411
412        // Create partition column (same values as id for Identity transform)
413        let partition_values = Int32Array::from(vec![1, 2, 1, 3, 2, 3, 1]);
414        let partition_struct = StructArray::from(vec![(
415            Arc::new(partition_field),
416            Arc::new(partition_values) as ArrayRef,
417        )]);
418
419        let batch = RecordBatch::try_new(input_schema.clone(), vec![
420            Arc::new(id_array),
421            Arc::new(data_array),
422            Arc::new(partition_struct),
423        ])
424        .expect("Failed to create RecordBatch");
425
426        // Split using the pre-computed partition column
427        let mut partitioned_batches = partition_splitter
428            .split(&batch)
429            .expect("Failed to split RecordBatch");
430
431        partitioned_batches.sort_by_key(|(partition_key, _)| {
432            if let PrimitiveLiteral::Int(i) = partition_key.data().fields()[0]
433                .as_ref()
434                .unwrap()
435                .as_primitive_literal()
436                .unwrap()
437            {
438                i
439            } else {
440                panic!("The partition value is not a int");
441            }
442        });
443
444        assert_eq!(partitioned_batches.len(), 3);
445
446        // Helper to extract id and name values from a batch
447        let extract_values = |batch: &RecordBatch| -> (Vec<i32>, Vec<String>) {
448            let id_col = batch
449                .column(0)
450                .as_any()
451                .downcast_ref::<Int32Array>()
452                .unwrap();
453            let name_col = batch
454                .column(1)
455                .as_any()
456                .downcast_ref::<StringArray>()
457                .unwrap();
458            (
459                id_col.values().to_vec(),
460                name_col.iter().map(|s| s.unwrap().to_string()).collect(),
461            )
462        };
463
464        // Verify partition 1: id=1, names=["a", "c", "g"]
465        let (key, batch) = &partitioned_batches[0];
466        assert_eq!(key.data(), &Struct::from_iter(vec![Some(Literal::int(1))]));
467        let (ids, names) = extract_values(batch);
468        assert_eq!(ids, vec![1, 1, 1]);
469        assert_eq!(names, vec!["a", "c", "g"]);
470
471        // Verify partition 2: id=2, names=["b", "e"]
472        let (key, batch) = &partitioned_batches[1];
473        assert_eq!(key.data(), &Struct::from_iter(vec![Some(Literal::int(2))]));
474        let (ids, names) = extract_values(batch);
475        assert_eq!(ids, vec![2, 2]);
476        assert_eq!(names, vec!["b", "e"]);
477
478        // Verify partition 3: id=3, names=["d", "f"]
479        let (key, batch) = &partitioned_batches[2];
480        assert_eq!(key.data(), &Struct::from_iter(vec![Some(Literal::int(3))]));
481        let (ids, names) = extract_values(batch);
482        assert_eq!(ids, vec![3, 3]);
483        assert_eq!(names, vec!["d", "f"]);
484    }
