18//! This module contains the location generator and file name generator for generating path of data file.
19
20use std::sync::Arc;
21use std::sync::atomic::AtomicU64;
22
23use crate::spec::{DataFileFormat, TableMetadata};
24use crate::{Error, ErrorKind, Result};
25
26/// `LocationGenerator` used to generate the location of data file.
27pub trait LocationGenerator: Clone + Send + 'static {
28    /// Generate an absolute path for the given file name.
29    /// e.g
30    /// For file name "part-00000.parquet", the generated location maybe "/table/data/part-00000.parquet"
31    fn generate_location(&self, file_name: &str) -> String;
32}
33
34const WRITE_DATA_LOCATION: &str = "write.data.path";
35const WRITE_FOLDER_STORAGE_LOCATION: &str = "write.folder-storage.path";
36const DEFAULT_DATA_DIR: &str = "/data";
37
38#[derive(Clone, Debug)]
39/// `DefaultLocationGenerator` used to generate the data dir location of data file.
40/// The location is generated based on the table location and the data location in table properties.
41pub struct DefaultLocationGenerator {
42    dir_path: String,
43}
44
45impl DefaultLocationGenerator {
46    /// Create a new `DefaultLocationGenerator`.
47    pub fn new(table_metadata: TableMetadata) -> Result<Self> {
48        let table_location = table_metadata.location();
49        let rel_dir_path = {
50            let prop = table_metadata.properties();
51            let data_location = prop
52                .get(WRITE_DATA_LOCATION)
53                .or(prop.get(WRITE_FOLDER_STORAGE_LOCATION));
54            if let Some(data_location) = data_location {
55                data_location.strip_prefix(table_location).ok_or_else(|| {
56                    Error::new(
57                        ErrorKind::DataInvalid,
58                        format!(
59                            "data location {} is not a subpath of table location {}",
60                            data_location, table_location
61                        ),
62                    )
63                })?
64            } else {
65                DEFAULT_DATA_DIR
66            }
67        };
68
69        Ok(Self {
70            dir_path: format!("{}{}", table_location, rel_dir_path),
71        })
72    }
73}
74
75impl LocationGenerator for DefaultLocationGenerator {
76    fn generate_location(&self, file_name: &str) -> String {
77        format!("{}/{}", self.dir_path, file_name)
78    }
79}

81/// `FileNameGeneratorTrait` used to generate file name for data file. The file name can be passed to `LocationGenerator` to generate the location of the file.
82pub trait FileNameGenerator: Clone + Send + 'static {
83    /// Generate a file name.
84    fn generate_file_name(&self) -> String;
85}
86
87/// `DefaultFileNameGenerator` used to generate file name for data file. The file name can be
88/// passed to `LocationGenerator` to generate the location of the file.
89/// The file name format is "{prefix}-{file_count}[-{suffix}].{file_format}".
90#[derive(Clone, Debug)]
91pub struct DefaultFileNameGenerator {
92    prefix: String,
93    suffix: String,
94    format: String,
95    file_count: Arc<AtomicU64>,
96}
97
98impl DefaultFileNameGenerator {
99    /// Create a new `FileNameGenerator`.
100    pub fn new(prefix: String, suffix: Option<String>, format: DataFileFormat) -> Self {
101        let suffix = if let Some(suffix) = suffix {
102            format!("-{}", suffix)
103        } else {
104            "".to_string()
105        };
106
107        Self {
108            prefix,
109            suffix,
110            format: format.to_string(),
111            file_count: Arc::new(AtomicU64::new(0)),
112        }
113    }
114}
115
116impl FileNameGenerator for DefaultFileNameGenerator {
117    fn generate_file_name(&self) -> String {
118        let file_id = self
119            .file_count
120            .fetch_add(1, std::sync::atomic::Ordering::Relaxed);
121        format!(
122            "{}-{:05}{}.{}",
123            self.prefix, file_id, self.suffix, self.format
124        )
125    }
126}

128#[cfg(test)]
129pub(crate) mod test {
130    use std::collections::HashMap;
131
132    use uuid::Uuid;
133
134    use super::LocationGenerator;
135    use crate::spec::{FormatVersion, PartitionSpec, StructType, TableMetadata};
136    use crate::writer::file_writer::location_generator::{
137        FileNameGenerator, WRITE_DATA_LOCATION, WRITE_FOLDER_STORAGE_LOCATION,
138    };
139
140    #[derive(Clone)]
141    pub(crate) struct MockLocationGenerator {
142        root: String,
143    }
144
145    impl MockLocationGenerator {
146        pub(crate) fn new(root: String) -> Self {
147            Self { root }
148        }
149    }
150
151    impl LocationGenerator for MockLocationGenerator {
152        fn generate_location(&self, file_name: &str) -> String {
153            format!("{}/{}", self.root, file_name)
154        }
155    }
156
157    #[test]
158    fn test_default_location_generate() {
159        let mut table_metadata = TableMetadata {
160            format_version: FormatVersion::V2,
161            table_uuid: Uuid::parse_str("fb072c92-a02b-11e9-ae9c-1bb7bc9eca94").unwrap(),
162            location: "s3://data.db/table".to_string(),
163            last_updated_ms: 1515100955770,
164            last_column_id: 1,
165            schemas: HashMap::new(),
166            current_schema_id: 1,
167            partition_specs: HashMap::new(),
168            default_spec: PartitionSpec::unpartition_spec().into(),
169            default_partition_type: StructType::new(vec![]),
170            last_partition_id: 1000,
171            default_sort_order_id: 0,
172            sort_orders: HashMap::from_iter(vec![]),
173            snapshots: HashMap::default(),
174            current_snapshot_id: None,
175            last_sequence_number: 1,
176            properties: HashMap::new(),
177            snapshot_log: Vec::new(),
178            metadata_log: vec![],
179            refs: HashMap::new(),
180            statistics: HashMap::new(),
181            partition_statistics: HashMap::new(),
182            encryption_keys: HashMap::new(),
183        };
184
185        let file_name_genertaor = super::DefaultFileNameGenerator::new(
186            "part".to_string(),
187            Some("test".to_string()),
188            crate::spec::DataFileFormat::Parquet,
189        );
190
191        // test default data location
192        let location_generator =
193            super::DefaultLocationGenerator::new(table_metadata.clone()).unwrap();
194        let location =
195            location_generator.generate_location(&file_name_genertaor.generate_file_name());
196        assert_eq!(location, "s3://data.db/table/data/part-00000-test.parquet");
197
198        // test custom data location
199        table_metadata.properties.insert(
200            WRITE_FOLDER_STORAGE_LOCATION.to_string(),
201            "s3://data.db/table/data_1".to_string(),
202        );
203        let location_generator =
204            super::DefaultLocationGenerator::new(table_metadata.clone()).unwrap();
205        let location =
206            location_generator.generate_location(&file_name_genertaor.generate_file_name());
207        assert_eq!(
208            location,
209            "s3://data.db/table/data_1/part-00001-test.parquet"
210        );
211
212        table_metadata.properties.insert(
213            WRITE_DATA_LOCATION.to_string(),
214            "s3://data.db/table/data_2".to_string(),
215        );
216        let location_generator =
217            super::DefaultLocationGenerator::new(table_metadata.clone()).unwrap();
218        let location =
219            location_generator.generate_location(&file_name_genertaor.generate_file_name());
220        assert_eq!(
221            location,
222            "s3://data.db/table/data_2/part-00002-test.parquet"
223        );
224
225        // test invalid data location
226        table_metadata.properties.insert(
227            WRITE_DATA_LOCATION.to_string(),
228            // invalid table location
229            "s3://data.db/data_3".to_string(),
230        );
231        let location_generator = super::DefaultLocationGenerator::new(table_metadata.clone());
232        assert!(location_generator.is_err());
233    }
