18use std::ops::BitOrAssign;
19
20use roaring::RoaringTreemap;
21use roaring::bitmap::Iter;
22use roaring::treemap::BitmapIter;
23
24#[derive(Debug, Default)]
25pub struct DeleteVector {
26    inner: RoaringTreemap,
27}
28
29impl DeleteVector {
30    #[allow(unused)]
31    pub(crate) fn new(roaring_treemap: RoaringTreemap) -> DeleteVector {
32        DeleteVector {
33            inner: roaring_treemap,
34        }
35    }
36
37    pub fn iter(&self) -> DeleteVectorIterator {
38        let outer = self.inner.bitmaps();
39        DeleteVectorIterator { outer, inner: None }
40    }
41}
42
43// Ideally, we'd just wrap `roaring::RoaringTreemap`'s iterator, `roaring::treemap::Iter` here.
44// But right now, it does not have a corresponding implementation of `roaring::bitmap::Iter::advance_to`,
45// which is very handy in ArrowReader::build_deletes_row_selection.
46// There is a PR open on roaring to add this (https://github.com/RoaringBitmap/roaring-rs/pull/314)
47// and if that gets merged then we can simplify `DeleteVectorIterator` here, refactoring `advance_to`
48// to just a wrapper around the underlying iterator's method.
49pub struct DeleteVectorIterator<'a> {
50    // NB: `BitMapIter` was only exposed publicly in https://github.com/RoaringBitmap/roaring-rs/pull/316
51    // which is not yet released. As a consequence our Cargo.toml temporarily uses a git reference for
52    // the roaring dependency.
53    outer: BitmapIter<'a>,
54    inner: Option<DeleteVectorIteratorInner<'a>>,
55}
56
57struct DeleteVectorIteratorInner<'a> {
58    high_bits: u32,
59    bitmap_iter: Iter<'a>,
60}
61
62impl Iterator for DeleteVectorIterator<'_> {
63    type Item = u64;
64
65    fn next(&mut self) -> Option<Self::Item> {
66        if let Some(inner) = &mut self.inner {
67            if let Some(inner_next) = inner.bitmap_iter.next() {
68                return Some(u64::from(inner.high_bits) << 32 | u64::from(inner_next));
69            }
70        }
71
72        if let Some((high_bits, next_bitmap)) = self.outer.next() {
73            self.inner = Some(DeleteVectorIteratorInner {
74                high_bits,
75                bitmap_iter: next_bitmap.iter(),
76            })
77        } else {
78            return None;
79        }
80
81        self.next()
82    }
83}
84
85impl DeleteVectorIterator<'_> {
86    pub fn advance_to(&mut self, pos: u64) {
87        let hi = (pos >> 32) as u32;
88        let lo = pos as u32;
89
90        let Some(ref mut inner) = self.inner else {
91            return;
92        };
93
94        while inner.high_bits < hi {
95            let Some((next_hi, next_bitmap)) = self.outer.next() else {
96                return;
97            };
98
99            *inner = DeleteVectorIteratorInner {
100                high_bits: next_hi,
101                bitmap_iter: next_bitmap.iter(),
102            }
103        }
104
105        inner.bitmap_iter.advance_to(lo);
106    }
107}
108
109impl BitOrAssign for DeleteVector {
110    fn bitor_assign(&mut self, other: Self) {
111        self.inner.bitor_assign(&other.inner);
112    }
113}
