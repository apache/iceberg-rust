34type OffsetIndex = Vec<OffsetIndexMetaData>;

36const IN_PREDICATE_LIMIT: usize = 200;
37
38enum MissingColBehavior {
39    CantMatch,
40    MightMatch,
41}
42
43enum PageNullCount {
44    AllNull,
45    NoneNull,
46    SomeNull,
47    Unknown,
48}
49
50impl PageNullCount {
51    fn from_row_and_null_counts(num_rows: usize, null_count: Option<i64>) -> Self {
52        match (num_rows, null_count) {
53            (x, Some(y)) if x == y as usize => PageNullCount::AllNull,
54            (_, Some(0)) => PageNullCount::NoneNull,
55            (_, Some(_)) => PageNullCount::SomeNull,
56            _ => PageNullCount::Unknown,
57        }
58    }
59}

61pub(crate) struct PageIndexEvaluator<'a> {
62    column_index: &'a [Index],
63    offset_index: &'a OffsetIndex,
64    row_group_metadata: &'a RowGroupMetaData,
65    iceberg_field_id_to_parquet_column_index: &'a HashMap<i32, usize>,
66    snapshot_schema: &'a Schema,
67    row_count_cache: HashMap<usize, Vec<usize>>,
68}
69
70impl<'a> PageIndexEvaluator<'a> {
71    pub(crate) fn new(
72        column_index: &'a [Index],
73        offset_index: &'a OffsetIndex,
74        row_group_metadata: &'a RowGroupMetaData,
75        field_id_map: &'a HashMap<i32, usize>,
76        snapshot_schema: &'a Schema,
77    ) -> Self {
78        Self {
79            column_index,
80            offset_index,
81            row_group_metadata,
82            iceberg_field_id_to_parquet_column_index: field_id_map,
83            snapshot_schema,
84            row_count_cache: HashMap::new(),
85        }
86    }
87
88    /// Evaluate this `PageIndexEvaluator`'s filter predicate against a
89    /// specific page's column index entry in a parquet file's page index.
90    /// [`ArrowReader`] uses the resulting [`RowSelection`] to reject
91    /// pages within a parquet file's row group that cannot contain rows
92    /// matching the filter predicate.
93    pub(crate) fn eval(
94        filter: &'a BoundPredicate,
95        column_index: &'a [Index],
96        offset_index: &'a OffsetIndex,
97        row_group_metadata: &'a RowGroupMetaData,
98        field_id_map: &'a HashMap<i32, usize>,
99        snapshot_schema: &'a Schema,
100    ) -> Result<Vec<RowSelector>> {
101        if row_group_metadata.num_rows() == 0 {
102            return Ok(vec![]);
103        }
104
105        let mut evaluator = Self::new(
106            column_index,
107            offset_index,
108            row_group_metadata,
109            field_id_map,
110            snapshot_schema,
111        );
112
113        Ok(visit(&mut evaluator, filter)?.iter().copied().collect())
114    }
115
116    fn select_all_rows(&self) -> Result<RowSelection> {
117        Ok(vec![RowSelector::select(
118            self.row_group_metadata.num_rows() as usize
119        )]
120        .into())
121    }
122
123    fn skip_all_rows(&self) -> Result<RowSelection> {
124        Ok(vec![RowSelector::skip(
125            self.row_group_metadata.num_rows() as usize
126        )]
127        .into())
128    }
129
130    fn calc_row_selection<F>(
131        &mut self,
132        field_id: i32,
133        predicate: F,
134        missing_col_behavior: MissingColBehavior,
135    ) -> Result<RowSelection>
136    where
137        F: Fn(Option<Datum>, Option<Datum>, PageNullCount) -> Result<bool>,
138    {
139        let Some(&parquet_column_index) =
140            self.iceberg_field_id_to_parquet_column_index.get(&field_id)
141        else {
142            // if the snapshot's column is not present in the row group,
143            // exit early
144            return match missing_col_behavior {
145                MissingColBehavior::CantMatch => self.skip_all_rows(),
146                MissingColBehavior::MightMatch => self.select_all_rows(),
147            };
148        };
149
150        let Some(field) = self.snapshot_schema.field_by_id(field_id) else {
151            return Err(Error::new(
152                ErrorKind::Unexpected,
153                format!("Field with id {} missing from snapshot schema", field_id),
154            ));
155        };
156
157        let Some(field_type) = field.field_type.as_primitive_type() else {
158            return Err(Error::new(
159                ErrorKind::Unexpected,
160                format!(
161                    "Field with id {} not convertible to primitive type",
162                    field_id
163                ),
164            ));
165        };
166
167        let Some(column_index) = self.column_index.get(parquet_column_index) else {
168            // This should not happen, but we fail soft anyway so that the scan is still
169            // successful, just a bit slower
170            return self.select_all_rows();
171        };
172
173        let row_counts = {
174            // Caches row count calculations for columns that appear multiple times in
175            // the predicate
176            match self.row_count_cache.get(&parquet_column_index) {
177                Some(count) => count.clone(),
178                None => {
179                    let Some(offset_index) = self.offset_index.get(parquet_column_index) else {
180                        // if we have a column index, we should always have an offset index.
181                        return Err(Error::new(
182                            ErrorKind::Unexpected,
183                            format!("Missing offset index for field id {}", field_id),
184                        ));
185                    };
186
187                    let count = self.calc_row_counts(offset_index);
188                    self.row_count_cache
189                        .insert(parquet_column_index, count.clone());
190
191                    count
192                }
193            }
194        };
195
196        let Some(page_filter) = Self::apply_predicate_to_column_index(
197            predicate,
198            field_type,
199            column_index,
200            &row_counts,
201        )?
202        else {
203            return self.select_all_rows();
204        };
205
206        let row_selectors: Vec<_> = row_counts
207            .iter()
208            .zip(page_filter.iter())
209            .map(|(&row_count, &is_selected)| {
210                if is_selected {
211                    RowSelector::select(row_count)
212                } else {
213                    RowSelector::skip(row_count)
214                }
215            })
216            .collect();
217
218        Ok(row_selectors.into())
219    }
220
221    /// Returns a list of row counts per page
222    fn calc_row_counts(&self, offset_index: &OffsetIndexMetaData) -> Vec<usize> {
223        let mut remaining_rows = self.row_group_metadata.num_rows() as usize;
224        let mut row_counts = Vec::with_capacity(self.offset_index.len());
225
226        let page_locations = offset_index.page_locations();
227        for (idx, page_location) in page_locations.iter().enumerate() {
228            let row_count = if idx < page_locations.len() - 1 {
229                let row_count = (page_locations[idx + 1].first_row_index
230                    - page_location.first_row_index) as usize;
231                remaining_rows -= row_count;
232                row_count
233            } else {
234                remaining_rows
235            };
236
237            row_counts.push(row_count);
238        }
239
240        row_counts
241    }

243    fn apply_predicate_to_column_index<F>(
244        predicate: F,
245        field_type: &PrimitiveType,
246        column_index: &Index,
247        row_counts: &[usize],
248    ) -> Result<Option<Vec<bool>>>
249    where
250        F: Fn(Option<Datum>, Option<Datum>, PageNullCount) -> Result<bool>,
251    {
252        let result: Result<Vec<bool>> = match column_index {
253            Index::NONE => {
254                return Ok(None);
255            }
256            Index::BOOLEAN(idx) => idx
257                .indexes
258                .iter()
259                .zip(row_counts.iter())
260                .map(|(item, &row_count)| {
261                    predicate(
262                        item.min.map(|val| {
263                            Datum::new(field_type.clone(), PrimitiveLiteral::Boolean(val))
264                        }),
265                        item.max.map(|val| {
266                            Datum::new(field_type.clone(), PrimitiveLiteral::Boolean(val))
267                        }),
268                        PageNullCount::from_row_and_null_counts(row_count, item.null_count),
269                    )
270                })
271                .collect(),
272            Index::INT32(idx) => idx
273                .indexes
274                .iter()
275                .zip(row_counts.iter())
276                .map(|(item, &row_count)| {
277                    predicate(
278                        item.min
279                            .map(|val| Datum::new(field_type.clone(), PrimitiveLiteral::Int(val))),
280                        item.max
281                            .map(|val| Datum::new(field_type.clone(), PrimitiveLiteral::Int(val))),
282                        PageNullCount::from_row_and_null_counts(row_count, item.null_count),
283                    )
284                })
285                .collect(),
286            Index::INT64(idx) => idx
287                .indexes
288                .iter()
289                .zip(row_counts.iter())
290                .map(|(item, &row_count)| {
291                    predicate(
292                        item.min
293                            .map(|val| Datum::new(field_type.clone(), PrimitiveLiteral::Long(val))),
294                        item.max
295                            .map(|val| Datum::new(field_type.clone(), PrimitiveLiteral::Long(val))),
296                        PageNullCount::from_row_and_null_counts(row_count, item.null_count),
297                    )
298                })
299                .collect(),
300            Index::FLOAT(idx) => idx
301                .indexes
302                .iter()
303                .zip(row_counts.iter())
304                .map(|(item, &row_count)| {
305                    predicate(
306                        item.min.map(|val| {
307                            Datum::new(
308                                field_type.clone(),
309                                PrimitiveLiteral::Float(OrderedFloat::from(val)),
310                            )
311                        }),
312                        item.max.map(|val| {
313                            Datum::new(
314                                field_type.clone(),
315                                PrimitiveLiteral::Float(OrderedFloat::from(val)),
316                            )
317                        }),
318                        PageNullCount::from_row_and_null_counts(row_count, item.null_count),
319                    )
320                })
321                .collect(),
322            Index::DOUBLE(idx) => idx
323                .indexes
324                .iter()
325                .zip(row_counts.iter())
326                .map(|(item, &row_count)| {
327                    predicate(
328                        item.min.map(|val| {
329                            Datum::new(
330                                field_type.clone(),
331                                PrimitiveLiteral::Double(OrderedFloat::from(val)),
332                            )
333                        }),
334                        item.max.map(|val| {
335                            Datum::new(
336                                field_type.clone(),
337                                PrimitiveLiteral::Double(OrderedFloat::from(val)),
338                            )
339                        }),
340                        PageNullCount::from_row_and_null_counts(row_count, item.null_count),
341                    )
342                })
343                .collect(),
344            Index::BYTE_ARRAY(idx) => idx
345                .indexes
346                .iter()
347                .zip(row_counts.iter())
348                .map(|(item, &row_count)| {
349                    predicate(
350                        item.min.clone().map(|val| {
351                            Datum::new(
352                                field_type.clone(),
353                                PrimitiveLiteral::String(
354                                    String::from_utf8(val.data().to_vec()).unwrap(),
355                                ),
356                            )
357                        }),
358                        item.max.clone().map(|val| {
359                            Datum::new(
360                                field_type.clone(),
361                                PrimitiveLiteral::String(
362                                    String::from_utf8(val.data().to_vec()).unwrap(),
363                                ),
364                            )
365                        }),
366                        PageNullCount::from_row_and_null_counts(row_count, item.null_count),
367                    )
368                })
369                .collect(),
370            Index::FIXED_LEN_BYTE_ARRAY(_) => {
371                return Err(Error::new(
372                    ErrorKind::FeatureUnsupported,
373                    "unsupported 'FIXED_LEN_BYTE_ARRAY' index type in column_index",
374                ));
375            }
376            Index::INT96(_) => {
377                return Err(Error::new(
378                    ErrorKind::FeatureUnsupported,
379                    "unsupported 'INT96' index type in column_index",
380                ));
381            }
382        };
383
384        Ok(Some(result?))
385    }
386
387    fn visit_inequality(
388        &mut self,
389        reference: &BoundReference,
390        datum: &Datum,
391        cmp_fn: fn(&Datum, &Datum) -> bool,
392        use_lower_bound: bool,
393    ) -> Result<RowSelection> {
394        let field_id = reference.field().id;
395
396        self.calc_row_selection(
397            field_id,
398            |min, max, null_count| {
399                if matches!(null_count, PageNullCount::AllNull) {
400                    return Ok(false);
401                }
402
403                if datum.is_nan() {
404                    // NaN indicates unreliable bounds.
405                    return Ok(true);
406                }
407
408                let bound = if use_lower_bound { min } else { max };
409
410                if let Some(bound) = bound {
411                    if cmp_fn(&bound, datum) {
412                        return Ok(true);
413                    }
414
415                    return Ok(false);
416                }
417
418                Ok(true)
419            },
420            MissingColBehavior::MightMatch,
421        )
422    }

425impl BoundPredicateVisitor for PageIndexEvaluator<'_> {
426    type T = RowSelection;
427
428    fn always_true(&mut self) -> Result<RowSelection> {
429        self.select_all_rows()
430    }
431
432    fn always_false(&mut self) -> Result<RowSelection> {
433        self.skip_all_rows()
434    }
435
436    fn and(&mut self, lhs: RowSelection, rhs: RowSelection) -> Result<RowSelection> {
437        Ok(lhs.intersection(&rhs))
438    }
439
440    fn or(&mut self, lhs: RowSelection, rhs: RowSelection) -> Result<RowSelection> {
441        Ok(lhs.union(&rhs))
442    }
443
444    fn not(&mut self, _: RowSelection) -> Result<RowSelection> {
445        Err(Error::new(
446            ErrorKind::Unexpected,
447            "NOT unsupported at this point. NOT-rewrite should be performed first",
448        ))
449    }
450
451    fn is_null(
452        &mut self,
453        reference: &BoundReference,
454        _predicate: &BoundPredicate,
455    ) -> Result<RowSelection> {
456        let field_id = reference.field().id;
457
458        self.calc_row_selection(
459            field_id,
460            |_max, _min, null_count| Ok(!matches!(null_count, PageNullCount::NoneNull)),
461            MissingColBehavior::MightMatch,
462        )
463    }
464
465    fn not_null(
466        &mut self,
467        reference: &BoundReference,
468        _predicate: &BoundPredicate,
469    ) -> Result<RowSelection> {
470        let field_id = reference.field().id;
471
472        self.calc_row_selection(
473            field_id,
474            |_max, _min, null_count| Ok(!matches!(null_count, PageNullCount::AllNull)),
475            MissingColBehavior::CantMatch,
476        )
477    }
478
479    fn is_nan(
480        &mut self,
481        reference: &BoundReference,
482        _predicate: &BoundPredicate,
483    ) -> Result<RowSelection> {
484        // NaN counts not present in ColumnChunkMetadata Statistics.
485        // Only float columns can be NaN.
486        if reference.field().field_type.is_floating_type() {
487            self.select_all_rows()
488        } else {
489            self.skip_all_rows()
490        }
491    }
492
493    fn not_nan(
494        &mut self,
495        _reference: &BoundReference,
496        _predicate: &BoundPredicate,
497    ) -> Result<RowSelection> {
498        // NaN counts not present in ColumnChunkMetadata Statistics
499        self.select_all_rows()
500    }
501
502    fn less_than(
503        &mut self,
504        reference: &BoundReference,
505        datum: &Datum,
506        _predicate: &BoundPredicate,
507    ) -> Result<RowSelection> {
508        self.visit_inequality(reference, datum, PartialOrd::lt, true)
509    }

511    fn less_than_or_eq(
512        &mut self,
513        reference: &BoundReference,
514        datum: &Datum,
515        _predicate: &BoundPredicate,
516    ) -> Result<RowSelection> {
517        self.visit_inequality(reference, datum, PartialOrd::le, true)
518    }

520    fn greater_than(
521        &mut self,
522        reference: &BoundReference,
523        datum: &Datum,
524        _predicate: &BoundPredicate,
525    ) -> Result<RowSelection> {
526        self.visit_inequality(reference, datum, PartialOrd::gt, false)
527    }
528
529    fn greater_than_or_eq(
530        &mut self,
531        reference: &BoundReference,
532        datum: &Datum,
533        _predicate: &BoundPredicate,
534    ) -> Result<RowSelection> {
535        self.visit_inequality(reference, datum, PartialOrd::ge, false)
536    }
537
538    fn eq(
539        &mut self,
540        reference: &BoundReference,
541        datum: &Datum,
542        _predicate: &BoundPredicate,
543    ) -> Result<RowSelection> {
544        let field_id = reference.field().id;
545
546        self.calc_row_selection(
547            field_id,
548            |min, max, nulls| {
549                if matches!(nulls, PageNullCount::AllNull) {
550                    return Ok(false);
551                }
552
553                if let Some(min) = min {
554                    if min.gt(datum) {
555                        return Ok(false);
556                    }
557                }
558
559                if let Some(max) = max {
560                    if max.lt(datum) {
561                        return Ok(false);
562                    }
563                }
564
565                Ok(true)
566            },
567            MissingColBehavior::CantMatch,
568        )
569    }
570
571    fn not_eq(
572        &mut self,
573        _reference: &BoundReference,
574        _datum: &Datum,
575        _predicate: &BoundPredicate,
576    ) -> Result<RowSelection> {
577        // Because the bounds are not necessarily a min or max value,
578        // this cannot be answered using them. notEq(col, X) with (X, Y)
579        // doesn't guarantee that X is a value in col.
580        self.select_all_rows()
581    }
582
583    fn starts_with(
584        &mut self,
585        reference: &BoundReference,
586        datum: &Datum,
587        _predicate: &BoundPredicate,
588    ) -> Result<RowSelection> {
589        let field_id = reference.field().id;
590
591        let PrimitiveLiteral::String(datum) = datum.literal() else {
592            return Err(Error::new(
593                ErrorKind::Unexpected,
594                "Cannot use StartsWith operator on non-string values",
595            ));
596        };
597
598        self.calc_row_selection(
599            field_id,
600            |min, max, nulls| {
601                if matches!(nulls, PageNullCount::AllNull) {
602                    return Ok(false);
603                }
604
605                if let Some(lower_bound) = min {
606                    let PrimitiveLiteral::String(lower_bound) = lower_bound.literal() else {
607                        return Err(Error::new(
608                            ErrorKind::Unexpected,
609                            "Cannot use StartsWith operator on non-string lower_bound value",
610                        ));
611                    };
612
613                    let prefix_length = lower_bound.chars().count().min(datum.chars().count());
614
615                    // truncate lower bound so that its length
616                    // is not greater than the length of prefix
617                    let truncated_lower_bound =
618                        lower_bound.chars().take(prefix_length).collect::<String>();
619                    if datum < &truncated_lower_bound {
620                        return Ok(false);
621                    }
622                }
623
624                if let Some(upper_bound) = max {
625                    let PrimitiveLiteral::String(upper_bound) = upper_bound.literal() else {
626                        return Err(Error::new(
627                            ErrorKind::Unexpected,
628                            "Cannot use StartsWith operator on non-string upper_bound value",
629                        ));
630                    };
631
632                    let prefix_length = upper_bound.chars().count().min(datum.chars().count());
633
634                    // truncate upper bound so that its length
635                    // is not greater than the length of prefix
636                    let truncated_upper_bound =
637                        upper_bound.chars().take(prefix_length).collect::<String>();
638                    if datum > &truncated_upper_bound {
639                        return Ok(false);
640                    }
641                }
642
643                Ok(true)
644            },
645            MissingColBehavior::CantMatch,
646        )
647    }
648
649    fn not_starts_with(
650        &mut self,
651        reference: &BoundReference,
652        datum: &Datum,
653        _predicate: &BoundPredicate,
654    ) -> Result<RowSelection> {
655        let field_id = reference.field().id;
656
657        // notStartsWith will match unless all values must start with the prefix.
658        // This happens when the lower and upper bounds both start with the prefix.
659
660        let PrimitiveLiteral::String(prefix) = datum.literal() else {
661            return Err(Error::new(
662                ErrorKind::Unexpected,
663                "Cannot use StartsWith operator on non-string values",
664            ));
665        };
666
667        self.calc_row_selection(
668            field_id,
669            |min, max, nulls| {
670                if !matches!(nulls, PageNullCount::NoneNull) {
671                    return Ok(true);
672                }
673
674                let Some(lower_bound) = min else {
675                    return Ok(true);
676                };
677
678                let PrimitiveLiteral::String(lower_bound_str) = lower_bound.literal() else {
679                    return Err(Error::new(
680                        ErrorKind::Unexpected,
681                        "Cannot use NotStartsWith operator on non-string lower_bound value",
682                    ));
683                };
684
685                if lower_bound_str < prefix {
686                    // if lower is shorter than the prefix then lower doesn't start with the prefix
687                    return Ok(true);
688                }
689
690                let prefix_len = prefix.chars().count();
691
692                if lower_bound_str.chars().take(prefix_len).collect::<String>() == *prefix {
693                    // lower bound matches the prefix
694
695                    let Some(upper_bound) = max else {
696                        return Ok(true);
697                    };
698
699                    let PrimitiveLiteral::String(upper_bound) = upper_bound.literal() else {
700                        return Err(Error::new(
701                            ErrorKind::Unexpected,
702                            "Cannot use NotStartsWith operator on non-string upper_bound value",
703                        ));
704                    };
705
706                    // if upper is shorter than the prefix then upper can't start with the prefix
707                    if upper_bound.chars().count() < prefix_len {
708                        return Ok(true);
709                    }
710
711                    if upper_bound.chars().take(prefix_len).collect::<String>() == *prefix {
712                        // both bounds match the prefix, so all rows must match the
713                        // prefix and therefore do not satisfy the predicate
714                        return Ok(false);
715                    }
716                }
717
718                Ok(true)
719            },
720            MissingColBehavior::MightMatch,
721        )
722    }

724    fn r#in(
725        &mut self,
726        reference: &BoundReference,
727        literals: &FnvHashSet<Datum>,
728        _predicate: &BoundPredicate,
729    ) -> Result<RowSelection> {
730        let field_id = reference.field().id;
731
732        if literals.len() > IN_PREDICATE_LIMIT {
733            // skip evaluating the predicate if the number of values is too big
734            return self.select_all_rows();
735        }
736        self.calc_row_selection(
737            field_id,
738            |min, max, nulls| {
739                if matches!(nulls, PageNullCount::AllNull) {
740                    return Ok(false);
741                }
742
743                match (min, max) {
744                    (Some(min), Some(max)) => {
745                        if literals
746                            .iter()
747                            .all(|datum| datum.lt(&min) || datum.gt(&max))
748                        {
749                            // if all values are outside the bounds, rows cannot match.
750                            return Ok(false);
751                        }
752                    }
753                    (Some(min), _) => {
754                        if !literals.iter().any(|datum| datum.ge(&min)) {
755                            // if none of the values are greater than the min bound, rows cant match
756                            return Ok(false);
757                        }
758                    }
759                    (_, Some(max)) => {
760                        if !literals.iter().any(|datum| datum.le(&max)) {
761                            // if all values are greater than upper bound, rows cannot match.
762                            return Ok(false);
763                        }
764                    }
765
766                    _ => {}
767                }
768
769                Ok(true)
770            },
771            MissingColBehavior::CantMatch,
772        )
773    }
774
775    fn not_in(
776        &mut self,
777        _reference: &BoundReference,
778        _literals: &FnvHashSet<Datum>,
779        _predicate: &BoundPredicate,
780    ) -> Result<RowSelection> {
781        // Because the bounds are not necessarily a min or max value,
782        // this cannot be answered using them. notIn(col, {X, ...})
783        // with (X, Y) doesn't guarantee that X is a value in col.
784        self.select_all_rows()
785    }
786}
