35const ELEMENT_ID: &str = "element-id";
36const FILED_ID_PROP: &str = "field-id";
37const KEY_ID: &str = "key-id";
38const VALUE_ID: &str = "value-id";
39const UUID_BYTES: usize = 16;
40const UUID_LOGICAL_TYPE: &str = "uuid";
41const MAP_LOGICAL_TYPE: &str = "map";
42// # TODO: https://github.com/apache/iceberg-rust/issues/86
43// This const may better to maintain in avro-rs.
44const LOGICAL_TYPE: &str = "logicalType";

46struct SchemaToAvroSchema {
47    schema: String,
48}
49
50type AvroSchemaOrField = Either<AvroSchema, AvroRecordField>;
51
52impl SchemaVisitor for SchemaToAvroSchema {
53    type T = AvroSchemaOrField;
54
55    fn schema(&mut self, _schema: &Schema, value: AvroSchemaOrField) -> Result<AvroSchemaOrField> {
56        let mut avro_schema = value.unwrap_left();
57
58        if let AvroSchema::Record(record) = &mut avro_schema {
59            record.name = Name::from(self.schema.as_str());
60        } else {
61            return Err(Error::new(
62                ErrorKind::Unexpected,
63                "Schema result must be avro record!",
64            ));
65        }
66
67        Ok(Either::Left(avro_schema))
68    }
69
70    fn field(
71        &mut self,
72        field: &NestedFieldRef,
73        avro_schema: AvroSchemaOrField,
74    ) -> Result<AvroSchemaOrField> {
75        let mut field_schema = avro_schema.unwrap_left();
76        if let AvroSchema::Record(record) = &mut field_schema {
77            record.name = Name::from(format!("r{}", field.id).as_str());
78        }
79
80        if !field.required {
81            field_schema = avro_optional(field_schema)?;
82        }
83
84        let mut avro_record_field = AvroRecordField {
85            name: field.name.clone(),
86            schema: field_schema,
87            order: RecordFieldOrder::Ignore,
88            position: 0,
89            doc: field.doc.clone(),
90            aliases: None,
91            default: None,
92            custom_attributes: Default::default(),
93        };
94
95        if !field.required {
96            avro_record_field.default = Some(Value::Null);
97        }
98        avro_record_field.custom_attributes.insert(
99            FILED_ID_PROP.to_string(),
100            Value::Number(Number::from(field.id)),
101        );
102
103        Ok(Either::Right(avro_record_field))
104    }
105
106    fn r#struct(
107        &mut self,
108        _struct: &StructType,
109        results: Vec<AvroSchemaOrField>,
110    ) -> Result<AvroSchemaOrField> {
111        let avro_fields = results.into_iter().map(|r| r.unwrap_right()).collect_vec();
112
113        Ok(Either::Left(
114            // The name of this record schema should be determined later, by schema name or field
115            // name, here we use a temporary placeholder to do it.
116            avro_record_schema("null", avro_fields)?,
117        ))
118    }
119
120    fn list(&mut self, list: &ListType, value: AvroSchemaOrField) -> Result<AvroSchemaOrField> {
121        let mut field_schema = value.unwrap_left();
122
123        if let AvroSchema::Record(record) = &mut field_schema {
124            record.name = Name::from(format!("r{}", list.element_field.id).as_str());
125        }
126
127        if !list.element_field.required {
128            field_schema = avro_optional(field_schema)?;
129        }
130
131        Ok(Either::Left(AvroSchema::Array(ArraySchema {
132            items: Box::new(field_schema),
133            attributes: BTreeMap::from([(
134                ELEMENT_ID.to_string(),
135                Value::Number(Number::from(list.element_field.id)),
136            )]),
137        })))
138    }
139
140    fn map(
141        &mut self,
142        map: &MapType,
143        key_value: AvroSchemaOrField,
144        value: AvroSchemaOrField,
145    ) -> Result<AvroSchemaOrField> {
146        let key_field_schema = key_value.unwrap_left();
147        let mut value_field_schema = value.unwrap_left();
148        if !map.value_field.required {
149            value_field_schema = avro_optional(value_field_schema)?;
150        }
151
152        if matches!(key_field_schema, AvroSchema::String) {
153            Ok(Either::Left(AvroSchema::Map(MapSchema {
154                types: Box::new(value_field_schema),
155                attributes: BTreeMap::from([
156                    (
157                        KEY_ID.to_string(),
158                        Value::Number(Number::from(map.key_field.id)),
159                    ),
160                    (
161                        VALUE_ID.to_string(),
162                        Value::Number(Number::from(map.value_field.id)),
163                    ),
164                ]),
165            })))
166        } else {
167            // Avro map requires that key must be string type. Here we convert it to array if key is
168            // not string type.
169            let key_field = {
170                let mut field = AvroRecordField {
171                    name: map.key_field.name.clone(),
172                    doc: None,
173                    aliases: None,
174                    default: None,
175                    schema: key_field_schema,
176                    order: RecordFieldOrder::Ascending,
177                    position: 0,
178                    custom_attributes: Default::default(),
179                };
180                field.custom_attributes.insert(
181                    FILED_ID_PROP.to_string(),
182                    Value::Number(Number::from(map.key_field.id)),
183                );
184                field
185            };
186
187            let value_field = {
188                let mut field = AvroRecordField {
189                    name: map.value_field.name.clone(),
190                    doc: None,
191                    aliases: None,
192                    default: None,
193                    schema: value_field_schema,
194                    order: RecordFieldOrder::Ignore,
195                    position: 0,
196                    custom_attributes: Default::default(),
197                };
198                field.custom_attributes.insert(
199                    FILED_ID_PROP.to_string(),
200                    Value::Number(Number::from(map.value_field.id)),
201                );
202                field
203            };
204
205            let fields = vec![key_field, value_field];
206            let item_avro_schema = avro_record_schema(
207                format!("k{}_v{}", map.key_field.id, map.value_field.id).as_str(),
208                fields,
209            )?;
210
211            Ok(Either::Left(AvroSchema::Array(ArraySchema {
212                items: Box::new(item_avro_schema),
213                attributes: BTreeMap::from([(
214                    LOGICAL_TYPE.to_string(),
215                    Value::String(MAP_LOGICAL_TYPE.to_string()),
216                )]),
217            })))
218        }
219    }
220
221    fn primitive(&mut self, p: &PrimitiveType) -> Result<AvroSchemaOrField> {
222        let avro_schema = match p {
223            PrimitiveType::Boolean => AvroSchema::Boolean,
224            PrimitiveType::Int => AvroSchema::Int,
225            PrimitiveType::Long => AvroSchema::Long,
226            PrimitiveType::Float => AvroSchema::Float,
227            PrimitiveType::Double => AvroSchema::Double,
228            PrimitiveType::Date => AvroSchema::Date,
229            PrimitiveType::Time => AvroSchema::TimeMicros,
230            PrimitiveType::Timestamp => AvroSchema::TimestampMicros,
231            PrimitiveType::Timestamptz => AvroSchema::TimestampMicros,
232            PrimitiveType::TimestampNs => AvroSchema::TimestampNanos,
233            PrimitiveType::TimestamptzNs => AvroSchema::TimestampNanos,
234            PrimitiveType::String => AvroSchema::String,
235            PrimitiveType::Uuid => avro_fixed_schema(UUID_BYTES, Some(UUID_LOGICAL_TYPE))?,
236            PrimitiveType::Fixed(len) => avro_fixed_schema((*len) as usize, None)?,
237            PrimitiveType::Binary => AvroSchema::Bytes,
238            PrimitiveType::Decimal { precision, scale } => {
239                avro_decimal_schema(*precision as usize, *scale as usize)?
240            }
241        };
242        Ok(Either::Left(avro_schema))
243    }
244}

246/// Converting iceberg schema to avro schema.
247pub(crate) fn schema_to_avro_schema(name: impl ToString, schema: &Schema) -> Result<AvroSchema> {
248    let mut converter = SchemaToAvroSchema {
249        schema: name.to_string(),
250    };
251
252    visit_schema(schema, &mut converter).map(Either::unwrap_left)
253}
254
255fn avro_record_schema(name: &str, fields: Vec<AvroRecordField>) -> Result<AvroSchema> {
256    let lookup = fields
257        .iter()
258        .enumerate()
259        .map(|f| (f.1.name.clone(), f.0))
260        .collect();
261
262    Ok(AvroSchema::Record(RecordSchema {
263        name: Name::new(name)?,
264        aliases: None,
265        doc: None,
266        fields,
267        lookup,
268        attributes: Default::default(),
269    }))
270}

272pub(crate) fn avro_fixed_schema(len: usize, logical_type: Option<&str>) -> Result<AvroSchema> {
273    let attributes = if let Some(logical_type) = logical_type {
274        BTreeMap::from([(
275            LOGICAL_TYPE.to_string(),
276            Value::String(logical_type.to_string()),
277        )])
278    } else {
279        Default::default()
280    };
281    Ok(AvroSchema::Fixed(FixedSchema {
282        name: Name::new(format!("fixed_{len}").as_str())?,
283        aliases: None,
284        doc: None,
285        size: len,
286        attributes,
287        default: None,
288    }))
289}

291pub(crate) fn avro_decimal_schema(precision: usize, scale: usize) -> Result<AvroSchema> {
292    // Avro decimal logical type annotates Avro bytes _or_ fixed types.
293    // https://avro.apache.org/docs/1.11.1/specification/_print/#decimal
294    // Iceberg spec: Stored as _fixed_ using the minimum number of bytes for the given precision.
295    // https://iceberg.apache.org/spec/#avro
296    Ok(AvroSchema::Decimal(DecimalSchema {
297        precision,
298        scale,
299        inner: Box::new(AvroSchema::Fixed(FixedSchema {
300            // Name is not restricted by the spec.
301            // Refer to iceberg-python https://github.com/apache/iceberg-python/blob/d8bc1ca9af7957ce4d4db99a52c701ac75db7688/pyiceberg/utils/schema_conversion.py#L574-L582
302            name: Name::new(&format!("decimal_{precision}_{scale}")).unwrap(),
303            aliases: None,
304            doc: None,
305            size: crate::spec::Type::decimal_required_bytes(precision as u32)? as usize,
306            attributes: Default::default(),
307            default: None,
308        })),
309    }))
310}
311
312fn avro_optional(avro_schema: AvroSchema) -> Result<AvroSchema> {
313    Ok(AvroSchema::Union(UnionSchema::new(vec![
314        AvroSchema::Null,
315        avro_schema,
316    ])?))
317}
318
319fn is_avro_optional(avro_schema: &AvroSchema) -> bool {
320    match avro_schema {
321        AvroSchema::Union(union) => union.is_nullable(),
322        _ => false,
323    }
324}

326/// Post order avro schema visitor.
327pub(crate) trait AvroSchemaVisitor {
328    type T;
329
330    fn record(&mut self, record: &RecordSchema, fields: Vec<Self::T>) -> Result<Self::T>;
331
332    fn union(&mut self, union: &UnionSchema, options: Vec<Self::T>) -> Result<Self::T>;
333
334    fn array(&mut self, array: &ArraySchema, item: Self::T) -> Result<Self::T>;
335    fn map(&mut self, map: &MapSchema, value: Self::T) -> Result<Self::T>;
336    // There are two representation for iceberg map in avro: array of key-value records, or map when keys are strings (optional),
337    // ref: https://iceberg.apache.org/spec/#avro
338    fn map_array(&mut self, array: &RecordSchema, key: Self::T, value: Self::T) -> Result<Self::T>;
339
340    fn primitive(&mut self, schema: &AvroSchema) -> Result<Self::T>;
341}
342
343/// Visit avro schema in post order visitor.
344pub(crate) fn visit<V: AvroSchemaVisitor>(schema: &AvroSchema, visitor: &mut V) -> Result<V::T> {
345    match schema {
346        AvroSchema::Record(record) => {
347            let field_results = record
348                .fields
349                .iter()
350                .map(|f| visit(&f.schema, visitor))
351                .collect::<Result<Vec<V::T>>>()?;
352
353            visitor.record(record, field_results)
354        }
355        AvroSchema::Union(union) => {
356            let option_results = union
357                .variants()
358                .iter()
359                .map(|f| visit(f, visitor))
360                .collect::<Result<Vec<V::T>>>()?;
361
362            visitor.union(union, option_results)
363        }
364        AvroSchema::Array(item) => {
365            if let Some(logical_type) = item
366                .attributes
367                .get(LOGICAL_TYPE)
368                .and_then(|v| Value::as_str(v))
369            {
370                if logical_type == MAP_LOGICAL_TYPE {
371                    if let AvroSchema::Record(record_schema) = &*item.items {
372                        let key = visit(&record_schema.fields[0].schema, visitor)?;
373                        let value = visit(&record_schema.fields[1].schema, visitor)?;
374                        return visitor.map_array(record_schema, key, value);
375                    } else {
376                        return Err(Error::new(
377                            ErrorKind::DataInvalid,
378                            "Can't convert avro map schema, item is not a record.",
379                        ));
380                    }
381                } else {
382                    return Err(Error::new(
383                        ErrorKind::FeatureUnsupported,
384                        format!(
385                            "Logical type {logical_type} is not support in iceberg array type.",
386                        ),
387                    ));
388                }
389            }
390            let item_result = visit(&item.items, visitor)?;
391            visitor.array(item, item_result)
392        }
393        AvroSchema::Map(inner) => {
394            let item_result = visit(&inner.types, visitor)?;
395            visitor.map(inner, item_result)
396        }
397        schema => visitor.primitive(schema),
398    }
399}

401struct AvroSchemaToSchema;
402
403impl AvroSchemaToSchema {
404    /// A convenient way to get element id(i32) from attributes.
405    #[inline]
406    fn get_element_id_from_attributes(
407        attributes: &BTreeMap<String, Value>,
408        name: &str,
409    ) -> Result<i32> {
410        attributes
411            .get(name)
412            .ok_or_else(|| {
413                Error::new(
414                    ErrorKind::DataInvalid,
415                    "Can't convert avro array schema, missing element id.",
416                )
417            })?
418            .as_i64()
419            .ok_or_else(|| {
420                Error::new(
421                    ErrorKind::DataInvalid,
422                    "Can't convert avro array schema, element id is not a valid i64 number.",
423                )
424            })?
425            .try_into()
426            .map_err(|_| {
427                Error::new(
428                    ErrorKind::DataInvalid,
429                    "Can't convert avro array schema, element id is not a valid i32.",
430                )
431            })
432    }
433}
434
435impl AvroSchemaVisitor for AvroSchemaToSchema {
436    // Only `AvroSchema::Null` will return `None`
437    type T = Option<Type>;
438
439    fn record(
440        &mut self,
441        record: &RecordSchema,
442        field_types: Vec<Option<Type>>,
443    ) -> Result<Option<Type>> {
444        let mut fields = Vec::with_capacity(field_types.len());
445        for (avro_field, field_type) in record.fields.iter().zip_eq(field_types) {
446            let field_id =
447                Self::get_element_id_from_attributes(&avro_field.custom_attributes, FILED_ID_PROP)?;
448
449            let optional = is_avro_optional(&avro_field.schema);
450
451            let mut field =
452                NestedField::new(field_id, &avro_field.name, field_type.unwrap(), !optional);
453
454            if let Some(doc) = &avro_field.doc {
455                field = field.with_doc(doc);
456            }
457
458            fields.push(field.into());
459        }
460
461        Ok(Some(Type::Struct(StructType::new(fields))))
462    }
463
464    fn union(
465        &mut self,
466        union: &UnionSchema,
467        mut options: Vec<Option<Type>>,
468    ) -> Result<Option<Type>> {
469        ensure_data_valid!(
470            options.len() <= 2 && !options.is_empty(),
471            "Can't convert avro union type {:?} to iceberg.",
472            union
473        );
474
475        if options.len() > 1 {
476            ensure_data_valid!(
477                options[0].is_none(),
478                "Can't convert avro union type {:?} to iceberg.",
479                union
480            );
481        }
482
483        if options.len() == 1 {
484            Ok(Some(options.remove(0).unwrap()))
485        } else {
486            Ok(Some(options.remove(1).unwrap()))
487        }
488    }
489
490    fn array(&mut self, array: &ArraySchema, item: Option<Type>) -> Result<Self::T> {
491        let element_field_id = Self::get_element_id_from_attributes(&array.attributes, ELEMENT_ID)?;
492        let element_field = NestedField::list_element(
493            element_field_id,
494            item.unwrap(),
495            !is_avro_optional(&array.items),
496        )
497        .into();
498        Ok(Some(Type::List(ListType { element_field })))
499    }
500
501    fn map(&mut self, map: &MapSchema, value: Option<Type>) -> Result<Option<Type>> {
502        let key_field_id = Self::get_element_id_from_attributes(&map.attributes, KEY_ID)?;
503        let key_field =
504            NestedField::map_key_element(key_field_id, Type::Primitive(PrimitiveType::String));
505        let value_field_id = Self::get_element_id_from_attributes(&map.attributes, VALUE_ID)?;
506        let value_field = NestedField::map_value_element(
507            value_field_id,
508            value.unwrap(),
509            !is_avro_optional(&map.types),
510        );
511        Ok(Some(Type::Map(MapType {
512            key_field: key_field.into(),
513            value_field: value_field.into(),
514        })))
515    }
516
517    fn primitive(&mut self, schema: &AvroSchema) -> Result<Option<Type>> {
518        let schema_type = match schema {
519            AvroSchema::Decimal(decimal) => {
520                Type::decimal(decimal.precision as u32, decimal.scale as u32)?
521            }
522            AvroSchema::Date => Type::Primitive(PrimitiveType::Date),
523            AvroSchema::TimeMicros => Type::Primitive(PrimitiveType::Time),
524            AvroSchema::TimestampMicros => Type::Primitive(PrimitiveType::Timestamp),
525            AvroSchema::TimestampNanos => Type::Primitive(PrimitiveType::TimestampNs),
526            AvroSchema::Boolean => Type::Primitive(PrimitiveType::Boolean),
527            AvroSchema::Int => Type::Primitive(PrimitiveType::Int),
528            AvroSchema::Long => Type::Primitive(PrimitiveType::Long),
529            AvroSchema::Float => Type::Primitive(PrimitiveType::Float),
530            AvroSchema::Double => Type::Primitive(PrimitiveType::Double),
531            AvroSchema::String | AvroSchema::Enum(_) => Type::Primitive(PrimitiveType::String),
532            AvroSchema::Fixed(fixed) => {
533                if let Some(logical_type) = fixed.attributes.get(LOGICAL_TYPE) {
534                    let logical_type = logical_type.as_str().ok_or_else(|| {
535                        Error::new(
536                            ErrorKind::DataInvalid,
537                            "logicalType in attributes of avro schema is not a string type",
538                        )
539                    })?;
540                    match logical_type {
541                        UUID_LOGICAL_TYPE => Type::Primitive(PrimitiveType::Uuid),
542                        ty => {
543                            return Err(Error::new(
544                                ErrorKind::FeatureUnsupported,
545                                format!(
546                                    "Logical type {ty} is not support in iceberg primitive type.",
547                                ),
548                            ));
549                        }
550                    }
551                } else {
552                    Type::Primitive(PrimitiveType::Fixed(fixed.size as u64))
553                }
554            }
555            AvroSchema::Bytes => Type::Primitive(PrimitiveType::Binary),
556            AvroSchema::Null => return Ok(None),
557            _ => {
558                return Err(Error::new(
559                    ErrorKind::Unexpected,
560                    "Unable to convert avro {schema} to iceberg primitive type.",
561                ));
562            }
563        };
564
565        Ok(Some(schema_type))
566    }
567
568    fn map_array(
569        &mut self,
570        array: &RecordSchema,
571        key: Option<Type>,
572        value: Option<Type>,
573    ) -> Result<Self::T> {
574        let key = key.ok_or_else(|| {
575            Error::new(
576                ErrorKind::DataInvalid,
577                "Can't convert avro map schema, missing key schema.",
578            )
579        })?;
580        let value = value.ok_or_else(|| {
581            Error::new(
582                ErrorKind::DataInvalid,
583                "Can't convert avro map schema, missing value schema.",
584            )
585        })?;
586        let key_id = Self::get_element_id_from_attributes(
587            &array.fields[0].custom_attributes,
588            FILED_ID_PROP,
589        )?;
590        let value_id = Self::get_element_id_from_attributes(
591            &array.fields[1].custom_attributes,
592            FILED_ID_PROP,
593        )?;
594        let key_field = NestedField::map_key_element(key_id, key);
595        let value_field = NestedField::map_value_element(
596            value_id,
597            value,
598            !is_avro_optional(&array.fields[1].schema),
599        );
600        Ok(Some(Type::Map(MapType {
601            key_field: key_field.into(),
602            value_field: value_field.into(),
603        })))
604    }
605}

607// # TODO
608// Fix this when we have used `avro_schema_to_schema` inner.
609#[allow(unused)]
610/// Converts avro schema to iceberg schema.
611pub(crate) fn avro_schema_to_schema(avro_schema: &AvroSchema) -> Result<Schema> {
612    if let AvroSchema::Record(_) = avro_schema {
613        let mut converter = AvroSchemaToSchema;
614        let schema_type =
615            visit(avro_schema, &mut converter)?.expect("Iceberg schema should not be none.");
616        if let Type::Struct(s) = schema_type {
617            Schema::builder()
618                .with_fields(s.fields().iter().cloned())
619                .build()
620        } else {
621            Err(Error::new(
622                ErrorKind::Unexpected,
623                format!("Expected to convert avro record schema to struct type, but {schema_type}"),
624            ))
625        }
626    } else {
627        Err(Error::new(
628            ErrorKind::DataInvalid,
629            "Can't convert non record avro schema to iceberg schema: {avro_schema}",
630        ))
631    }
632}

657    /// Help function to check schema conversion between avro and iceberg:
658    /// 1. avro to iceberg
659    /// 2. iceberg to avro
660    /// 3. iceberg to avro to iceberg back
661    fn check_schema_conversion(avro_schema: AvroSchema, iceberg_schema: Schema) {
