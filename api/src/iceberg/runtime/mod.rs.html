18// This module contains the async runtime abstraction for iceberg.
19
20use std::future::Future;
21use std::pin::Pin;
22use std::task::{Context, Poll};
23
24pub enum JoinHandle<T> {
25    #[cfg(feature = "tokio")]
26    Tokio(tokio::task::JoinHandle<T>),
27    #[cfg(all(feature = "async-std", not(feature = "tokio")))]
28    AsyncStd(async_std::task::JoinHandle<T>),
29    #[cfg(all(not(feature = "async-std"), not(feature = "tokio")))]
30    Unimplemented(Box<T>),
31}
32
33impl<T: Send + 'static> Future for JoinHandle<T> {
34    type Output = T;
35
36    fn poll(self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Self::Output> {
37        match self.get_mut() {
38            #[cfg(feature = "tokio")]
39            JoinHandle::Tokio(handle) => Pin::new(handle)
40                .poll(cx)
41                .map(|h| h.expect("tokio spawned task failed")),
42            #[cfg(all(feature = "async-std", not(feature = "tokio")))]
43            JoinHandle::AsyncStd(handle) => Pin::new(handle).poll(cx),
44            #[cfg(all(not(feature = "async-std"), not(feature = "tokio")))]
45            JoinHandle::Unimplemented(_) => unimplemented!("no runtime has been enabled"),
46        }
47    }
48}
49
50#[allow(dead_code)]
51pub fn spawn<F>(f: F) -> JoinHandle<F::Output>
52where
53    F: Future + Send + 'static,
54    F::Output: Send + 'static,
55{
56    #[cfg(feature = "tokio")]
57    return JoinHandle::Tokio(tokio::task::spawn(f));
58
59    #[cfg(all(feature = "async-std", not(feature = "tokio")))]
60    return JoinHandle::AsyncStd(async_std::task::spawn(f));
61
62    #[cfg(all(not(feature = "async-std"), not(feature = "tokio")))]
63    unimplemented!("no runtime has been enabled")
64}
65
66#[allow(dead_code)]
67pub fn spawn_blocking<F, T>(f: F) -> JoinHandle<T>
68where
69    F: FnOnce() -> T + Send + 'static,
70    T: Send + 'static,
71{
72    #[cfg(feature = "tokio")]
73    return JoinHandle::Tokio(tokio::task::spawn_blocking(f));
74
75    #[cfg(all(feature = "async-std", not(feature = "tokio")))]
76    return JoinHandle::AsyncStd(async_std::task::spawn_blocking(f));
77
78    #[cfg(all(not(feature = "async-std"), not(feature = "tokio")))]
79    unimplemented!("no runtime has been enabled")
80}

82#[cfg(test)]
83mod tests {
84    use super::*;
85
86    #[cfg(feature = "tokio")]
87    #[tokio::test]
88    async fn test_tokio_spawn() {
89        let handle = spawn(async { 1 + 1 });
90        assert_eq!(handle.await, 2);
91    }
92
93    #[cfg(feature = "tokio")]
94    #[tokio::test]
95    async fn test_tokio_spawn_blocking() {
96        let handle = spawn_blocking(|| 1 + 1);
97        assert_eq!(handle.await, 2);
98    }
99
100    #[cfg(all(feature = "async-std", not(feature = "tokio")))]
101    #[async_std::test]
102    async fn test_async_std_spawn() {
103        let handle = spawn(async { 1 + 1 });
104        assert_eq!(handle.await, 2);
105    }
106
107    #[cfg(all(feature = "async-std", not(feature = "tokio")))]
108    #[async_std::test]
109    async fn test_async_std_spawn_blocking() {
110        let handle = spawn_blocking(|| 1 + 1);

112    }
