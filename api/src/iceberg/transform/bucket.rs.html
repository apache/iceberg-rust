26#[derive(Debug)]
27pub struct Bucket {
28    mod_n: u32,
29}
30
31impl Bucket {
32    pub fn new(mod_n: u32) -> Self {
33        Self { mod_n }
34    }
35}
36
37impl Bucket {
38    /// When switch the hash function, we only need to change this function.
39    #[inline]
40    fn hash_bytes(mut v: &[u8]) -> i32 {
41        murmur3::murmur3_32(&mut v, 0).unwrap() as i32
42    }
43
44    #[inline]
45    fn hash_int(v: i32) -> i32 {
46        Self::hash_long(v as i64)
47    }
48
49    #[inline]
50    fn hash_long(v: i64) -> i32 {
51        Self::hash_bytes(v.to_le_bytes().as_slice())
52    }
53
54    /// v is days from unix epoch
55    #[inline]
56    fn hash_date(v: i32) -> i32 {
57        Self::hash_int(v)
58    }
59
60    /// v is microseconds from midnight
61    #[inline]
62    fn hash_time(v: i64) -> i32 {
63        Self::hash_long(v)
64    }
65
66    /// v is microseconds from unix epoch
67    #[inline]
68    fn hash_timestamp(v: i64) -> i32 {
69        Self::hash_long(v)
70    }
71
72    #[inline]
73    fn hash_str(s: &str) -> i32 {
74        Self::hash_bytes(s.as_bytes())
75    }
76
77    /// Decimal values are hashed using the minimum number of bytes required to hold the unscaled value as a twoâ€™s complement big-endian
78    /// ref: https://iceberg.apache.org/spec/#appendix-b-32-bit-hash-requirements
79    #[inline]
80    fn hash_decimal(v: i128) -> i32 {
81        let bytes = v.to_be_bytes();
82        if let Some(start) = bytes.iter().position(|&x| x != 0) {
83            Self::hash_bytes(&bytes[start..])
84        } else {
85            Self::hash_bytes(&[0])
86        }
87    }
88
89    /// def bucket_N(x) = (murmur3_x86_32_hash(x) & Integer.MAX_VALUE) % N
90    /// ref: https://iceberg.apache.org/spec/#partitioning
91    #[inline]
92    fn bucket_n(&self, v: i32) -> i32 {
93        (v & i32::MAX) % (self.mod_n as i32)
94    }
95
96    #[inline]
97    fn bucket_int(&self, v: i32) -> i32 {
98        self.bucket_n(Self::hash_int(v))
99    }
100
101    #[inline]
102    fn bucket_long(&self, v: i64) -> i32 {
103        self.bucket_n(Self::hash_long(v))
104    }
105
106    #[inline]
107    fn bucket_decimal(&self, v: i128) -> i32 {
108        self.bucket_n(Self::hash_decimal(v))
109    }
110
111    #[inline]
112    fn bucket_date(&self, v: i32) -> i32 {
113        self.bucket_n(Self::hash_date(v))
114    }
115
116    #[inline]
117    fn bucket_time(&self, v: i64) -> i32 {
118        self.bucket_n(Self::hash_time(v))
119    }
120
121    #[inline]
122    fn bucket_timestamp(&self, v: i64) -> i32 {
123        self.bucket_n(Self::hash_timestamp(v))
124    }
125
126    #[inline]
127    fn bucket_str(&self, v: &str) -> i32 {
128        self.bucket_n(Self::hash_str(v))
129    }
130
131    #[inline]
132    fn bucket_bytes(&self, v: &[u8]) -> i32 {
133        self.bucket_n(Self::hash_bytes(v))
134    }
135}

137impl TransformFunction for Bucket {
138    fn transform(&self, input: ArrayRef) -> crate::Result<ArrayRef> {
139        let res: arrow_array::Int32Array = match input.data_type() {
140            DataType::Int32 => input
141                .as_any()
142                .downcast_ref::<arrow_array::Int32Array>()
143                .unwrap()
144                .unary(|v| self.bucket_int(v)),
145            DataType::Int64 => input
146                .as_any()
147                .downcast_ref::<arrow_array::Int64Array>()
148                .unwrap()
149                .unary(|v| self.bucket_long(v)),
150            DataType::Decimal128(_, _) => input
151                .as_any()
152                .downcast_ref::<arrow_array::Decimal128Array>()
153                .unwrap()
154                .unary(|v| self.bucket_decimal(v)),
155            DataType::Date32 => input
156                .as_any()
157                .downcast_ref::<arrow_array::Date32Array>()
158                .unwrap()
159                .unary(|v| self.bucket_date(v)),
160            DataType::Time64(TimeUnit::Microsecond) => input
161                .as_any()
162                .downcast_ref::<arrow_array::Time64MicrosecondArray>()
163                .unwrap()
164                .unary(|v| self.bucket_time(v)),
165            DataType::Timestamp(TimeUnit::Microsecond, _) => input
166                .as_any()
167                .downcast_ref::<arrow_array::TimestampMicrosecondArray>()
168                .unwrap()
169                .unary(|v| self.bucket_timestamp(v)),
170            DataType::Time64(TimeUnit::Nanosecond) => input
171                .as_any()
172                .downcast_ref::<arrow_array::Time64NanosecondArray>()
173                .unwrap()
174                .unary(|v| self.bucket_time(v / 1000)),
175            DataType::Timestamp(TimeUnit::Nanosecond, _) => input
176                .as_any()
177                .downcast_ref::<arrow_array::TimestampNanosecondArray>()
178                .unwrap()
179                .unary(|v| self.bucket_timestamp(v / 1000)),
180            DataType::Utf8 => arrow_array::Int32Array::from_iter(
181                input
182                    .as_any()
183                    .downcast_ref::<arrow_array::StringArray>()
184                    .unwrap()
185                    .iter()
186                    .map(|v| v.map(|v| self.bucket_str(v))),
187            ),
188            DataType::LargeUtf8 => arrow_array::Int32Array::from_iter(
189                input
190                    .as_any()
191                    .downcast_ref::<arrow_array::LargeStringArray>()
192                    .unwrap()
193                    .iter()
194                    .map(|v| v.map(|v| self.bucket_str(v))),
195            ),
196            DataType::Binary => arrow_array::Int32Array::from_iter(
197                input
198                    .as_any()
199                    .downcast_ref::<arrow_array::BinaryArray>()
200                    .unwrap()
201                    .iter()
202                    .map(|v| v.map(|v| self.bucket_bytes(v))),
203            ),
204            DataType::LargeBinary => arrow_array::Int32Array::from_iter(
205                input
206                    .as_any()
207                    .downcast_ref::<arrow_array::LargeBinaryArray>()
208                    .unwrap()
209                    .iter()
210                    .map(|v| v.map(|v| self.bucket_bytes(v))),
211            ),
212            DataType::FixedSizeBinary(_) => arrow_array::Int32Array::from_iter(
213                input
214                    .as_any()
215                    .downcast_ref::<arrow_array::FixedSizeBinaryArray>()
216                    .unwrap()
217                    .iter()
218                    .map(|v| v.map(|v| self.bucket_bytes(v))),
219            ),
220            _ => {
221                return Err(crate::Error::new(
222                    crate::ErrorKind::FeatureUnsupported,
223                    format!(
224                        "Unsupported data type for bucket transform: {:?}",
225                        input.data_type()
226                    ),
227                ));
228            }
229        };
230        Ok(Arc::new(res))
231    }
232
233    fn transform_literal(&self, input: &Datum) -> crate::Result<Option<Datum>> {
234        let val = match (input.data_type(), input.literal()) {
235            (PrimitiveType::Int, PrimitiveLiteral::Int(v)) => self.bucket_int(*v),
236            (PrimitiveType::Long, PrimitiveLiteral::Long(v)) => self.bucket_long(*v),
237            (PrimitiveType::Decimal { .. }, PrimitiveLiteral::Int128(v)) => self.bucket_decimal(*v),
238            (PrimitiveType::Date, PrimitiveLiteral::Int(v)) => self.bucket_date(*v),
239            (PrimitiveType::Time, PrimitiveLiteral::Long(v)) => self.bucket_time(*v),
240            (PrimitiveType::Timestamp, PrimitiveLiteral::Long(v)) => self.bucket_timestamp(*v),
241            (PrimitiveType::Timestamptz, PrimitiveLiteral::Long(v)) => self.bucket_timestamp(*v),
242            (PrimitiveType::TimestampNs, PrimitiveLiteral::Long(v)) => {
243                self.bucket_timestamp(*v / 1000)
244            }
245            (PrimitiveType::TimestamptzNs, PrimitiveLiteral::Long(v)) => {
246                self.bucket_timestamp(*v / 1000)
247            }
248            (PrimitiveType::String, PrimitiveLiteral::String(v)) => self.bucket_str(v.as_str()),
249            (PrimitiveType::Uuid, PrimitiveLiteral::UInt128(v)) => {
250                self.bucket_bytes(uuid::Uuid::from_u128(*v).as_ref())
251            }
252            (PrimitiveType::Binary, PrimitiveLiteral::Binary(v)) => self.bucket_bytes(v.as_ref()),
253            (PrimitiveType::Fixed(_), PrimitiveLiteral::Binary(v)) => self.bucket_bytes(v.as_ref()),
254            _ => {
255                return Err(crate::Error::new(
256                    crate::ErrorKind::FeatureUnsupported,
257                    format!(
258                        "Unsupported data type for bucket transform: {:?}",
259                        input.data_type()
260                    ),
261                ));
262            }
263        };
264        Ok(Some(Datum::int(val)))
265    }
266}

948    fn test_transform_timestamp_nanos_and_micros_array_equivalence() {
949        let bucket = Bucket::new(100);
950        let micros_value = 1510871468000000;
951        let nanos_value = micros_value * 1000;
952
953        let micro_array = TimestampMicrosecondArray::from_iter_values(vec![micros_value]);
954        let nano_array = TimestampNanosecondArray::from_iter_values(vec![nanos_value]);
955
956        let transformed_micro: ArrayRef = bucket.transform(Arc::new(micro_array)).unwrap();
957        let transformed_nano: ArrayRef = bucket.transform(Arc::new(nano_array)).unwrap();
958
959        let micro_result = transformed_micro
960            .as_any()
961            .downcast_ref::<Int32Array>()
962            .unwrap();
963        let nano_result = transformed_nano
964            .as_any()
965            .downcast_ref::<Int32Array>()
966            .unwrap();
967
968        assert_eq!(micro_result.value(0), nano_result.value(0));
