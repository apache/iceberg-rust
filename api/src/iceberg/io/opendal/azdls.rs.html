33/// Parses adls.* prefixed configuration properties.
34pub(crate) fn azdls_config_parse(mut properties: HashMap<String, String>) -> Result<AzdlsConfig> {
35    let mut config = AzdlsConfig::default();
36
37    if let Some(_conn_str) = properties.remove(ADLS_CONNECTION_STRING) {
38        return Err(Error::new(
39            ErrorKind::FeatureUnsupported,
40            "Azdls: connection string currently not supported",
41        ));
42    }
43
44    if let Some(account_name) = properties.remove(ADLS_ACCOUNT_NAME) {
45        config.account_name = Some(account_name);
46    }
47
48    if let Some(account_key) = properties.remove(ADLS_ACCOUNT_KEY) {
49        config.account_key = Some(account_key);
50    }
51
52    if let Some(sas_token) = properties.remove(ADLS_SAS_TOKEN) {
53        config.sas_token = Some(sas_token);
54    }
55
56    if let Some(tenant_id) = properties.remove(ADLS_TENANT_ID) {
57        config.tenant_id = Some(tenant_id);
58    }
59
60    if let Some(client_id) = properties.remove(ADLS_CLIENT_ID) {
61        config.client_id = Some(client_id);
62    }
63
64    if let Some(client_secret) = properties.remove(ADLS_CLIENT_SECRET) {
65        config.client_secret = Some(client_secret);
66    }
67
68    if let Some(authority_host) = properties.remove(ADLS_AUTHORITY_HOST) {
69        config.authority_host = Some(authority_host);
70    }
71
72    Ok(config)
73}

75/// Builds an OpenDAL operator from the AzdlsConfig and path.
76///
77/// The path is expected to include the scheme in a format like:
78/// `abfss://<myfs>@<myaccount>.dfs.core.windows.net/mydir/myfile.parquet`.
79pub(crate) fn azdls_create_operator<'a>(
80    absolute_path: &'a str,
81    config: &AzdlsConfig,
82    configured_scheme: &AzureStorageScheme,
83) -> Result<(opendal::Operator, &'a str)> {
84    let path = absolute_path.parse::<AzureStoragePath>()?;
85    match_path_with_config(&path, config, configured_scheme)?;
86
87    let op = azdls_config_build(config, &path)?;
88
89    // Paths to files in ADLS tend to be written in fully qualified form,
90    // including their filesystem and account name.
91    // OpenDAL's operator methods expect only the relative path, so we split it
92    // off and save it for later use.
93    let relative_path_len = path.path.len();
94    let (_, relative_path) = absolute_path.split_at(absolute_path.len() - relative_path_len);
95
96    Ok((op, relative_path))
97}

99/// Note that `abf[s]` and `wasb[s]` variants have different implications:
100/// - `abfs[s]` is used to refer to files in ADLS Gen2, backed by blob storage;
101///   paths are expected to contain the `dfs` storage service.
102/// - `wasb[s]` is used to refer to files in Blob Storage directly; paths are
103///   expected to contain the `blob` storage service.
104#[derive(Clone, Debug, PartialEq, Serialize, Deserialize)]
105pub enum AzureStorageScheme {
106    Abfs,
107    Abfss,
108    Wasb,
109    Wasbs,
110}
111
112impl AzureStorageScheme {
113    // Returns the respective encrypted or plain-text HTTP scheme.
114    pub fn as_http_scheme(&self) -> &str {
115        match self {
116            AzureStorageScheme::Abfs | AzureStorageScheme::Wasb => "http",
117            AzureStorageScheme::Abfss | AzureStorageScheme::Wasbs => "https",
118        }
119    }
120}
121
122impl Display for AzureStorageScheme {
123    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
124        match self {
125            AzureStorageScheme::Abfs => write!(f, "abfs"),
126            AzureStorageScheme::Abfss => write!(f, "abfss"),
127            AzureStorageScheme::Wasb => write!(f, "wasb"),
128            AzureStorageScheme::Wasbs => write!(f, "wasbs"),
129        }
130    }
131}
132
133impl FromStr for AzureStorageScheme {
134    type Err = Error;
135
136    fn from_str(s: &str) -> Result<Self> {
137        match s {
138            "abfs" => Ok(AzureStorageScheme::Abfs),
139            "abfss" => Ok(AzureStorageScheme::Abfss),
140            "wasb" => Ok(AzureStorageScheme::Wasb),
141            "wasbs" => Ok(AzureStorageScheme::Wasbs),
142            _ => Err(Error::new(
143                ErrorKind::DataInvalid,
144                format!("Unexpected Azure Storage scheme: {s}"),
145            )),
146        }
147    }
148}

150/// Validates whether the given path matches what's configured for the backend.
151fn match_path_with_config(
152    path: &AzureStoragePath,
153    config: &AzdlsConfig,
154    configured_scheme: &AzureStorageScheme,
155) -> Result<()> {
156    ensure_data_valid!(
157        &path.scheme == configured_scheme,
158        "Storage::Azdls: Scheme mismatch: configured {}, passed {}",
159        configured_scheme,
160        path.scheme
161    );
162
163    if let Some(ref configured_account_name) = config.account_name {
164        ensure_data_valid!(
165            &path.account_name == configured_account_name,
166            "Storage::Azdls: Account name mismatch: configured {}, path {}",
167            configured_account_name,
168            path.account_name
169        );
170    }
171
172    if let Some(ref configured_endpoint) = config.endpoint {
173        let passed_http_scheme = path.scheme.as_http_scheme();
174        ensure_data_valid!(
175            configured_endpoint.starts_with(passed_http_scheme),
176            "Storage::Azdls: Endpoint {} does not use the expected http scheme {}.",
177            configured_endpoint,
178            passed_http_scheme
179        );
180
181        let ends_with_expected_suffix = configured_endpoint
182            .trim_end_matches('/')
183            .ends_with(&path.endpoint_suffix);
184        ensure_data_valid!(
185            ends_with_expected_suffix,
186            "Storage::Azdls: Endpoint suffix {} used with configured endpoint {}.",
187            path.endpoint_suffix,
188            configured_endpoint,
189        );
190    }
191
192    Ok(())
193}

195fn azdls_config_build(config: &AzdlsConfig, path: &AzureStoragePath) -> Result<opendal::Operator> {
196    let mut builder = config.clone().into_builder();
197
198    if config.endpoint.is_none() {
199        // If no endpoint is provided, we construct it from the fully-qualified path.
200        builder = builder.endpoint(&path.as_endpoint());
201    }
202    builder = builder.filesystem(&path.filesystem);
203
204    Ok(opendal::Operator::new(builder)?.finish())
205}

207/// Represents a fully qualified path to blob/ file in Azure Storage.
208#[derive(Debug, PartialEq)]
209struct AzureStoragePath {
210    /// The scheme of the URL, e.g., `abfss`, `abfs`, `wasbs`, or `wasb`.
211    scheme: AzureStorageScheme,
212
213    /// Under Blob Storage, this is considered the _container_.
214    filesystem: String,
215
216    account_name: String,
217
218    /// The endpoint suffix, e.g., `core.windows.net` for the public cloud
219    /// endpoint.
220    endpoint_suffix: String,
221
222    /// Path to the file.
223    ///
224    /// It is relative to the `root` of the `AzdlsConfig`.
225    path: String,
226}
227
228impl AzureStoragePath {
229    /// Converts the AzureStoragePath into a full endpoint URL.
230    ///
231    /// This is possible because the path is fully qualified.
232    fn as_endpoint(&self) -> String {
233        format!(
234            "{}://{}.dfs.{}",
235            self.scheme.as_http_scheme(),
236            self.account_name,
237            self.endpoint_suffix
238        )
239    }
240}
241
242impl FromStr for AzureStoragePath {
243    type Err = Error;
244
245    fn from_str(path: &str) -> Result<Self> {
246        let url = Url::parse(path)?;
247
248        let filesystem = url.username();
249        ensure_data_valid!(
250            !filesystem.is_empty(),
251            "AzureStoragePath: No container or filesystem name in path: {}",
252            path
253        );
254
255        let (account_name, storage_service, endpoint_suffix) = parse_azure_storage_endpoint(&url)?;
256        let scheme = validate_storage_and_scheme(storage_service, url.scheme())?;
257
258        Ok(AzureStoragePath {
259            scheme,
260            filesystem: filesystem.to_string(),
261            account_name: account_name.to_string(),
262            endpoint_suffix: endpoint_suffix.to_string(),
263            path: url.path().to_string(),
264        })
265    }
266}
267
268fn parse_azure_storage_endpoint(url: &Url) -> Result<(&str, &str, &str)> {
269    let host = url.host_str().ok_or(Error::new(
270        ErrorKind::DataInvalid,
271        "AzureStoragePath: No host",
272    ))?;
273
274    let (account_name, endpoint) = host.split_once('.').ok_or(Error::new(
275        ErrorKind::DataInvalid,
276        "AzureStoragePath: No account name",
277    ))?;
278    if account_name.is_empty() {
279        return Err(Error::new(
280            ErrorKind::DataInvalid,
281            "AzureStoragePath: No account name",
282        ));
283    }
284
285    let (storage, endpoint_suffix) = endpoint.split_once('.').ok_or(Error::new(
286        ErrorKind::DataInvalid,
287        "AzureStoragePath: No storage service",
288    ))?;
289
290    Ok((account_name, storage, endpoint_suffix))
291}
292
293fn validate_storage_and_scheme(
294    storage_service: &str,
295    scheme_str: &str,
296) -> Result<AzureStorageScheme> {
297    let scheme = scheme_str.parse::<AzureStorageScheme>()?;
298    match scheme {
299        AzureStorageScheme::Abfss | AzureStorageScheme::Abfs => {
300            ensure_data_valid!(
301                storage_service == "dfs",
302                "AzureStoragePath: Unexpected storage service for abfs[s]: {}",
303                storage_service
304            );
305            Ok(scheme)
306        }
307        AzureStorageScheme::Wasbs | AzureStorageScheme::Wasb => {
308            ensure_data_valid!(
309                storage_service == "blob",
310                "AzureStoragePath: Unexpected storage service for wasb[s]: {}",
311                storage_service
312            );
313            Ok(scheme)
314        }
315    }
316}
