18use std::sync::Arc;
19
20use opendal::layers::RetryLayer;
21#[cfg(feature = "storage-gcs")]
22use opendal::services::GcsConfig;
23#[cfg(feature = "storage-oss")]
24use opendal::services::OssConfig;
25#[cfg(feature = "storage-s3")]
26use opendal::services::S3Config;
27use opendal::{Operator, Scheme};
28
29use super::FileIOBuilder;
30use crate::{Error, ErrorKind};
31
32/// The storage carries all supported storage services in iceberg
33#[derive(Debug)]
34pub(crate) enum Storage {
35    #[cfg(feature = "storage-memory")]
36    Memory(Operator),
37    #[cfg(feature = "storage-fs")]
38    LocalFs,
39    #[cfg(feature = "storage-s3")]
40    S3 {
41        /// s3 storage could have `s3://` and `s3a://`.
42        /// Storing the scheme string here to return the correct path.
43        scheme_str: String,
44        config: Arc<S3Config>,
45    },
46    #[cfg(feature = "storage-oss")]
47    Oss { config: Arc<OssConfig> },
48    #[cfg(feature = "storage-gcs")]
49    Gcs { config: Arc<GcsConfig> },
50}
51
52impl Storage {
53    /// Convert iceberg config to opendal config.
54    pub(crate) fn build(file_io_builder: FileIOBuilder) -> crate::Result<Self> {
55        let (scheme_str, props) = file_io_builder.into_parts();
56        let scheme = Self::parse_scheme(&scheme_str)?;
57
58        match scheme {
59            #[cfg(feature = "storage-memory")]
60            Scheme::Memory => Ok(Self::Memory(super::memory_config_build()?)),
61            #[cfg(feature = "storage-fs")]
62            Scheme::Fs => Ok(Self::LocalFs),
63            #[cfg(feature = "storage-s3")]
64            Scheme::S3 => Ok(Self::S3 {
65                scheme_str,
66                config: super::s3_config_parse(props)?.into(),
67            }),
68            #[cfg(feature = "storage-gcs")]
69            Scheme::Gcs => Ok(Self::Gcs {
70                config: super::gcs_config_parse(props)?.into(),
71            }),
72            #[cfg(feature = "storage-oss")]
73            Scheme::Oss => Ok(Self::Oss {
74                config: super::oss_config_parse(props)?.into(),
75            }),
76            // Update doc on [`FileIO`] when adding new schemes.
77            _ => Err(Error::new(
78                ErrorKind::FeatureUnsupported,
79                format!("Constructing file io from scheme: {scheme} not supported now",),
80            )),
81        }
82    }
83
84    /// Creates operator from path.
85    ///
86    /// # Arguments
87    ///
88    /// * path: It should be *absolute* path starting with scheme string used to construct [`FileIO`].
89    ///
90    /// # Returns
91    ///
92    /// The return value consists of two parts:
93    ///
94    /// * An [`opendal::Operator`] instance used to operate on file.
95    /// * Relative path to the root uri of [`opendal::Operator`].
96    pub(crate) fn create_operator<'a>(
97        &self,
98        path: &'a impl AsRef<str>,
99    ) -> crate::Result<(Operator, &'a str)> {
100        let path = path.as_ref();
101        let (operator, relative_path): (Operator, &str) = match self {
102            #[cfg(feature = "storage-memory")]
103            Storage::Memory(op) => {
104                if let Some(stripped) = path.strip_prefix("memory:/") {
105                    Ok::<_, crate::Error>((op.clone(), stripped))
106                } else {
107                    Ok::<_, crate::Error>((op.clone(), &path[1..]))
108                }
109            }
110            #[cfg(feature = "storage-fs")]
111            Storage::LocalFs => {
112                let op = super::fs_config_build()?;
113
114                if let Some(stripped) = path.strip_prefix("file:/") {
115                    Ok::<_, crate::Error>((op, stripped))
116                } else {
117                    Ok::<_, crate::Error>((op, &path[1..]))
118                }
119            }
120            #[cfg(feature = "storage-s3")]
121            Storage::S3 { scheme_str, config } => {
122                let op = super::s3_config_build(config, path)?;
123                let op_info = op.info();
124
125                // Check prefix of s3 path.
126                let prefix = format!("{}://{}/", scheme_str, op_info.name());
127                if path.starts_with(&prefix) {
128                    Ok((op, &path[prefix.len()..]))
129                } else {
130                    Err(Error::new(
131                        ErrorKind::DataInvalid,
132                        format!("Invalid s3 url: {}, should start with {}", path, prefix),
133                    ))
134                }
135            }
136
137            #[cfg(feature = "storage-gcs")]
138            Storage::Gcs { config } => {
139                let operator = super::gcs_config_build(config, path)?;
140                let prefix = format!("gs://{}/", operator.info().name());
141                if path.starts_with(&prefix) {
142                    Ok((operator, &path[prefix.len()..]))
143                } else {
144                    Err(Error::new(
145                        ErrorKind::DataInvalid,
146                        format!("Invalid gcs url: {}, should start with {}", path, prefix),
147                    ))
148                }
149            }
150            #[cfg(feature = "storage-oss")]
151            Storage::Oss { config } => {
152                let op = super::oss_config_build(config, path)?;
153
154                // Check prefix of oss path.
155                let prefix = format!("oss://{}/", op.info().name());
156                if path.starts_with(&prefix) {
157                    Ok((op, &path[prefix.len()..]))
158                } else {
159                    Err(Error::new(
160                        ErrorKind::DataInvalid,
161                        format!("Invalid oss url: {}, should start with {}", path, prefix),
162                    ))
163                }
164            }
165            #[cfg(all(
166                not(feature = "storage-s3"),
167                not(feature = "storage-fs"),
168                not(feature = "storage-gcs"),
169                not(feature = "storage-oss")
170            ))]
171            _ => Err(Error::new(
172                ErrorKind::FeatureUnsupported,
173                "No storage service has been enabled",
174            )),
175        }?;
176
177        // Transient errors are common for object stores; however there's no
178        // harm in retrying temporary failures for other storage backends as well.
179        let operator = operator.layer(RetryLayer::new());
180
181        Ok((operator, relative_path))
182    }
183
184    /// Parse scheme.
185    fn parse_scheme(scheme: &str) -> crate::Result<Scheme> {
186        match scheme {
187            "memory" => Ok(Scheme::Memory),
188            "file" | "" => Ok(Scheme::Fs),
189            "s3" | "s3a" => Ok(Scheme::S3),
190            "gs" | "gcs" => Ok(Scheme::Gcs),
191            "oss" => Ok(Scheme::Oss),
192            s => Ok(s.parse::<Scheme>()?),
193        }
194    }
195}
