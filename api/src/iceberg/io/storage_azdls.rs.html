28/// A connection string.
29///
30/// Note, this string is parsed first, and any other passed adls.* properties
31/// will override values from the connection string.
32const ADLS_CONNECTION_STRING: &str = "adls.connection-string";
33
34/// The account that you want to connect to.
35pub const ADLS_ACCOUNT_NAME: &str = "adls.account-name";
36
37/// The key to authentication against the account.
38pub const ADLS_ACCOUNT_KEY: &str = "adls.account-key";
39
40/// The shared access signature.
41pub const ADLS_SAS_TOKEN: &str = "adls.sas-token";
42
43/// The tenant-id.
44pub const ADLS_TENANT_ID: &str = "adls.tenant-id";
45
46/// The client-id.
47pub const ADLS_CLIENT_ID: &str = "adls.client-id";
48
49/// The client-secret.
50pub const ADLS_CLIENT_SECRET: &str = "adls.client-secret";
51
52/// Parses adls.* prefixed configuration properties.
53pub(crate) fn azdls_config_parse(mut properties: HashMap<String, String>) -> Result<AzdlsConfig> {
54    let mut config = AzdlsConfig::default();
55
56    if let Some(_conn_str) = properties.remove(ADLS_CONNECTION_STRING) {
57        return Err(Error::new(
58            ErrorKind::FeatureUnsupported,
59            "Azdls: connection string currently not supported",
60        ));
61    }
62
63    if let Some(account_name) = properties.remove(ADLS_ACCOUNT_NAME) {
64        config.account_name = Some(account_name);
65    }
66
67    if let Some(account_key) = properties.remove(ADLS_ACCOUNT_KEY) {
68        config.account_key = Some(account_key);
69    }
70
71    if let Some(sas_token) = properties.remove(ADLS_SAS_TOKEN) {
72        config.sas_token = Some(sas_token);
73    }
74
75    if let Some(tenant_id) = properties.remove(ADLS_TENANT_ID) {
76        config.tenant_id = Some(tenant_id);
77    }
78
79    if let Some(client_id) = properties.remove(ADLS_CLIENT_ID) {
80        config.client_id = Some(client_id);
81    }
82
83    if let Some(client_secret) = properties.remove(ADLS_CLIENT_SECRET) {
84        config.client_secret = Some(client_secret);
85    }
86
87    Ok(config)
88}

90/// Builds an OpenDAL operator from the AzdlsConfig and path.
91///
92/// The path is expected to include the scheme in a format like:
93/// `abfss://<myfs>@<myaccount>.dfs.core.windows.net/mydir/myfile.parquet`.
94pub(crate) fn azdls_create_operator<'a>(
95    absolute_path: &'a str,
96    config: &AzdlsConfig,
97    configured_scheme: &AzureStorageScheme,
98) -> Result<(opendal::Operator, &'a str)> {
99    let path = absolute_path.parse::<AzureStoragePath>()?;
100    match_path_with_config(&path, config, configured_scheme)?;
101
102    let op = azdls_config_build(config, &path)?;
103
104    // Paths to files in ADLS tend to be written in fully qualified form,
105    // including their filesystem and account name.
106    // OpenDAL's operator methods expect only the relative path, so we split it
107    // off and save it for later use.
108    let relative_path_len = path.path.len();
109    let (_, relative_path) = absolute_path.split_at(absolute_path.len() - relative_path_len);
110
111    Ok((op, relative_path))
112}

114/// Note that `abf[s]` and `wasb[s]` variants have different implications:
115/// - `abfs[s]` is used to refer to files in ADLS Gen2, backed by blob storage;
116///   paths are expected to contain the `dfs` storage service.
117/// - `wasb[s]` is used to refer to files in Blob Storage directly; paths are
118///   expected to contain the `blob` storage service.
119#[derive(Debug, PartialEq)]
120pub(crate) enum AzureStorageScheme {
121    Abfs,
122    Abfss,
123    Wasb,
124    Wasbs,
125}
126
127impl AzureStorageScheme {
128    // Returns the respective encrypted or plain-text HTTP scheme.
129    pub fn as_http_scheme(&self) -> &str {
130        match self {
131            AzureStorageScheme::Abfs | AzureStorageScheme::Wasb => "http",
132            AzureStorageScheme::Abfss | AzureStorageScheme::Wasbs => "https",
133        }
134    }
135}
136
137impl Display for AzureStorageScheme {
138    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
139        match self {
140            AzureStorageScheme::Abfs => write!(f, "abfs"),
141            AzureStorageScheme::Abfss => write!(f, "abfss"),
142            AzureStorageScheme::Wasb => write!(f, "wasb"),
143            AzureStorageScheme::Wasbs => write!(f, "wasbs"),
144        }
145    }
146}
147
148impl FromStr for AzureStorageScheme {
149    type Err = Error;
150
151    fn from_str(s: &str) -> Result<Self> {
152        match s {
153            "abfs" => Ok(AzureStorageScheme::Abfs),
154            "abfss" => Ok(AzureStorageScheme::Abfss),
155            "wasb" => Ok(AzureStorageScheme::Wasb),
156            "wasbs" => Ok(AzureStorageScheme::Wasbs),
157            _ => Err(Error::new(
158                ErrorKind::DataInvalid,
159                format!("Unexpected Azure Storage scheme: {}", s),
160            )),
161        }
162    }
163}

165/// Validates whether the given path matches what's configured for the backend.
166fn match_path_with_config(
167    path: &AzureStoragePath,
168    config: &AzdlsConfig,
169    configured_scheme: &AzureStorageScheme,
170) -> Result<()> {
171    ensure_data_valid!(
172        &path.scheme == configured_scheme,
173        "Storage::Azdls: Scheme mismatch: configured {}, passed {}",
174        configured_scheme,
175        path.scheme
176    );
177
178    if let Some(ref configured_account_name) = config.account_name {
179        ensure_data_valid!(
180            &path.account_name == configured_account_name,
181            "Storage::Azdls: Account name mismatch: configured {}, path {}",
182            configured_account_name,
183            path.account_name
184        );
185    }
186
187    if let Some(ref configured_endpoint) = config.endpoint {
188        let passed_http_scheme = path.scheme.as_http_scheme();
189        ensure_data_valid!(
190            configured_endpoint.starts_with(passed_http_scheme),
191            "Storage::Azdls: Endpoint {} does not use the expected http scheme {}.",
192            configured_endpoint,
193            passed_http_scheme
194        );
195
196        let ends_with_expected_suffix = configured_endpoint
197            .trim_end_matches('/')
198            .ends_with(&path.endpoint_suffix);
199        ensure_data_valid!(
200            ends_with_expected_suffix,
201            "Storage::Azdls: Endpoint suffix {} used with configured endpoint {}.",
202            path.endpoint_suffix,
203            configured_endpoint,
204        );
205    }
206
207    Ok(())
208}

210fn azdls_config_build(config: &AzdlsConfig, path: &AzureStoragePath) -> Result<opendal::Operator> {
211    let mut builder = config.clone().into_builder();
212
213    if config.endpoint.is_none() {
214        // If no endpoint is provided, we construct it from the fully-qualified path.
215        builder = builder.endpoint(&path.as_endpoint());
216    }
217    builder = builder.filesystem(&path.filesystem);
218
219    Ok(opendal::Operator::new(builder)?.finish())
220}

222/// Represents a fully qualified path to blob/ file in Azure Storage.
223#[derive(Debug, PartialEq)]
224struct AzureStoragePath {
225    /// The scheme of the URL, e.g., `abfss`, `abfs`, `wasbs`, or `wasb`.
226    scheme: AzureStorageScheme,
227
228    /// Under Blob Storage, this is considered the _container_.
229    filesystem: String,
230
231    account_name: String,
232
233    /// The endpoint suffix, e.g., `core.windows.net` for the public cloud
234    /// endpoint.
235    endpoint_suffix: String,
236
237    /// Path to the file.
238    ///
239    /// It is relative to the `root` of the `AzdlsConfig`.
240    path: String,
241}
242
243impl AzureStoragePath {
244    /// Converts the AzureStoragePath into a full endpoint URL.
245    ///
246    /// This is possible because the path is fully qualified.
247    fn as_endpoint(&self) -> String {
248        format!(
249            "{}://{}.dfs.{}",
250            self.scheme.as_http_scheme(),
251            self.account_name,
252            self.endpoint_suffix
253        )
254    }
255}
256
257impl FromStr for AzureStoragePath {
258    type Err = Error;
259
260    fn from_str(path: &str) -> Result<Self> {
261        let url = Url::parse(path)?;
262
263        let filesystem = url.username();
264        ensure_data_valid!(
265            !filesystem.is_empty(),
266            "AzureStoragePath: No container or filesystem name in path: {}",
267            path
268        );
269
270        let (account_name, storage_service, endpoint_suffix) = parse_azure_storage_endpoint(&url)?;
271        let scheme = validate_storage_and_scheme(storage_service, url.scheme())?;
272
273        Ok(AzureStoragePath {
274            scheme,
275            filesystem: filesystem.to_string(),
276            account_name: account_name.to_string(),
277            endpoint_suffix: endpoint_suffix.to_string(),
278            path: url.path().to_string(),
279        })
280    }
281}
282
283fn parse_azure_storage_endpoint(url: &Url) -> Result<(&str, &str, &str)> {
284    let host = url.host_str().ok_or(Error::new(
285        ErrorKind::DataInvalid,
286        "AzureStoragePath: No host",
287    ))?;
288
289    let (account_name, endpoint) = host.split_once('.').ok_or(Error::new(
290        ErrorKind::DataInvalid,
291        "AzureStoragePath: No account name",
292    ))?;
293    if account_name.is_empty() {
294        return Err(Error::new(
295            ErrorKind::DataInvalid,
296            "AzureStoragePath: No account name",
297        ));
298    }
299
300    let (storage, endpoint_suffix) = endpoint.split_once('.').ok_or(Error::new(
301        ErrorKind::DataInvalid,
302        "AzureStoragePath: No storage service",
303    ))?;
304
305    Ok((account_name, storage, endpoint_suffix))
306}
307
308fn validate_storage_and_scheme(
309    storage_service: &str,
310    scheme_str: &str,
311) -> Result<AzureStorageScheme> {
312    let scheme = scheme_str.parse::<AzureStorageScheme>()?;
313    match scheme {
314        AzureStorageScheme::Abfss | AzureStorageScheme::Abfs => {
315            ensure_data_valid!(
316                storage_service == "dfs",
317                "AzureStoragePath: Unexpected storage service for abfs[s]: {}",
318                storage_service
319            );
320            Ok(scheme)
321        }
322        AzureStorageScheme::Wasbs | AzureStorageScheme::Wasb => {
323            ensure_data_valid!(
324                storage_service == "blob",
325                "AzureStoragePath: Unexpected storage service for wasb[s]: {}",
326                storage_service
327            );
328            Ok(scheme)
329        }
330    }
