29/// FileIO implementation, used to manipulate files in underlying storage.
30///
31/// # Note
32///
33/// All path passed to `FileIO` must be absolute path starting with scheme string used to construct `FileIO`.
34/// For example, if you construct `FileIO` with `s3a` scheme, then all path passed to `FileIO` must start with `s3a://`.
35///
36/// Supported storages:
37///
38/// | Storage            | Feature Flag     | Schemes    |
39/// |--------------------|-------------------|------------|
40/// | Local file system  | `storage-fs`      | `file`     |
41/// | Memory             | `storage-memory`  | `memory`   |
42/// | S3                 | `storage-s3`      | `s3`, `s3a`|
43/// | GCS                | `storage-gcs`     | `gs`, `gcs`|
44#[derive(Clone, Debug)]
45pub struct FileIO {
46    builder: FileIOBuilder,
47
48    inner: Arc<Storage>,
49}
50
51impl FileIO {
52    /// Convert FileIO into [`FileIOBuilder`] which used to build this FileIO.
53    ///
54    /// This function is useful when you want serialize and deserialize FileIO across
55    /// distributed systems.
56    pub fn into_builder(self) -> FileIOBuilder {
57        self.builder
58    }
59
60    /// Try to infer file io scheme from path. See [`FileIO`] for supported schemes.
61    ///
62    /// - If it's a valid url, for example `s3://bucket/a`, url scheme will be used, and the rest of the url will be ignored.
63    /// - If it's not a valid url, will try to detect if it's a file path.
64    ///
65    /// Otherwise will return parsing error.
66    pub fn from_path(path: impl AsRef<str>) -> crate::Result<FileIOBuilder> {
67        let url = Url::parse(path.as_ref())
68            .map_err(Error::from)
69            .or_else(|e| {
70                Url::from_file_path(path.as_ref()).map_err(|_| {
71                    Error::new(
72                        ErrorKind::DataInvalid,
73                        "Input is neither a valid url nor path",
74                    )
75                    .with_context("input", path.as_ref().to_string())
76                    .with_source(e)
77                })
78            })?;
79
80        Ok(FileIOBuilder::new(url.scheme()))
81    }
82
83    /// Deletes file.
84    ///
85    /// # Arguments
86    ///
87    /// * path: It should be *absolute* path starting with scheme string used to construct [`FileIO`].
88    pub async fn delete(&self, path: impl AsRef<str>) -> Result<()> {
89        let (op, relative_path) = self.inner.create_operator(&path)?;
90        Ok(op.delete(relative_path).await?)
91    }
92
93    /// Remove the path and all nested dirs and files recursively.
94    ///
95    /// # Arguments
96    ///
97    /// * path: It should be *absolute* path starting with scheme string used to construct [`FileIO`].
98    #[deprecated(note = "use remove_dir_all instead", since = "0.4.0")]
99    pub async fn remove_all(&self, path: impl AsRef<str>) -> Result<()> {
100        let (op, relative_path) = self.inner.create_operator(&path)?;
101        Ok(op.remove_all(relative_path).await?)
102    }
103
104    /// Remove the path and all nested dirs and files recursively.
105    ///
106    /// # Arguments
107    ///
108    /// * path: It should be *absolute* path starting with scheme string used to construct [`FileIO`].
109    ///
110    /// # Behavior
111    ///
112    /// - If the path is a file or not exist, this function will be no-op.
113    /// - If the path is a empty directory, this function will remove the directory itself.
114    /// - If the path is a non-empty directory, this function will remove the directory and all nested files and directories.
115    pub async fn remove_dir_all(&self, path: impl AsRef<str>) -> Result<()> {
116        let (op, relative_path) = self.inner.create_operator(&path)?;
117        let path = if relative_path.ends_with('/') {
118            relative_path.to_string()
119        } else {
120            format!("{relative_path}/")
121        };
122        Ok(op.remove_all(&path).await?)
123    }
124
125    /// Check file exists.
126    ///
127    /// # Arguments
128    ///
129    /// * path: It should be *absolute* path starting with scheme string used to construct [`FileIO`].
130    pub async fn exists(&self, path: impl AsRef<str>) -> Result<bool> {
131        let (op, relative_path) = self.inner.create_operator(&path)?;
132        Ok(op.exists(relative_path).await?)
133    }
134
135    /// Creates input file.
136    ///
137    /// # Arguments
138    ///
139    /// * path: It should be *absolute* path starting with scheme string used to construct [`FileIO`].
140    pub fn new_input(&self, path: impl AsRef<str>) -> Result<InputFile> {
141        let (op, relative_path) = self.inner.create_operator(&path)?;
142        let path = path.as_ref().to_string();
143        let relative_path_pos = path.len() - relative_path.len();
144        Ok(InputFile {
145            op,
146            path,
147            relative_path_pos,
148        })
149    }
150
151    /// Creates output file.
152    ///
153    /// # Arguments
154    ///
155    /// * path: It should be *absolute* path starting with scheme string used to construct [`FileIO`].
156    pub fn new_output(&self, path: impl AsRef<str>) -> Result<OutputFile> {
157        let (op, relative_path) = self.inner.create_operator(&path)?;
158        let path = path.as_ref().to_string();
159        let relative_path_pos = path.len() - relative_path.len();
160        Ok(OutputFile {
161            op,
162            path,
163            relative_path_pos,
164        })
165    }
166}

168/// Builder for [`FileIO`].
169#[derive(Clone, Debug)]
170pub struct FileIOBuilder {
171    /// This is used to infer scheme of operator.
172    ///
173    /// If this is `None`, then [`FileIOBuilder::build`](FileIOBuilder::build) will build a local file io.
174    scheme_str: Option<String>,
175    /// Arguments for operator.
176    props: HashMap<String, String>,
177}
178
179impl FileIOBuilder {
180    /// Creates a new builder with scheme.
181    /// See [`FileIO`] for supported schemes.
182    pub fn new(scheme_str: impl ToString) -> Self {
183        Self {
184            scheme_str: Some(scheme_str.to_string()),
185            props: HashMap::default(),
186        }
187    }
188
189    /// Creates a new builder for local file io.
190    pub fn new_fs_io() -> Self {
191        Self {
192            scheme_str: None,
193            props: HashMap::default(),
194        }
195    }
196
197    /// Fetch the scheme string.
198    ///
199    /// The scheme_str will be empty if it's None.
200    pub fn into_parts(self) -> (String, HashMap<String, String>) {
201        (self.scheme_str.unwrap_or_default(), self.props)
202    }
203
204    /// Add argument for operator.
205    pub fn with_prop(mut self, key: impl ToString, value: impl ToString) -> Self {
206        self.props.insert(key.to_string(), value.to_string());
207        self
208    }
209
210    /// Add argument for operator.
211    pub fn with_props(
212        mut self,
213        args: impl IntoIterator<Item = (impl ToString, impl ToString)>,
214    ) -> Self {
215        self.props
216            .extend(args.into_iter().map(|e| (e.0.to_string(), e.1.to_string())));
217        self
218    }
219
220    /// Builds [`FileIO`].
221    pub fn build(self) -> Result<FileIO> {
222        let storage = Storage::build(self.clone())?;
223        Ok(FileIO {
224            builder: self,
225            inner: Arc::new(storage),
226        })
227    }
228}

259/// Input file is used for reading from files.
260#[derive(Debug)]
261pub struct InputFile {
262    op: Operator,
263    // Absolution path of file.
264    path: String,
265    // Relative path of file to uri, starts at [`relative_path_pos`]
266    relative_path_pos: usize,
267}
268
269impl InputFile {
270    /// Absolute path to root uri.
271    pub fn location(&self) -> &str {
272        &self.path
273    }
274
275    /// Check if file exists.
276    pub async fn exists(&self) -> crate::Result<bool> {
277        Ok(self.op.exists(&self.path[self.relative_path_pos..]).await?)
278    }
279
280    /// Fetch and returns metadata of file.
281    pub async fn metadata(&self) -> crate::Result<FileMetadata> {
282        let meta = self.op.stat(&self.path[self.relative_path_pos..]).await?;
283
284        Ok(FileMetadata {
285            size: meta.content_length(),
286        })
287    }
288
289    /// Read and returns whole content of file.
290    ///
291    /// For continuous reading, use [`Self::reader`] instead.
292    pub async fn read(&self) -> crate::Result<Bytes> {
293        Ok(self
294            .op
295            .read(&self.path[self.relative_path_pos..])
296            .await?
297            .to_bytes())
298    }
299
300    /// Creates [`FileRead`] for continuous reading.
301    ///
302    /// For one-time reading, use [`Self::read`] instead.
303    pub async fn reader(&self) -> crate::Result<impl FileRead + use<>> {
304        Ok(self.op.reader(&self.path[self.relative_path_pos..]).await?)
305    }
306}

308/// Trait for writing file.
309///
310/// # TODO
311///
312/// It's possible for us to remove the async_trait, but we need to figure
313/// out how to handle the object safety.
314#[async_trait::async_trait]
315pub trait FileWrite: Send + Unpin + 'static {
316    /// Write bytes to file.
317    ///
318    /// TODO: we can support writing non-contiguous bytes in the future.
319    async fn write(&mut self, bs: Bytes) -> crate::Result<()>;
320
321    /// Close file.
322    ///
323    /// Calling close on closed file will generate an error.
324    async fn close(&mut self) -> crate::Result<()>;
325}
326
327#[async_trait::async_trait]
328impl FileWrite for opendal::Writer {
329    async fn write(&mut self, bs: Bytes) -> crate::Result<()> {
330        Ok(opendal::Writer::write(self, bs).await?)
331    }
332
333    async fn close(&mut self) -> crate::Result<()> {
334        let _ = opendal::Writer::close(self).await?;
335        Ok(())
336    }
337}

339/// Output file is used for writing to files..
340#[derive(Debug)]
341pub struct OutputFile {
342    op: Operator,
343    // Absolution path of file.
344    path: String,
345    // Relative path of file to uri, starts at [`relative_path_pos`]
346    relative_path_pos: usize,
347}
348
349impl OutputFile {
350    /// Relative path to root uri.
351    pub fn location(&self) -> &str {
352        &self.path
353    }
354
355    /// Checks if file exists.
356    pub async fn exists(&self) -> Result<bool> {
357        Ok(self.op.exists(&self.path[self.relative_path_pos..]).await?)
358    }
359
360    /// Deletes file.
361    ///
362    /// If the file does not exist, it will not return error.
363    pub async fn delete(&self) -> Result<()> {
364        Ok(self.op.delete(&self.path[self.relative_path_pos..]).await?)
365    }
366
367    /// Converts into [`InputFile`].
368    pub fn to_input_file(self) -> InputFile {
369        InputFile {
370            op: self.op,
371            path: self.path,
372            relative_path_pos: self.relative_path_pos,
373        }
374    }
375
376    /// Create a new output file with given bytes.
377    ///
378    /// # Notes
379    ///
380    /// Calling `write` will overwrite the file if it exists.
381    /// For continuous writing, use [`Self::writer`].
382    pub async fn write(&self, bs: Bytes) -> crate::Result<()> {
383        let mut writer = self.writer().await?;
384        writer.write(bs).await?;
385        writer.close().await
386    }
387
388    /// Creates output file for continuous writing.
389    ///
390    /// # Notes
391    ///
392    /// For one-time writing, use [`Self::write`] instead.
393    pub async fn writer(&self) -> crate::Result<Box<dyn FileWrite>> {
394        Ok(Box::new(
395            self.op.writer(&self.path[self.relative_path_pos..]).await?,
396        ))
397    }
398}

453        let tmp_dir = TempDir::new().unwrap();
454
455        let a_path = format!("{}/{}", tmp_dir.path().to_str().unwrap(), "a.txt");
456        let sub_dir_path = format!("{}/sub", tmp_dir.path().to_str().unwrap());
457        let b_path = format!("{}/{}", sub_dir_path, "b.txt");
458        let c_path = format!("{}/{}", sub_dir_path, "c.txt");
459        write_to_file("Iceberg loves rust.", &a_path);
460        write_to_file("Iceberg loves rust.", &b_path);
461        write_to_file("Iceberg loves rust.", &c_path);
462
463        let file_io = create_local_file_io();
464        assert!(file_io.exists(&a_path).await.unwrap());
465
466        // Remove a file should be no-op.
467        file_io.remove_dir_all(&a_path).await.unwrap();
468        assert!(file_io.exists(&a_path).await.unwrap());
469
470        // Remove a not exist dir should be no-op.
471        file_io.remove_dir_all("not_exists/").await.unwrap();
472
473        // Remove a dir should remove all files in it.
474        file_io.remove_dir_all(&sub_dir_path).await.unwrap();
475        assert!(!file_io.exists(&b_path).await.unwrap());
476        assert!(!file_io.exists(&c_path).await.unwrap());
477        assert!(file_io.exists(&a_path).await.unwrap());
478
479        file_io.delete(&a_path).await.unwrap();
480        assert!(!file_io.exists(&a_path).await.unwrap());
481    }
