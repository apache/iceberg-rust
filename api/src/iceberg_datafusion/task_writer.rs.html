18//! TaskWriter for DataFusion integration.
19//!
20//! This module provides a high-level writer that handles partitioning and routing
21//! of RecordBatch data to Iceberg tables.
22
23use datafusion::arrow::array::RecordBatch;
24use iceberg::Result;
25use iceberg::arrow::RecordBatchPartitionSplitter;
26use iceberg::spec::{DataFile, PartitionSpecRef, SchemaRef};
27use iceberg::writer::IcebergWriterBuilder;
28use iceberg::writer::partitioning::PartitioningWriter;
29use iceberg::writer::partitioning::clustered_writer::ClusteredWriter;
30use iceberg::writer::partitioning::fanout_writer::FanoutWriter;
31use iceberg::writer::partitioning::unpartitioned_writer::UnpartitionedWriter;

33/// High-level writer for DataFusion that handles partitioning and routing of RecordBatch data.
34///
35/// TaskWriter coordinates writing data to Iceberg tables by:
36/// - Selecting the appropriate partitioning strategy (unpartitioned, fanout, or clustered)
37/// - Initializing the partition splitter in the constructor for partitioned tables
38/// - Routing data to the underlying writer
39/// - Collecting all written data files
40///
41/// # Type Parameters
42///
43/// * `B` - The IcebergWriterBuilder type used to create underlying writers
44///
45/// # Example
46///
47/// ```rust,ignore
48/// use iceberg::spec::{PartitionSpec, Schema};
49/// use iceberg::writer::base_writer::data_file_writer::DataFileWriterBuilder;
50/// use iceberg_datafusion::writer::task_writer::TaskWriter;
51///
52/// // Create a TaskWriter for an unpartitioned table
53/// let task_writer = TaskWriter::new(
54///     data_file_writer_builder,
55///     false, // fanout_enabled
56///     schema,
57///     partition_spec,
58/// );
59///
60/// // Write data
61/// task_writer.write(record_batch).await?;
62///
63/// // Close and get data files
64/// let data_files = task_writer.close().await?;
65/// ```
66pub(crate) struct TaskWriter<B: IcebergWriterBuilder> {
67    /// The underlying writer (UnpartitionedWriter, FanoutWriter, or ClusteredWriter)
68    writer: SupportedWriter<B>,
69    /// Partition splitter for partitioned tables (initialized in constructor)
70    partition_splitter: Option<RecordBatchPartitionSplitter>,
71}
72
73/// Internal enum to hold the different writer types.
74///
75/// This enum allows TaskWriter to work with different partitioning strategies
76/// while maintaining a unified interface.
77enum SupportedWriter<B: IcebergWriterBuilder> {
78    /// Writer for unpartitioned tables
79    Unpartitioned(UnpartitionedWriter<B>),
80    /// Writer for partitioned tables with unsorted data (maintains multiple active writers)
81    Fanout(FanoutWriter<B>),
82    /// Writer for partitioned tables with sorted data (maintains single active writer)
83    Clustered(ClusteredWriter<B>),
84}
85
86impl<B: IcebergWriterBuilder> TaskWriter<B> {
87    /// Create a new TaskWriter.
88    ///
89    /// # Parameters
90    ///
91    /// * `writer_builder` - The IcebergWriterBuilder to use for creating underlying writers
92    /// * `fanout_enabled` - If true, use FanoutWriter for partitioned tables; otherwise use ClusteredWriter
93    /// * `schema` - The Iceberg schema reference
94    /// * `partition_spec` - The partition specification reference
95    ///
96    /// # Returns
97    ///
98    /// Returns a new TaskWriter instance.
99    ///
100    /// # Writer Selection Logic
101    ///
102    /// - If partition_spec is unpartitioned: creates UnpartitionedWriter
103    /// - If partition_spec is partitioned AND fanout_enabled is true: creates FanoutWriter
104    /// - If partition_spec is partitioned AND fanout_enabled is false: creates ClusteredWriter
105    ///
106    /// # Example
107    ///
108    /// ```rust,ignore
109    /// use iceberg::spec::{PartitionSpec, Schema};
110    /// use iceberg::writer::base_writer::data_file_writer::DataFileWriterBuilder;
111    /// use iceberg_datafusion::writer::task_writer::TaskWriter;
112    ///
113    /// // Create a TaskWriter for an unpartitioned table
114    /// let task_writer = TaskWriter::new(
115    ///     data_file_writer_builder,
116    ///     false, // fanout_enabled
117    ///     schema,
118    ///     partition_spec,
119    /// );
120    /// ```
121    pub fn try_new(
122        writer_builder: B,
123        fanout_enabled: bool,
124        schema: SchemaRef,
125        partition_spec: PartitionSpecRef,
126    ) -> Result<Self> {
127        let writer = if partition_spec.is_unpartitioned() {
128            SupportedWriter::Unpartitioned(UnpartitionedWriter::new(writer_builder))
129        } else if fanout_enabled {
130            SupportedWriter::Fanout(FanoutWriter::new(writer_builder))
131        } else {
132            SupportedWriter::Clustered(ClusteredWriter::new(writer_builder))
133        };
134
135        // Initialize partition splitter in constructor for partitioned tables
136        let partition_splitter = if !partition_spec.is_unpartitioned() {
137            Some(
138                RecordBatchPartitionSplitter::try_new_with_precomputed_values(
139                    schema.clone(),
140                    partition_spec.clone(),
141                )?,
142            )
143        } else {
144            None
145        };
146
147        Ok(Self {
148            writer,
149            partition_splitter,
150        })
151    }
152
153    /// Write a RecordBatch to the TaskWriter.
154    ///
155    /// For partitioned tables, uses the partition splitter to split
156    /// the batch by partition key and route each partition to the underlying writer.
157    /// For unpartitioned tables, data is written directly without splitting.
158    ///
159    /// # Parameters
160    ///
161    /// * `batch` - The RecordBatch to write
162    ///
163    /// # Returns
164    ///
165    /// Returns `Ok(())` on success, or an error if the write fails.
166    ///
167    /// # Errors
168    ///
169    /// This method will return an error if:
170    /// - Splitting the batch by partition fails
171    /// - Writing to the underlying writer fails
172    ///
173    /// # Example
174    ///
175    /// ```rust,ignore
176    /// use arrow_array::RecordBatch;
177    /// use iceberg_datafusion::writer::task_writer::TaskWriter;
178    ///
179    /// // Write a RecordBatch
180    /// task_writer.write(record_batch).await?;
181    /// ```
182    pub async fn write(&mut self, batch: RecordBatch) -> Result<()> {
183        match &mut self.writer {
184            SupportedWriter::Unpartitioned(writer) => {
185                // Unpartitioned: write directly without splitting
186                writer.write(batch).await
187            }
188            SupportedWriter::Fanout(writer) => {
189                Self::write_partitioned_batches(writer, &self.partition_splitter, &batch).await
190            }
191            SupportedWriter::Clustered(writer) => {
192                Self::write_partitioned_batches(writer, &self.partition_splitter, &batch).await
193            }
194        }
195    }
196
197    /// Helper method to split and write partitioned data.
198    ///
199    /// This method handles the common logic for both FanoutWriter and ClusteredWriter:
200    /// - Splits the batch by partition key using the partition splitter
201    /// - Writes each partition to the underlying writer with its corresponding partition key
202    ///
203    /// # Parameters
204    ///
205    /// * `writer` - The underlying PartitioningWriter (FanoutWriter or ClusteredWriter)
206    /// * `partition_splitter` - The partition splitter
207    /// * `batch` - The RecordBatch to write
208    ///
209    /// # Returns
210    ///
211    /// Returns `Ok(())` on success, or an error if the operation fails.
212    async fn write_partitioned_batches<W: PartitioningWriter>(
213        writer: &mut W,
214        partition_splitter: &Option<RecordBatchPartitionSplitter>,
215        batch: &RecordBatch,
216    ) -> Result<()> {
217        // Split batch by partition
218        let splitter = partition_splitter
219            .as_ref()
220            .expect("Partition splitter should be initialized");
221        let partitioned_batches = splitter.split(batch)?;
222
223        // Write each partition
224        for (partition_key, partition_batch) in partitioned_batches {
225            writer.write(partition_key, partition_batch).await?;
226        }
227
228        Ok(())
229    }
230
231    /// Close the TaskWriter and return all written data files.
232    ///
233    /// This method consumes the TaskWriter to prevent further use.
234    ///
235    /// # Returns
236    ///
237    /// Returns a `Vec<DataFile>` containing all written files, or an error if closing fails.
238    ///
239    /// # Errors
240    ///
241    /// This method will return an error if:
242    /// - Closing the underlying writer fails
243    /// - Any I/O operation fails during the close process
244    ///
245    /// # Example
246    ///
247    /// ```rust,ignore
248    /// use iceberg_datafusion::writer::task_writer::TaskWriter;
249    ///
250    /// // Close the writer and get all data files
251    /// let data_files = task_writer.close().await?;
252    /// ```
253    pub async fn close(self) -> Result<Vec<DataFile>> {
254        match self.writer {
255            SupportedWriter::Unpartitioned(writer) => writer.close().await,
256            SupportedWriter::Fanout(writer) => writer.close().await,
257            SupportedWriter::Clustered(writer) => writer.close().await,
258        }
259    }
260}

342    fn create_writer_builder(
343        temp_dir: &TempDir,
344        schema: Arc<iceberg::spec::Schema>,
345    ) -> Result<
346        DataFileWriterBuilder<
347            ParquetWriterBuilder,
348            DefaultLocationGenerator,
349            DefaultFileNameGenerator,
350        >,
351    > {
352        let file_io = FileIOBuilder::new_fs_io().build()?;
353        let location_gen = DefaultLocationGenerator::with_data_location(
354            temp_dir.path().to_str().unwrap().to_string(),
355        );
356        let file_name_gen =
357            DefaultFileNameGenerator::new("test".to_string(), None, DataFileFormat::Parquet);
358        let parquet_writer_builder =
359            ParquetWriterBuilder::new(WriterProperties::builder().build(), schema);
360        let rolling_writer_builder = RollingFileWriterBuilder::new_with_default_file_size(
361            parquet_writer_builder,
362            file_io,
363            location_gen,
364            file_name_gen,
365        );
366        Ok(DataFileWriterBuilder::new(rolling_writer_builder))
367    }

371        let temp_dir = TempDir::new()?;
372        let schema = create_test_schema()?;
373        let arrow_schema = create_arrow_schema();
374
375        // Create unpartitioned spec
376        let partition_spec = Arc::new(PartitionSpec::builder(schema.clone()).build()?);
377
378        let writer_builder = create_writer_builder(&temp_dir, schema.clone())?;
379        let mut task_writer = TaskWriter::try_new(writer_builder, false, schema, partition_spec)?;

381        // Write data

383            Arc::new(Int32Array::from(vec![1, 2, 3])),

394
395        Ok(())

398    /// Helper to verify partition data files
399    fn verify_partition_files(
400        data_files: &[iceberg::spec::DataFile],
401        expected_total: u64,
402    ) -> HashMap<String, u64> {
403        let total_records: u64 = data_files.iter().map(|f| f.record_count()).sum();
404        assert_eq!(total_records, expected_total, "Total record count mismatch");
405
406        let mut partition_counts = HashMap::new();
407        for data_file in data_files {
408            let partition_value = data_file.partition();
409            let region_literal = partition_value.fields()[0]
410                .as_ref()
411                .expect("Partition value should not be null");
412            let region = match region_literal
413                .as_primitive_literal()
414                .expect("Should be primitive literal")
415            {
416                iceberg::spec::PrimitiveLiteral::String(s) => s.clone(),
417                _ => panic!("Expected string partition value"),
418            };
419
420            *partition_counts.entry(region.clone()).or_insert(0) += data_file.record_count();
421
422            // Verify file path contains partition information
423            assert!(
424                data_file.file_path().contains("region="),
425                "File path should contain partition info"
426            );
427        }
428        partition_counts
429    }

432    async fn test_task_writer_partitioned_fanout() -> Result<()> {
433        let temp_dir = TempDir::new()?;
434        let schema = create_test_schema()?;
435        let arrow_schema = create_arrow_schema_with_partition();
436
437        let partition_spec = Arc::new(
438            PartitionSpec::builder(schema.clone())
439                .with_spec_id(1)
440                .add_partition_field("region", "region", iceberg::spec::Transform::Identity)?
441                .build()?,
442        );
443
444        let writer_builder = create_writer_builder(&temp_dir, schema.clone())?;
445        let mut task_writer = TaskWriter::try_new(writer_builder, true, schema, partition_spec)?;
446
447        // Create partition column
448        let partition_field = Field::new("region", DataType::Utf8, false).with_metadata(
449            HashMap::from([(PARQUET_FIELD_ID_META_KEY.to_string(), "1000".to_string())]),
450        );
451        let partition_values = StringArray::from(vec!["US", "EU", "US", "EU"]);
452        let partition_struct = StructArray::from(vec![(
453            Arc::new(partition_field),
454            Arc::new(partition_values) as ArrayRef,
455        )]);
456
457        let batch = RecordBatch::try_new(arrow_schema, vec![
458            Arc::new(Int32Array::from(vec![1, 2, 3, 4])),
459            Arc::new(StringArray::from(vec!["Alice", "Bob", "Charlie", "Dave"])),
460            Arc::new(StringArray::from(vec!["US", "EU", "US", "EU"])),
461            Arc::new(partition_struct),
462        ])?;
463
464        task_writer.write(batch).await?;
465        let data_files = task_writer.close().await?;
466
467        let partition_counts = verify_partition_files(&data_files, 4);
468        assert_eq!(partition_counts.get("US"), Some(&2));
469        assert_eq!(partition_counts.get("EU"), Some(&2));
470
471        Ok(())
472    }
