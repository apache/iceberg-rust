18use datafusion::catalog::TableProvider;
19use datafusion::physical_expr::EquivalenceProperties;
20use datafusion::physical_plan::execution_plan::{Boundedness, EmissionType};
21use datafusion::physical_plan::stream::RecordBatchStreamAdapter;
22use datafusion::physical_plan::{DisplayAs, ExecutionPlan, Partitioning, PlanProperties};
23use futures::TryStreamExt;
24
25use crate::metadata_table::IcebergMetadataTableProvider;
26
27#[derive(Debug)]
28pub struct IcebergMetadataScan {
29    provider: IcebergMetadataTableProvider,
30    properties: PlanProperties,
31}
32
33impl IcebergMetadataScan {
34    pub fn new(provider: IcebergMetadataTableProvider) -> Self {
35        let properties = PlanProperties::new(
36            EquivalenceProperties::new(provider.schema()),
37            Partitioning::UnknownPartitioning(1),
38            EmissionType::Incremental,
39            Boundedness::Bounded,
40        );
41        Self {
42            provider,
43            properties,
44        }
45    }
46}
47
48impl DisplayAs for IcebergMetadataScan {
49    fn fmt_as(
50        &self,
51        _t: datafusion::physical_plan::DisplayFormatType,
52        f: &mut std::fmt::Formatter,
53    ) -> std::fmt::Result {
54        write!(f, "IcebergMetadataScan")
55    }
56}
57
58impl ExecutionPlan for IcebergMetadataScan {
59    fn name(&self) -> &str {
60        "IcebergMetadataScan"
61    }
62
63    fn as_any(&self) -> &dyn std::any::Any {
64        self
65    }
66
67    fn properties(&self) -> &PlanProperties {
68        &self.properties
69    }
70
71    fn children(&self) -> Vec<&std::sync::Arc<dyn ExecutionPlan>> {
72        vec![]
73    }
74
75    fn with_new_children(
76        self: std::sync::Arc<Self>,
77        _children: Vec<std::sync::Arc<dyn ExecutionPlan>>,
78    ) -> datafusion::error::Result<std::sync::Arc<dyn ExecutionPlan>> {
79        Ok(self)
80    }
81
82    fn execute(
83        &self,
84        _partition: usize,
85        _context: std::sync::Arc<datafusion::execution::TaskContext>,
86    ) -> datafusion::error::Result<datafusion::execution::SendableRecordBatchStream> {
87        let fut = self.provider.clone().scan();
88        let stream = futures::stream::once(fut).try_flatten();
89        let schema = self.provider.schema();
90        Ok(Box::pin(RecordBatchStreamAdapter::new(schema, stream)))
91    }
92}
