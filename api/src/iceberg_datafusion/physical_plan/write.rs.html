59/// An execution plan node that writes data to an Iceberg table.
60///
61/// This execution plan takes input data from a child execution plan and writes it to an Iceberg table.
62/// It handles the creation of data files in the appropriate format and returns information about the written files as its output.
63///
64/// The output of this execution plan is a record batch containing a single column with serialized
65/// data file information that can be used for committing the write operation to the table.
66#[derive(Debug)]
67pub(crate) struct IcebergWriteExec {
68    table: Table,
69    input: Arc<dyn ExecutionPlan>,
70    result_schema: ArrowSchemaRef,
71    plan_properties: PlanProperties,
72}
73
74impl IcebergWriteExec {
75    pub fn new(table: Table, input: Arc<dyn ExecutionPlan>, schema: ArrowSchemaRef) -> Self {
76        let plan_properties = Self::compute_properties(&input, schema);
77
78        Self {
79            table,
80            input,
81            result_schema: Self::make_result_schema(),
82            plan_properties,
83        }
84    }
85
86    fn compute_properties(
87        input: &Arc<dyn ExecutionPlan>,
88        schema: ArrowSchemaRef,
89    ) -> PlanProperties {
90        PlanProperties::new(
91            EquivalenceProperties::new(schema),
92            Partitioning::UnknownPartitioning(input.output_partitioning().partition_count()),
93            EmissionType::Final,
94            Boundedness::Bounded,
95        )
96    }
97
98    // Create a record batch with serialized data files
99    fn make_result_batch(data_files: Vec<String>) -> DFResult<RecordBatch> {
100        let files_array = Arc::new(StringArray::from(data_files)) as ArrayRef;
101
102        RecordBatch::try_new(Self::make_result_schema(), vec![files_array]).map_err(|e| {
103            DataFusionError::ArrowError(e, Some("Failed to make result batch".to_string()))
104        })
105    }
106
107    fn make_result_schema() -> ArrowSchemaRef {
108        // Define a schema.
109        Arc::new(ArrowSchema::new(vec![Field::new(
110            DATA_FILES_COL_NAME,
111            DataType::Utf8,
112            false,
113        )]))
114    }
115}

117impl DisplayAs for IcebergWriteExec {
118    fn fmt_as(&self, t: DisplayFormatType, f: &mut Formatter) -> std::fmt::Result {
119        match t {
120            DisplayFormatType::Default => {
121                write!(f, "IcebergWriteExec: table={}", self.table.identifier())
122            }
123            DisplayFormatType::Verbose => {
124                write!(
125                    f,
126                    "IcebergWriteExec: table={}, result_schema={:?}",
127                    self.table.identifier(),
128                    self.result_schema
129                )
130            }
131            DisplayFormatType::TreeRender => {
132                write!(f, "IcebergWriteExec: table={}", self.table.identifier())
133            }
134        }
135    }
136}

138impl ExecutionPlan for IcebergWriteExec {
139    fn name(&self) -> &str {
140        "IcebergWriteExec"
141    }
142
143    fn as_any(&self) -> &dyn Any {
144        self
145    }
146
147    fn properties(&self) -> &PlanProperties {
148        &self.plan_properties
149    }
150
151    fn children(&self) -> Vec<&Arc<dyn ExecutionPlan>> {
152        vec![&self.input]
153    }
154
155    fn with_new_children(
156        self: Arc<Self>,
157        children: Vec<Arc<dyn ExecutionPlan>>,
158    ) -> DFResult<Arc<dyn ExecutionPlan>> {
159        if children.len() != 1 {
160            return Err(DataFusionError::Internal(format!(
161                "IcebergWriteExec expects exactly one child, but provided {}",
162                children.len()
163            )));
164        }
165
166        Ok(Arc::new(Self::new(
167            self.table.clone(),
168            Arc::clone(&children[0]),
169            self.schema(),
170        )))
171    }

173    /// Executes the write operation for the given partition.
174    ///
175    /// This function:
176    /// 1. Sets up a data file writer based on the table's configuration
177    /// 2. Processes input data from the child execution plan
178    /// 3. Writes the data to files using the configured writer
179    /// 4. Returns a stream containing information about the written data files
180    ///
181    /// The output of this function is a stream of record batches with the following structure:
182    ///
183    /// ```text
184    /// +------------------+
185    /// | data_files       |
186    /// +------------------+
187    /// | "{"file_path":.. |  <- JSON string representing a data file
188    /// +------------------+
189    /// ```
190    ///
191    /// Each row in the output contains a JSON string representing a data file that was written.
192    ///
193    /// This output can be used by a subsequent operation to commit the added files to the table.
194    fn execute(
195        &self,
196        partition: usize,
197        context: Arc<TaskContext>,
198    ) -> DFResult<SendableRecordBatchStream> {
199        if !self
200            .table
201            .metadata()
202            .default_partition_spec()
203            .is_unpartitioned()
204        {
205            // TODO add support for partitioned tables
206            return Err(DataFusionError::NotImplemented(
207                "IcebergWriteExec does not support partitioned tables yet".to_string(),
208            ));
209        }
210
211        let spec_id = self.table.metadata().default_partition_spec_id();
212        let partition_type = self.table.metadata().default_partition_type().clone();
213        let format_version = self.table.metadata().format_version();
214
215        // Check data file format
216        let file_format = DataFileFormat::from_str(
217            self.table
218                .metadata()
219                .properties()
220                .get(PROPERTY_DEFAULT_FILE_FORMAT)
221                .unwrap_or(&PROPERTY_DEFAULT_FILE_FORMAT_DEFAULT.to_string()),
222        )
223        .map_err(to_datafusion_error)?;
224        if file_format != DataFileFormat::Parquet {
225            return Err(to_datafusion_error(Error::new(
226                ErrorKind::FeatureUnsupported,
227                format!(
228                    "File format {} is not supported for insert_into yet!",
229                    file_format
230                ),
231            )));
232        }
233
234        // Create data file writer builder
235        let parquet_file_writer_builder = ParquetWriterBuilder::new_with_match_mode(
236            WriterProperties::default(),
237            self.table.metadata().current_schema().clone(),
238            None,
239            FieldMatchMode::Name,
240            self.table.file_io().clone(),
241            DefaultLocationGenerator::new(self.table.metadata().clone())
242                .map_err(to_datafusion_error)?,
243            // todo filename prefix/suffix should be configurable
244            DefaultFileNameGenerator::new(Uuid::now_v7().to_string(), None, file_format),
245        );
246        let target_file_size = match self
247            .table
248            .metadata()
249            .properties()
250            .get(PROPERTY_WRITE_TARGET_FILE_SIZE_BYTES)
251        {
252            Some(value_str) => value_str
253                .parse::<usize>()
254                .map_err(|e| {
255                    Error::new(
256                        ErrorKind::DataInvalid,
257                        "Invalid value for write.target-file-size-bytes",
258                    )
259                    .with_source(e)
260                })
261                .map_err(to_datafusion_error)?,
262            None => PROPERTY_WRITE_TARGET_FILE_SIZE_BYTES_DEFAULT,
263        };
264        let rolling_writer_builder =
265            RollingFileWriterBuilder::new(parquet_file_writer_builder, target_file_size);
266        let data_file_writer_builder =
267            DataFileWriterBuilder::new(rolling_writer_builder, None, spec_id);
268
269        // Get input data
270        let data = execute_input_stream(
271            Arc::clone(&self.input),
272            Arc::new(
273                schema_to_arrow_schema(self.table.metadata().current_schema())
274                    .map_err(to_datafusion_error)?,
275            ),
276            partition,
277            Arc::clone(&context),
278        )?;
279
280        // Create write stream
281        let stream = futures::stream::once(async move {
282            let mut writer = data_file_writer_builder
283                .build()
284                .await
285                .map_err(to_datafusion_error)?;
286            let mut input_stream = data;
287
288            while let Some(batch) = input_stream.next().await {
289                writer.write(batch?).await.map_err(to_datafusion_error)?;
290            }
291
292            let data_files = writer.close().await.map_err(to_datafusion_error)?;
293
294            // Convert builders to data files and then to JSON strings
295            let data_files_strs: Vec<String> = data_files
296                .into_iter()
297                .map(|data_file| {
298                    serialize_data_file_to_json(data_file, &partition_type, format_version)
299                        .map_err(to_datafusion_error)
300                })
301                .collect::<DFResult<Vec<String>>>()?;
302
303            Self::make_result_batch(data_files_strs)
304        })
305        .boxed();
306
307        Ok(Box::pin(RecordBatchStreamAdapter::new(
308            Arc::clone(&self.result_schema),
309            stream,
310        )))
311    }
312}

464    #[tokio::test]
465    async fn test_iceberg_write_exec() -> Result<()> {
466        // 1. Set up test environment
467        let iceberg_catalog = get_iceberg_catalog().await;
468        let namespace = NamespaceIdent::new("test_namespace".to_string());
469
470        // Create namespace
471        iceberg_catalog
472            .create_namespace(&namespace, HashMap::new())
473            .await?;
474
475        // Create schema
476        let schema = get_test_schema()?;
477
478        // Create table
479        let table_name = "test_table";
480        let table_location = temp_path();
481        let creation = get_table_creation(table_location, table_name, schema);
482        let table = iceberg_catalog.create_table(&namespace, creation).await?;
483
484        // 2. Create test data
485        let arrow_schema = Arc::new(ArrowSchema::new(vec![
486            Field::new("id", DataType::Int32, false).with_metadata(HashMap::from([(
487                PARQUET_FIELD_ID_META_KEY.to_string(),
488                "1".to_string(),
489            )])),
490            Field::new("name", DataType::Utf8, false).with_metadata(HashMap::from([(
491                PARQUET_FIELD_ID_META_KEY.to_string(),
492                "2".to_string(),
493            )])),
494        ]));
495
496        let id_array = Arc::new(Int32Array::from(vec![1, 2, 3])) as ArrayRef;
497        let name_array = Arc::new(StringArray::from(vec!["Alice", "Bob", "Charlie"])) as ArrayRef;
498
499        let batch = RecordBatch::try_new(arrow_schema.clone(), vec![id_array, name_array])
500            .map_err(|e| {
501                Error::new(
502                    ErrorKind::Unexpected,
503                    format!("Failed to create record batch: {}", e),
504                )
505            })?;
506
507        // 3. Create mock input execution plan
508        let input_plan = Arc::new(MockExecutionPlan::new(arrow_schema.clone(), vec![
509            batch.clone(),
510        ]));
511
512        // 4. Create IcebergWriteExec
513        let write_exec = IcebergWriteExec::new(table.clone(), input_plan, arrow_schema);
514
515        // 5. Execute the plan
516        let task_ctx = Arc::new(TaskContext::default());
517        let stream = write_exec.execute(0, task_ctx).map_err(|e| {
518            Error::new(
519                ErrorKind::Unexpected,
520                format!("Failed to execute plan: {}", e),
521            )
522        })?;
523
524        // Collect the results
525        let mut results = vec![];
526        let mut stream = stream;
527        while let Some(batch) = stream.next().await {
528            results.push(batch.map_err(|e| {
529                Error::new(ErrorKind::Unexpected, format!("Failed to get batch: {}", e))
530            })?);
531        }
532
533        // 6. Verify the results
534        assert_eq!(results.len(), 1, "Expected one result batch");
535        let result_batch = &results[0];
536
537        // Check schema
538        assert_eq!(
539            result_batch.schema().as_ref(),
540            &ArrowSchema::new(vec![Field::new(DATA_FILES_COL_NAME, DataType::Utf8, false)])
541        );
542
543        // Check data
544        assert_eq!(result_batch.num_rows(), 1, "Expected one data file");
545
546        // Get the data file JSON
547        let data_file_json = result_batch
548            .column(0)
549            .as_any()
550            .downcast_ref::<StringArray>()
551            .expect("Expected StringArray")
552            .value(0);
553
554        // Deserialize the data file JSON
555        let partition_type = table.metadata().default_partition_type();
556        let spec_id = table.metadata().default_partition_spec_id();
557        let schema = table.metadata().current_schema();
558
559        let data_file =
560            deserialize_data_file_from_json(data_file_json, spec_id, partition_type, schema)
561                .expect("Failed to deserialize data file JSON");
562
563        // Verify data file properties
564        assert_eq!(
565            data_file.record_count(),
566            3,
567            "Expected 3 records in the data file"
568        );
569        assert!(
570            data_file.file_size_in_bytes() > 0,
571            "File size should be greater than 0"
572        );
573        assert_eq!(
574            data_file.file_format(),
575            DataFileFormat::Parquet,
576            "Expected Parquet file format"
577        );
578
579        // Verify column statistics
580        assert!(
581            data_file.column_sizes().get(&1).unwrap() > &0,
582            "Column 1 size should be greater than 0"
583        );
584        assert!(
585            data_file.column_sizes().get(&2).unwrap() > &0,
586            "Column 2 size should be greater than 0"
587        );
588
589        assert_eq!(
590            *data_file.value_counts().get(&1).unwrap(),
591            3,
592            "Expected 3 values for column 1"
593        );
594        assert_eq!(
595            *data_file.value_counts().get(&2).unwrap(),
596            3,
597            "Expected 3 values for column 2"
598        );
599
600        // Verify lower and upper bounds
601        assert!(
602            data_file.lower_bounds().contains_key(&1) || data_file.lower_bounds().contains_key(&2),
603            "Expected lower bounds to contain at least one column"
604        );
605        assert!(
606            data_file.upper_bounds().contains_key(&1) || data_file.upper_bounds().contains_key(&2),
607            "Expected upper bounds to contain at least one column"
608        );
609
610        // Check that the file path exists
611        let file_path = data_file.file_path();
612        assert!(!file_path.is_empty(), "File path should not be empty");
613
614        // 7. Verify the file exists
615        let file_io = table.file_io();
616        assert!(file_io.exists(file_path).await?, "Data file should exist");
617
618        Ok(())
619    }
