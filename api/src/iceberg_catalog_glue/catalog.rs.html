45/// Glue catalog URI
46pub const GLUE_CATALOG_PROP_URI: &str = "uri";
47/// Glue catalog id
48pub const GLUE_CATALOG_PROP_CATALOG_ID: &str = "catalog_id";
49/// Glue catalog warehouse location
50pub const GLUE_CATALOG_PROP_WAREHOUSE: &str = "warehouse";

56impl Default for GlueCatalogBuilder {
57    fn default() -> Self {
58        Self(GlueCatalogConfig {
59            name: None,

61            catalog_id: None,
62            warehouse: "".to_string(),
63            props: HashMap::new(),
64        })

68impl CatalogBuilder for GlueCatalogBuilder {
69    type C = GlueCatalog;
70
71    fn load(
72        mut self,
73        name: impl Into<String>,
74        props: HashMap<String, String>,
75    ) -> impl Future<Output = Result<Self::C>> + Send {
76        self.0.name = Some(name.into());
77
78        if props.contains_key(GLUE_CATALOG_PROP_URI) {
79            self.0.uri = props.get(GLUE_CATALOG_PROP_URI).cloned()
80        }
81
82        if props.contains_key(GLUE_CATALOG_PROP_CATALOG_ID) {
83            self.0.catalog_id = props.get(GLUE_CATALOG_PROP_CATALOG_ID).cloned()
84        }
85
86        if props.contains_key(GLUE_CATALOG_PROP_WAREHOUSE) {
87            self.0.warehouse = props
88                .get(GLUE_CATALOG_PROP_WAREHOUSE)
89                .cloned()
90                .unwrap_or_default();
91        }
92
93        // Collect other remaining properties
94        self.0.props = props
95            .into_iter()
96            .filter(|(k, _)| {
97                k != GLUE_CATALOG_PROP_URI
98                    && k != GLUE_CATALOG_PROP_CATALOG_ID
99                    && k != GLUE_CATALOG_PROP_WAREHOUSE
100            })
101            .collect();
102
103        async move {
104            if self.0.name.is_none() {
105                return Err(Error::new(
106                    ErrorKind::DataInvalid,
107                    "Catalog name is required",
108                ));
109            }
110            if self.0.warehouse.is_empty() {
111                return Err(Error::new(
112                    ErrorKind::DataInvalid,
113                    "Catalog warehouse is required",
114                ));
115            }
116
117            GlueCatalog::new(self.0).await
118        }
119    }
120}

149impl GlueCatalog {
150    /// Create a new glue catalog
151    async fn new(config: GlueCatalogConfig) -> Result<Self> {
152        let sdk_config = create_sdk_config(&config.props, config.uri.as_ref()).await;
153        let mut file_io_props = config.props.clone();
154        if !file_io_props.contains_key(S3_ACCESS_KEY_ID) {
155            if let Some(access_key_id) = file_io_props.get(AWS_ACCESS_KEY_ID) {
156                file_io_props.insert(S3_ACCESS_KEY_ID.to_string(), access_key_id.to_string());
157            }
158        }
159        if !file_io_props.contains_key(S3_SECRET_ACCESS_KEY) {
160            if let Some(secret_access_key) = file_io_props.get(AWS_SECRET_ACCESS_KEY) {
161                file_io_props.insert(
162                    S3_SECRET_ACCESS_KEY.to_string(),
163                    secret_access_key.to_string(),
164                );
165            }
166        }
167        if !file_io_props.contains_key(S3_REGION) {
168            if let Some(region) = file_io_props.get(AWS_REGION_NAME) {
169                file_io_props.insert(S3_REGION.to_string(), region.to_string());
170            }
171        }
172        if !file_io_props.contains_key(S3_SESSION_TOKEN) {
173            if let Some(session_token) = file_io_props.get(AWS_SESSION_TOKEN) {
174                file_io_props.insert(S3_SESSION_TOKEN.to_string(), session_token.to_string());
175            }
176        }
177        if !file_io_props.contains_key(S3_ENDPOINT) {
178            if let Some(aws_endpoint) = config.uri.as_ref() {
179                file_io_props.insert(S3_ENDPOINT.to_string(), aws_endpoint.to_string());
180            }
181        }
182
183        let client = aws_sdk_glue::Client::new(&sdk_config);
184
185        let file_io = FileIO::from_path(&config.warehouse)?
186            .with_props(file_io_props)
187            .build()?;
188
189        Ok(GlueCatalog {
190            config,
191            client: GlueClient(client),
192            file_io,
193        })
194    }
195    /// Get the catalogs `FileIO`
196    pub fn file_io(&self) -> FileIO {
197        self.file_io.clone()
198    }

203    /// List namespaces from glue catalog.
204    ///
205    /// Glue doesn't support nested namespaces.
206    /// We will return an empty list if parent is some.
207    async fn list_namespaces(
208        &self,
209        parent: Option<&NamespaceIdent>,
210    ) -> Result<Vec<NamespaceIdent>> {
211        if parent.is_some() {
212            return Ok(vec![]);
213        }
214
215        let mut database_list: Vec<NamespaceIdent> = Vec::new();
216        let mut next_token: Option<String> = None;
217
218        loop {
219            let builder = match &next_token {
220                Some(token) => self.client.0.get_databases().next_token(token),
221                None => self.client.0.get_databases(),
222            };
223            let builder = with_catalog_id!(builder, self.config);
224            let resp = builder.send().await.map_err(from_aws_sdk_error)?;
225
226            let dbs: Vec<NamespaceIdent> = resp
227                .database_list()
228                .iter()
229                .map(|db| NamespaceIdent::new(db.name().to_string()))
230                .collect();
231
232            database_list.extend(dbs);
233
234            next_token = resp.next_token().map(ToOwned::to_owned);
235            if next_token.is_none() {
236                break;
237            }
238        }
239
240        Ok(database_list)
241    }

244    ///
245    /// Attempts to create a namespace defined by the `namespace`
246    /// parameter and configured with the specified `properties`.
247    ///
248    /// This function can return an error in the following situations:
249    ///
250    /// - Errors from `validate_namespace` if the namespace identifier does not
251    /// meet validation criteria.
252    /// - Errors from `convert_to_database` if the properties cannot be

260    ) -> Result<Namespace> {

264        let builder = with_catalog_id!(builder, self.config);
265
266        builder.send().await.map_err(from_aws_sdk_error)?;
267
268        Ok(Namespace::with_properties(namespace.clone(), properties))
269    }

271    /// Retrieves a namespace by its identifier.
272    ///
273    /// Validates the given namespace identifier and then queries the
274    /// underlying database client to fetch the corresponding namespace data.
275    /// Constructs a `Namespace` object with the retrieved data and returns it.
276    ///
277    /// This function can return an error in any of the following situations:
278    /// - If the provided namespace identifier fails validation checks
279    /// - If there is an error querying the database, returned by
280    /// `from_sdk_error`.
281    async fn get_namespace(&self, namespace: &NamespaceIdent) -> Result<Namespace> {
282        let db_name = validate_namespace(namespace)?;
283
284        let builder = self.client.0.get_database().name(&db_name);
285        let builder = with_catalog_id!(builder, self.config);
286
287        let resp = builder.send().await.map_err(from_aws_sdk_error)?;
288
289        match resp.database() {
290            Some(db) => {
291                let namespace = convert_to_namespace(db);
292                Ok(namespace)
293            }
294            None => Err(Error::new(
295                ErrorKind::DataInvalid,
296                format!("Database with name: {} does not exist", db_name),
297            )),
298        }
299    }

329                    return Ok(false);

331                Err(from_aws_sdk_error(err))
332            }
333        }

367    /// Asynchronously drops a namespace from the Glue Catalog.
368    ///
369    /// Checks if the namespace is empty. If it still contains tables the
370    /// namespace will not be dropped, but an error is returned instead.
371    ///
372    /// # Returns
373    /// A `Result<()>` indicating the outcome:
374    /// - `Ok(())` signifies successful namespace deletion.
375    /// - `Err(...)` signifies failure to drop the namespace due to validation
376    /// errors, connectivity issues, or Glue Catalog constraints.
377    async fn drop_namespace(&self, namespace: &NamespaceIdent) -> Result<()> {
378        let db_name = validate_namespace(namespace)?;
379        let table_list = self.list_tables(namespace).await?;
380
381        if !table_list.is_empty() {
382            return Err(Error::new(
383                ErrorKind::DataInvalid,
384                format!("Database with name: {} is not empty", &db_name),
385            ));
386        }
387
388        let builder = self.client.0.delete_database().name(db_name);
389        let builder = with_catalog_id!(builder, self.config);
390
391        builder.send().await.map_err(from_aws_sdk_error)?;
392
393        Ok(())
394    }

396    /// Asynchronously lists all tables within a specified namespace.
397    ///
398    /// # Returns
399    /// A `Result<Vec<TableIdent>>`, which is:
400    /// - `Ok(vec![...])` containing a vector of `TableIdent` instances, each
401    /// representing a table within the specified namespace.
402    /// - `Err(...)` if an error occurs during namespace validation or while
403    /// querying the database.
404    async fn list_tables(&self, namespace: &NamespaceIdent) -> Result<Vec<TableIdent>> {
405        let db_name = validate_namespace(namespace)?;
406
407        let mut table_list: Vec<TableIdent> = Vec::new();
408        let mut next_token: Option<String> = None;
409
410        loop {
411            let builder = match &next_token {
412                Some(token) => self
413                    .client
414                    .0
415                    .get_tables()
416                    .database_name(&db_name)
417                    .next_token(token),
418                None => self.client.0.get_tables().database_name(&db_name),
419            };
420            let builder = with_catalog_id!(builder, self.config);
421            let resp = builder.send().await.map_err(from_aws_sdk_error)?;
422
423            let tables: Vec<_> = resp
424                .table_list()
425                .iter()
426                .map(|tbl| TableIdent::new(namespace.clone(), tbl.name().to_string()))
427                .collect();
428
429            table_list.extend(tables);
430
431            next_token = resp.next_token().map(ToOwned::to_owned);
432            if next_token.is_none() {
433                break;
434            }
435        }
436
437        Ok(table_list)
438    }

440    /// Creates a new table within a specified namespace using the provided
441    /// table creation settings.
442    ///
443    /// # Returns
444    /// A `Result` wrapping a `Table` object representing the newly created
445    /// table.
446    ///
447    /// # Errors
448    /// This function may return an error in several cases, including invalid
449    /// namespace identifiers, failure to determine a default storage location,
450    /// issues generating or writing table metadata, and errors communicating
451    /// with the Glue Catalog.
452    async fn create_table(
453        &self,
454        namespace: &NamespaceIdent,
455        mut creation: TableCreation,
456    ) -> Result<Table> {
457        let db_name = validate_namespace(namespace)?;
458        let table_name = creation.name.clone();
459
460        let location = match &creation.location {
461            Some(location) => location.clone(),
462            None => {
463                let ns = self.get_namespace(namespace).await?;
464                let location =
465                    get_default_table_location(&ns, &db_name, &table_name, &self.config.warehouse);
466                creation.location = Some(location.clone());
467                location
468            }
469        };
470        let metadata = TableMetadataBuilder::from_table_creation(creation)?
471            .build()?
472            .metadata;
473        let metadata_location =
474            MetadataLocation::new_with_table_location(location.clone()).to_string();
475
476        metadata.write_to(&self.file_io, &metadata_location).await?;
477
478        let glue_table = convert_to_glue_table(
479            &table_name,
480            metadata_location.clone(),
481            &metadata,
482            metadata.properties(),
483            None,
484        )?;
485
486        let builder = self
487            .client
488            .0
489            .create_table()
490            .database_name(&db_name)
491            .table_input(glue_table);
492        let builder = with_catalog_id!(builder, self.config);
493
494        builder.send().await.map_err(from_aws_sdk_error)?;
495
496        Table::builder()
497            .file_io(self.file_io())
498            .metadata_location(metadata_location)
499            .metadata(metadata)
500            .identifier(TableIdent::new(NamespaceIdent::new(db_name), table_name))
501            .build()
502    }

504    /// Loads a table from the Glue Catalog and constructs a `Table` object
505    /// based on its metadata.
506    ///
507    /// # Returns
508    /// A `Result` wrapping a `Table` object that represents the loaded table.
509    ///
510    /// # Errors
511    /// This function may return an error in several scenarios, including:
512    /// - Failure to validate the namespace.
513    /// - Failure to retrieve the table from the Glue Catalog.
514    /// - Absence of metadata location information in the table's properties.
515    /// - Issues reading or deserializing the table's metadata file.
516    async fn load_table(&self, table: &TableIdent) -> Result<Table> {
517        let db_name = validate_namespace(table.namespace())?;
518        let table_name = table.name();
519
520        let builder = self
521            .client
522            .0
523            .get_table()
524            .database_name(&db_name)
525            .name(table_name);
526        let builder = with_catalog_id!(builder, self.config);
527
528        let glue_table_output = builder.send().await.map_err(from_aws_sdk_error)?;
529
530        match glue_table_output.table() {
531            None => Err(Error::new(
532                ErrorKind::TableNotFound,
533                format!(
534                    "Table object for database: {} and table: {} does not exist",
535                    db_name, table_name
536                ),
537            )),
538            Some(table) => {
539                let metadata_location = get_metadata_location(&table.parameters)?;
540
541                let metadata = TableMetadata::read_from(&self.file_io, &metadata_location).await?;
542
543                Table::builder()
544                    .file_io(self.file_io())
545                    .metadata_location(metadata_location)
546                    .metadata(metadata)
547                    .identifier(TableIdent::new(
548                        NamespaceIdent::new(db_name),
549                        table_name.to_owned(),
550                    ))
551                    .build()
552            }
553        }
554    }

619    /// Asynchronously renames a table within the database
620    /// or moves it between namespaces (databases).
621    ///
622    /// # Returns
623    /// - `Ok(())` on successful rename or move of the table.
624    /// - `Err(...)` if an error occurs during the process.
625    async fn rename_table(&self, src: &TableIdent, dest: &TableIdent) -> Result<()> {
626        let src_db_name = validate_namespace(src.namespace())?;
627        let dest_db_name = validate_namespace(dest.namespace())?;
628
629        let src_table_name = src.name();
630        let dest_table_name = dest.name();
631
632        let builder = self
633            .client
634            .0
635            .get_table()
636            .database_name(&src_db_name)
637            .name(src_table_name);
638        let builder = with_catalog_id!(builder, self.config);
639
640        let glue_table_output = builder.send().await.map_err(from_aws_sdk_error)?;
641
642        match glue_table_output.table() {
643            None => Err(Error::new(
644                ErrorKind::TableNotFound,
645                format!(
646                    "'Table' object for database: {} and table: {} does not exist",
647                    src_db_name, src_table_name
648                ),
649            )),
650            Some(table) => {
651                let rename_table_input = TableInput::builder()
652                    .name(dest_table_name)
653                    .set_parameters(table.parameters.clone())
654                    .set_storage_descriptor(table.storage_descriptor.clone())
655                    .set_table_type(table.table_type.clone())
656                    .set_description(table.description.clone())
657                    .build()
658                    .map_err(from_aws_build_error)?;
659
660                let builder = self
661                    .client
662                    .0
663                    .create_table()
664                    .database_name(&dest_db_name)
665                    .table_input(rename_table_input);
666                let builder = with_catalog_id!(builder, self.config);
667
668                builder.send().await.map_err(from_aws_sdk_error)?;
669
670                let drop_src_table_result = self.drop_table(src).await;
671
672                match drop_src_table_result {
673                    Ok(_) => Ok(()),
674                    Err(_) => {
675                        let err_msg_src_table = format!(
676                            "Failed to drop old table {}.{}.",
677                            src_db_name, src_table_name
678                        );
679
680                        let drop_dest_table_result = self.drop_table(dest).await;
681
682                        match drop_dest_table_result {
683                            Ok(_) => Err(Error::new(
684                                ErrorKind::Unexpected,
685                                format!(
686                                    "{} Rolled back table creation for {}.{}.",
687                                    err_msg_src_table, dest_db_name, dest_table_name
688                                ),
689                            )),
690                            Err(_) => Err(Error::new(
691                                ErrorKind::Unexpected,
692                                format!(
693                                    "{} Failed to roll back table creation for {}.{}. Please clean up manually.",
694                                    err_msg_src_table, dest_db_name, dest_table_name
695                                ),
696                            )),
697                        }
698                    }
699                }
700            }
701        }
702    }

704    /// registers an existing table into the Glue Catalog.
705    ///
706    /// Converts the provided table identifier and metadata location into a
707    /// Glue-compatible table representation, and attempts to create the
708    /// corresponding table in the Glue Catalog.
709    ///
710    /// # Returns
711    /// Returns `Ok(Table)` if the table is successfully registered and loaded.
712    /// If the registration fails due to validation issues, existing table conflicts,
713    /// metadata problems, or errors during the registration or loading process,
714    /// an `Err(...)` is returned.
715    async fn register_table(
716        &self,
717        table_ident: &TableIdent,
718        metadata_location: String,
719    ) -> Result<Table> {
720        let db_name = validate_namespace(table_ident.namespace())?;
721        let table_name = table_ident.name();
722        let metadata = TableMetadata::read_from(&self.file_io, &metadata_location).await?;
723
724        let table_input = convert_to_glue_table(
725            table_name,
726            metadata_location.clone(),
727            &metadata,
728            metadata.properties(),
729            None,
730        )?;
731
732        let builder = self
733            .client
734            .0
735            .create_table()
736            .database_name(&db_name)
737            .table_input(table_input);
738        let builder = with_catalog_id!(builder, self.config);
739
740        builder.send().await.map_err(|e| {
741            let error = e.into_service_error();
742            match error {
743                CreateTableError::EntityNotFoundException(_) => Error::new(
744                    ErrorKind::NamespaceNotFound,
745                    format!("Database {db_name} does not exist"),
746                ),
747                CreateTableError::AlreadyExistsException(_) => Error::new(
748                    ErrorKind::TableAlreadyExists,
749                    format!("Table {table_ident} already exists"),
750                ),
751                _ => Error::new(
752                    ErrorKind::Unexpected,
753                    format!("Failed to register table {table_ident} due to AWS SDK error"),
754                ),
755            }
756            .with_source(anyhow!("aws sdk error: {:?}", error))
757        })?;
758
759        Ok(Table::builder()
760            .identifier(table_ident.clone())
761            .metadata_location(metadata_location)
762            .metadata(metadata)
763            .file_io(self.file_io())
764            .build()?)
765    }

767    async fn update_table(&self, commit: TableCommit) -> Result<Table> {
768        let table_ident = commit.identifier().clone();
769        let table_namespace = validate_namespace(table_ident.namespace())?;
770        let current_table = self.load_table(&table_ident).await?;
771        let current_metadata_location = current_table.metadata_location_result()?.to_string();
772
773        let staged_table = commit.apply(current_table)?;
774        let staged_metadata_location = staged_table.metadata_location_result()?;
775
776        // Write new metadata
777        staged_table
778            .metadata()
779            .write_to(staged_table.file_io(), staged_metadata_location)
780            .await?;
781
782        // Persist staged table to Glue
783        let builder = self
784            .client
785            .0
786            .update_table()
787            .database_name(table_namespace)
788            .set_skip_archive(Some(true)) // todo make this configurable
789            .table_input(convert_to_glue_table(
790                table_ident.name(),
791                staged_metadata_location.to_string(),
792                staged_table.metadata(),
793                staged_table.metadata().properties(),
794                Some(current_metadata_location),
795            )?);
796        let builder = with_catalog_id!(builder, self.config);
797        let _ = builder.send().await.map_err(|e| {
798            let error = e.into_service_error();
799            match error {
800                UpdateTableError::EntityNotFoundException(_) => Error::new(
801                    ErrorKind::TableNotFound,
802                    format!("Table {table_ident} is not found"),
803                ),
804                UpdateTableError::ConcurrentModificationException(_) => Error::new(
805                    ErrorKind::CatalogCommitConflicts,
806                    format!("Commit failed for table: {table_ident}"),
807                )
808                .with_retryable(true),
809                _ => Error::new(
810                    ErrorKind::Unexpected,
811                    format!("Operation failed for table: {table_ident} for hitting aws sdk error"),
812                ),
813            }
814            .with_source(anyhow!("aws sdk error: {:?}", error))
815        })?;
816
817        Ok(staged_table)
818    }
