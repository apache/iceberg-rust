32pub struct DataFusionEngine {
33    datafusion: DataFusion,
34}
35
36#[async_trait::async_trait]
37impl EngineRunner for DataFusionEngine {
38    async fn run_slt_file(&mut self, path: &Path) -> Result<()> {
39        let content = std::fs::read_to_string(path)
40            .with_context(|| format!("Failed to read slt file {:?}", path))
41            .map_err(|e| anyhow!(e))?;
42
43        self.datafusion
44            .run(content.as_str())
45            .await
46            .with_context(|| format!("Failed to run slt file {:?}", path))
47            .map_err(|e| anyhow!(e))?;
48
49        Ok(())
50    }
51}
52
53impl DataFusionEngine {
54    pub async fn new(config: TomlTable) -> Result<Self> {
55        let session_config = SessionConfig::new().with_target_partitions(4);
56        let ctx = SessionContext::new_with_config(session_config);
57        ctx.register_catalog("default", Self::create_catalog(&config).await?);
58
59        Ok(Self {
60            datafusion: DataFusion::new(ctx, PathBuf::from("testdata"), ProgressBar::new(100)),
61        })
62    }
