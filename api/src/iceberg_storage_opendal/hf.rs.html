33/// Repository type of a HuggingFace Hub repository.
34#[derive(Debug, Clone, Copy, PartialEq, Eq)]
35pub(crate) enum HfRepoType {
36    /// Model repository (`models/` prefix).
37    Model,
38    /// Dataset repository (`datasets/` prefix).
39    Dataset,
40    /// Spaces application repository (`spaces/` prefix).
41    Space,
42    /// XET-backed object-storage bucket (`buckets/` prefix).
43    Bucket,
44}
45
46impl HfRepoType {
47    /// Parse a repo-type keyword (singular or plural) into the corresponding variant.
48    fn parse(s: &str) -> Option<Self> {
49        match s.to_lowercase().replace(' ', "").as_str() {
50            "model" | "models" => Some(Self::Model),
51            "dataset" | "datasets" => Some(Self::Dataset),
52            "space" | "spaces" => Some(Self::Space),
53            "bucket" | "buckets" => Some(Self::Bucket),
54            _ => None,
55        }
56    }
57
58    fn canonical(self) -> &'static str {
59        match self {
60            Self::Model => "models",
61            Self::Dataset => "datasets",
62            Self::Space => "spaces",
63            Self::Bucket => "buckets",
64        }
65    }
66}

68/// Parsed HuggingFace URI: `hf://<repo_type>/<repo_id>[@<revision>][/<path>]`.
69///
70/// `repo_type` must be explicitly specified — there is no implicit default.
71/// Only the fields required by this crate are stored; revision is consumed
72/// during parsing but not retained.
73#[derive(Debug, Clone, PartialEq, Eq)]
74pub(crate) struct HfUri {
75    pub repo_type: HfRepoType,
76    /// e.g. `"user/my-repo"`.
77    pub repo_id: String,
78    /// Path within the repository, e.g. `"train/data.parquet"`. Empty at repo root.
79    pub path: String,
80}
81
82impl HfUri {
83    /// Parse a full `hf://…` URI or the bare path portion (without scheme).
84    /// Returns `None` if the URI does not begin with a recognized repo-type prefix
85    /// (`models/`, `datasets/`, `spaces/`, or `buckets/`).
86    pub(crate) fn parse(full_uri: &str) -> Option<Self> {
87        let s = full_uri.strip_prefix("hf://").unwrap_or(full_uri);
88        if s.is_empty() {
89            return None;
90        }
91
92        // Require an explicit repo_type prefix — no implicit default.
93        let (first, rest) = s.split_once('/')?;
94        let repo_type = HfRepoType::parse(first)?;
95        let s = rest;
96
97        // Remaining: `<repo_id>[@<revision>][/<path_in_repo>]`
98        let (repo_id, path) = if s.contains('/') {
99            // Check if `@` appears in the first two slash-segments (the repo_id portion).
100            // This distinguishes "user/repo@rev/file" from "user/repo/path/@file".
101            let first_two = s.splitn(3, '/').take(2).collect::<Vec<_>>().join("/");
102            if first_two.contains('@') {
103                let (repo_id, rev_and_path) = s.split_once('@').unwrap();
104                let rev_and_path = rev_and_path.replace("%2F", "/");
105                (repo_id.to_string(), path_after_revision(&rev_and_path))
106            } else {
107                let segs: Vec<_> = s.splitn(3, '/').collect();
108                let repo_id = format!("{}/{}", segs[0], segs[1]);
109                let path = segs.get(2).copied().unwrap_or("").to_string();
110                (repo_id, path)
111            }
112        } else if let Some((repo_id, _)) = s.split_once('@') {
113            (repo_id.to_string(), String::new())
114        } else {
115            (s.to_string(), String::new())
116        };
117
118        Some(Self {
119            repo_type,
120            repo_id,
121            path,
122        })
123    }
124}

126/// Given the string after `@`, extract the path-in-repo, correctly skipping
127/// multi-segment special refs (`refs/convert/parquet`, `refs/pr/N`).
128/// These are the only two multi-segment special ref prefixes in HF's git model.
129fn path_after_revision(rev_and_path: &str) -> String {
130    if !rev_and_path.contains('/') {
131        return String::new();
132    }
133    if let Some(rest) = rev_and_path.strip_prefix("refs/convert/") {
134        return rest
135            .find('/')
136            .map_or(String::new(), |i| rest[i + 1..].to_string());
137    }
138    if let Some(rest) = rev_and_path.strip_prefix("refs/pr/") {
139        return rest
140            .find('/')
141            .map_or(String::new(), |i| rest[i + 1..].to_string());
142    }
143    rev_and_path
144        .split_once('/')
145        .map(|(_, path)| path.to_string())
146        .unwrap_or_default()
147}

153/// Parse iceberg `StorageConfig` properties into an opendal [`opendal::services::HfConfig`].
154pub(crate) fn hf_config_parse(m: HashMap<String, String>) -> Result<opendal::services::HfConfig> {
155    let mut cfg = opendal::services::HfConfig::default();
156    if let Some(token) = m.get(HF_TOKEN) {
157        cfg.token = Some(token.clone());
158    }
159    if let Some(endpoint) = m.get(HF_ENDPOINT) {
160        cfg.endpoint = Some(endpoint.clone());
161    }
162    if let Some(revision) = m.get(HF_REVISION) {
163        cfg.revision = Some(revision.clone());
164    }
165    Ok(cfg)
166}

168/// Build an [`Operator`] for the given `hf://…` path and return it together with
169/// the relative path-in-repo.
170///
171/// URI parsing is delegated to opendal's [`HfConfig::from_uri`]. The base config
172/// provides fallback values for `revision` and `endpoint`; the `token` is always
173/// taken from the base config and never from the URI.
174pub(crate) fn hf_config_build<'a>(
175    cfg: &opendal::services::HfConfig,
176    path: &'a str,
177) -> Result<(Operator, &'a str)> {
178    let uri = OperatorUri::new(path, Vec::<(String, String)>::new()).map_err(|e| {
179        Error::new(ErrorKind::DataInvalid, format!("Invalid hf url: {path}")).with_source(e)
180    })?;
181
182    let mut hf_cfg = opendal::services::HfConfig::from_uri(&uri).map_err(|e| {
183        Error::new(ErrorKind::DataInvalid, format!("Invalid hf url: {path}")).with_source(e)
184    })?;
185
186    // Token must come from config only, never from the URI.
187    hf_cfg.token = cfg.token.clone();
188
189    if hf_cfg.endpoint.is_none() {
190        hf_cfg.endpoint = cfg.endpoint.clone();
191    }
192    if hf_cfg.revision.is_none() {
193        hf_cfg.revision = cfg.revision.clone();
194    }
195
196    let parsed = HfUri::parse(path)
197        .ok_or_else(|| Error::new(ErrorKind::DataInvalid, format!("Invalid hf url: {path}")))?;
198    let relative_path = &path[path.len() - parsed.path.len()..];
199
200    let op = Operator::from_config(hf_cfg)
201        .map_err(from_opendal_error)?
202        .finish();
203    Ok((op, relative_path))
204}

206/// Returns a stable cache key for `delete_stream` batching: `"<repo_type>/<repo_id>"`
207/// (e.g. `"buckets/user/my-repo"`), without revision.
208/// Repo type is included so bucket and dataset paths to the same repo use separate operators.
209/// Falls back to the full path so that unparsable paths never share an operator accidentally.
210pub(crate) fn hf_batch_key(path: &str) -> String {
211    HfUri::parse(path)
212        .map(|u| format!("{}/{}", u.repo_type.canonical(), u.repo_id))
213        .unwrap_or_else(|| path.to_string())
214}

319    #[test]

321        assert!(HfUri::parse("hf://").is_none());
322        // bare repo-type, no repo_id
323        assert!(HfUri::parse("hf://datasets").is_none());
324        // missing repo-type prefix
