18use std::any::Any;
19use std::collections::HashMap;
20use std::path::Path;
21use std::sync::Arc;
22
23use anyhow::anyhow;
24use datafusion::catalog::{CatalogProvider, CatalogProviderList};
25use fs_err::read_to_string;
26use iceberg::CatalogBuilder;
27use iceberg_catalog_rest::RestCatalogBuilder;
28use iceberg_datafusion::IcebergCatalogProvider;
29use toml::{Table as TomlTable, Value};
30
31const CONFIG_NAME_CATALOGS: &str = "catalogs";
32
33#[derive(Debug)]
34pub struct IcebergCatalogList {
35    catalogs: HashMap<String, Arc<IcebergCatalogProvider>>,
36}
37
38impl IcebergCatalogList {
39    pub async fn parse(path: &Path) -> anyhow::Result<Self> {
40        let toml_table: TomlTable = toml::from_str(&read_to_string(path)?)?;
41        Self::parse_table(&toml_table).await
42    }
43
44    pub async fn parse_table(configs: &TomlTable) -> anyhow::Result<Self> {
45        if let Value::Array(catalogs_config) =
46            configs.get(CONFIG_NAME_CATALOGS).ok_or_else(|| {
47                anyhow::Error::msg(format!("{CONFIG_NAME_CATALOGS} entry not found in config"))
48            })?
49        {
50            let mut catalogs = HashMap::with_capacity(catalogs_config.len());
51            for config in catalogs_config {
52                if let Value::Table(table_config) = config {
53                    let (name, catalog_provider) =
54                        IcebergCatalogList::parse_one(table_config).await?;
55                    catalogs.insert(name, catalog_provider);
56                } else {
57                    return Err(anyhow!("{CONFIG_NAME_CATALOGS} entry must be a table"));
58                }
59            }
60            Ok(Self { catalogs })
61        } else {
62            Err(anyhow!("{CONFIG_NAME_CATALOGS} must be an array of table!"))
63        }
64    }
65
66    async fn parse_one(
67        config: &TomlTable,
68    ) -> anyhow::Result<(String, Arc<IcebergCatalogProvider>)> {
69        let name = config
70            .get("name")
71            .ok_or_else(|| anyhow::anyhow!("name not found for catalog"))?
72            .as_str()
73            .ok_or_else(|| anyhow::anyhow!("name is not string"))?;
74
75        let r#type = config
76            .get("type")
77            .ok_or_else(|| anyhow::anyhow!("type not found for catalog"))?
78            .as_str()
79            .ok_or_else(|| anyhow::anyhow!("type is not string"))?;
80
81        if r#type != "rest" {
82            return Err(anyhow::anyhow!("Only rest catalog is supported for now!"));
83        }
84
85        let catalog_config = config
86            .get("config")
87            .ok_or_else(|| anyhow::anyhow!("config not found for catalog {name}"))?
88            .as_table()
89            .ok_or_else(|| anyhow::anyhow!("config is not table for catalog {name}"))?;
90
91        // parse all config into props
92        let mut props = HashMap::with_capacity(catalog_config.len());
93        for (key, value) in catalog_config {
94            let value_str = value
95                .as_str()
96                .ok_or_else(|| anyhow::anyhow!("props {key} is not string"))?;
97            props.insert(key.to_string(), value_str.to_string());
98        }
99        let catalog = RestCatalogBuilder::default().load(name, props).await?;
100
101        Ok((
102            name.to_string(),
103            Arc::new(IcebergCatalogProvider::try_new(Arc::new(catalog)).await?),
104        ))
105    }
106}
107
108impl CatalogProviderList for IcebergCatalogList {
109    fn as_any(&self) -> &dyn Any {
110        self
111    }
112
113    fn register_catalog(
114        &self,
115        _name: String,
116        _catalog: Arc<dyn CatalogProvider>,
117    ) -> Option<Arc<dyn CatalogProvider>> {
118        tracing::error!("Registering catalog is not supported yet");
119        None
120    }
121
122    fn catalog_names(&self) -> Vec<String> {
123        self.catalogs.keys().cloned().collect()
124    }
125
126    fn catalog(&self, name: &str) -> Option<Arc<dyn CatalogProvider>> {
127        self.catalogs
128            .get(name)
129            .map(|c| c.clone() as Arc<dyn CatalogProvider>)
130    }
131}
