18use std::any::Any;
19use std::collections::HashMap;
20use std::path::Path;
21use std::sync::Arc;
22
23use anyhow::anyhow;
24use datafusion::catalog::{CatalogProvider, CatalogProviderList};
25use fs_err::read_to_string;
26use iceberg::CatalogBuilder;
27use iceberg::memory::MemoryCatalogBuilder;
28use iceberg_catalog_rest::RestCatalogBuilder;
29use iceberg_datafusion::IcebergCatalogProvider;
30use toml::{Table as TomlTable, Value};
31
32const CONFIG_NAME_CATALOGS: &str = "catalogs";
33
34#[derive(Debug)]
35pub struct IcebergCatalogList {
36    catalogs: HashMap<String, Arc<IcebergCatalogProvider>>,
37}
38
39impl IcebergCatalogList {
40    pub async fn parse(path: &Path) -> anyhow::Result<Self> {
41        let toml_table: TomlTable = toml::from_str(&read_to_string(path)?)?;
42        Self::parse_table(&toml_table).await
43    }
44
45    pub async fn parse_table(configs: &TomlTable) -> anyhow::Result<Self> {
46        if let Value::Array(catalogs_config) =
47            configs.get(CONFIG_NAME_CATALOGS).ok_or_else(|| {
48                anyhow::Error::msg(format!("{CONFIG_NAME_CATALOGS} entry not found in config"))
49            })?
50        {
51            let mut catalogs = HashMap::with_capacity(catalogs_config.len());
52            for config in catalogs_config {
53                if let Value::Table(table_config) = config {
54                    let (name, catalog_provider) =
55                        IcebergCatalogList::parse_one(table_config).await?;
56                    catalogs.insert(name, catalog_provider);
57                } else {
58                    return Err(anyhow!("{CONFIG_NAME_CATALOGS} entry must be a table"));
59                }
60            }
61            Ok(Self { catalogs })
62        } else {
63            Err(anyhow!("{CONFIG_NAME_CATALOGS} must be an array of table!"))
64        }
65    }
66
67    async fn parse_one(
68        config: &TomlTable,
69    ) -> anyhow::Result<(String, Arc<IcebergCatalogProvider>)> {
70        let name = config
71            .get("name")
72            .ok_or_else(|| anyhow::anyhow!("name not found for catalog"))?
73            .as_str()
74            .ok_or_else(|| anyhow::anyhow!("name is not string"))?;
75
76        let r#type = config
77            .get("type")
78            .ok_or_else(|| anyhow::anyhow!("type not found for catalog"))?
79            .as_str()
80            .ok_or_else(|| anyhow::anyhow!("type is not string"))?;
81
82        let catalog_config = config
83            .get("config")
84            .ok_or_else(|| anyhow::anyhow!("config not found for catalog {name}"))?
85            .as_table()
86            .ok_or_else(|| anyhow::anyhow!("config is not table for catalog {name}"))?;
87
88        // parse all config into props
89        let mut props = HashMap::with_capacity(catalog_config.len());
90        for (key, value) in catalog_config {
91            let value_str = value
92                .as_str()
93                .ok_or_else(|| anyhow::anyhow!("props {key} is not string"))?;
94            props.insert(key.to_string(), value_str.to_string());
95        }
96
97        // Create catalog based on type using the appropriate builder
98        let catalog: Arc<dyn iceberg::Catalog> = match r#type {
99            "rest" => Arc::new(RestCatalogBuilder::default().load(name, props).await?),
100            "memory" => Arc::new(MemoryCatalogBuilder::default().load(name, props).await?),
101            _ => {
102                return Err(anyhow::anyhow!(
103                    "Unsupported catalog type: '{type}'. Supported types: rest, memory"
104                ));
105            }
106        };
107
108        Ok((
109            name.to_string(),
110            Arc::new(IcebergCatalogProvider::try_new(catalog).await?),
111        ))
112    }
113}
114
115impl CatalogProviderList for IcebergCatalogList {
116    fn as_any(&self) -> &dyn Any {
117        self
118    }
119
120    fn register_catalog(
121        &self,
122        _name: String,
123        _catalog: Arc<dyn CatalogProvider>,
124    ) -> Option<Arc<dyn CatalogProvider>> {
125        tracing::error!("Registering catalog is not supported yet");
126        None
127    }
128
129    fn catalog_names(&self) -> Vec<String> {
130        self.catalogs.keys().cloned().collect()
131    }
132
133    fn catalog(&self, name: &str) -> Option<Arc<dyn CatalogProvider>> {
134        self.catalogs
135            .get(name)
136            .map(|c| c.clone() as Arc<dyn CatalogProvider>)
137    }
138}

140#[cfg(test)]
141mod tests {
142    use super::*;
143
144    #[tokio::test]
145    async fn test_parse_memory_catalog() {
146        let config = r#"
147            [[catalogs]]
148            name = "test_memory"
149            type = "memory"
150            [catalogs.config]
151            warehouse = "/tmp/test-warehouse"
152        "#;
153
154        let toml_table: TomlTable = toml::from_str(config).unwrap();
155        let catalog_list = IcebergCatalogList::parse_table(&toml_table).await.unwrap();
156
157        assert!(
158            catalog_list
159                .catalog_names()
160                .contains(&"test_memory".to_string())
161        );
162        assert!(catalog_list.catalog("test_memory").is_some());
163    }
164
165    #[tokio::test]
166    async fn test_parse_unsupported_catalog_type() {
167        let config = r#"
168            [[catalogs]]
169            name = "test_hive"
170            type = "hive"
171            [catalogs.config]
172            uri = "thrift://localhost:9083"
173        "#;
174
175        let toml_table: TomlTable = toml::from_str(config).unwrap();
176        let result = IcebergCatalogList::parse_table(&toml_table).await;
177
178        assert!(result.is_err());
179        let err_msg = result.unwrap_err().to_string();
180        assert!(err_msg.contains("Unsupported catalog type"));
181        assert!(err_msg.contains("hive"));
182        assert!(err_msg.contains("rest, memory"));
183    }
184
185    #[tokio::test]
186    async fn test_catalog_names() {
187        let config = r#"
188            [[catalogs]]
189            name = "catalog_one"
190            type = "memory"
191            [catalogs.config]
192            warehouse = "/tmp/warehouse1"
193
194            [[catalogs]]
195            name = "catalog_two"
196            type = "memory"
197            [catalogs.config]
198            warehouse = "/tmp/warehouse2"
199        "#;
200
201        let toml_table: TomlTable = toml::from_str(config).unwrap();
202        let catalog_list = IcebergCatalogList::parse_table(&toml_table).await.unwrap();
203
204        let names = catalog_list.catalog_names();
205        assert_eq!(names.len(), 2);
206        assert!(names.contains(&"catalog_one".to_string()));
207        assert!(names.contains(&"catalog_two".to_string()));
208    }
209}
