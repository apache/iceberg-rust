35// pyo3 0.28's CapsuleName only exposes `unsafe fn as_cstr() -> &CStr`,
36// so we accept &CStr to allow direct comparison without UTF-8 validation.
37pub(crate) fn validate_pycapsule(capsule: &Bound<PyCapsule>, name: &CStr) -> PyResult<()> {
38    let capsule_name = capsule.name()?;
39    match capsule_name {
40        None => Err(PyValueError::new_err(
41            "Expected PyCapsule to have name set.",
42        )),
43        Some(capsule_name) if unsafe { capsule_name.as_cstr() } != name => {
44            Err(PyValueError::new_err("PyCapsule name mismatch"))
45        }
46        _ => Ok(()),
47    }
48}

50pub(crate) fn ffi_logical_codec_from_pycapsule(
51    obj: Bound<PyAny>,
52) -> PyResult<FFI_LogicalExtensionCodec> {
53    let attr_name = "__datafusion_logical_extension_codec__";
54    let capsule = if obj.hasattr(attr_name)? {
55        obj.getattr(attr_name)?.call0()?
56    } else {
57        obj
58    };
59
60    let capsule_name = c"datafusion_logical_extension_codec";
61    let capsule = capsule.cast::<PyCapsule>()?;
62    validate_pycapsule(capsule, capsule_name)?;
63
64    let ptr = capsule.pointer_checked(Some(capsule_name))?;
65    let codec = unsafe { &*(ptr.as_ptr() as *const FFI_LogicalExtensionCodec) };
66    Ok(codec.clone())
67}

69#[pyclass(name = "IcebergDataFusionTable")]
70pub struct PyIcebergDataFusionTable {
71    inner: Arc<IcebergStaticTableProvider>,
72}
73
74#[pymethods]
75impl PyIcebergDataFusionTable {
76    #[new]
77    fn new(
78        identifier: Vec<String>,
79        metadata_location: String,
80        file_io_properties: Option<HashMap<String, String>>,
81    ) -> PyResult<Self> {
82        let runtime = runtime();
83
84        let provider = runtime.block_on(async {
85            let table_ident = TableIdent::from_strs(identifier)
86                .map_err(|e| PyRuntimeError::new_err(format!("Invalid table identifier: {e}")))?;
87
88            let factory = Arc::new(OpenDalResolvingStorageFactory::new());
89
90            let mut builder = FileIOBuilder::new(factory);
91
92            if let Some(props) = file_io_properties {
93                builder = builder.with_props(props);
94            }
95
96            let file_io = builder.build();
97
98            let static_table =
99                StaticTable::from_metadata_file(&metadata_location, table_ident, file_io)
100                    .await
101                    .map_err(|e| {
102                        PyRuntimeError::new_err(format!("Failed to load static table: {e}"))
103                    })?;
104
105            let table = static_table.into_table();
106
107            IcebergStaticTableProvider::try_new_from_table(table)
108                .await
109                .map_err(|e| {
110                    PyRuntimeError::new_err(format!("Failed to create table provider: {e}"))
111                })
112        })?;
113
114        Ok(Self {
115            inner: Arc::new(provider),
116        })
117    }
118
119    fn __datafusion_table_provider__<'py>(
120        &self,
121        py: Python<'py>,
122        session: Bound<'py, PyAny>,
123    ) -> PyResult<Bound<'py, PyCapsule>> {
124        let capsule_name = CString::new("datafusion_table_provider").unwrap();
125
126        let logical_codec = ffi_logical_codec_from_pycapsule(session)?;
127
128        let ffi_provider = FFI_TableProvider::new_with_ffi_codec(
129            self.inner.clone(),
130            false,
131            Some(runtime()),
132            logical_codec,
133        );
134
135        PyCapsule::new(py, ffi_provider, Some(capsule_name))
136    }
137}

139pub fn register_module(py: Python<'_>, m: &Bound<'_, PyModule>) -> PyResult<()> {
140    let this = PyModule::new(py, "datafusion")?;
141
142    this.add_class::<PyIcebergDataFusionTable>()?;
143
144    m.add_submodule(&this)?;
145    py.import("sys")?
146        .getattr("modules")?
147        .set_item("pyiceberg_core.datafusion", this)?;
148
149    Ok(())
150}
