18use std::collections::HashMap;

20use aws_config::{BehaviorVersion, Region, SdkConfig};
21use aws_sdk_s3tables::config::Credentials;
22use iceberg::{Error, ErrorKind, Result};
23use uuid::Uuid;

25/// Property aws profile name
26pub const AWS_PROFILE_NAME: &str = "profile_name";
27/// Property aws region
28pub const AWS_REGION_NAME: &str = "region_name";
29/// Property aws access key
30pub const AWS_ACCESS_KEY_ID: &str = "aws_access_key_id";
31/// Property aws secret access key
32pub const AWS_SECRET_ACCESS_KEY: &str = "aws_secret_access_key";
33/// Property aws session token
34pub const AWS_SESSION_TOKEN: &str = "aws_session_token";
35
36/// Creates an aws sdk configuration based on
37/// provided properties and an optional endpoint URL.
38pub(crate) async fn create_sdk_config(
39    properties: &HashMap<String, String>,
40    endpoint_url: Option<String>,
41) -> SdkConfig {
42    let mut config = aws_config::defaults(BehaviorVersion::latest());
43
44    if properties.is_empty() {
45        return config.load().await;
46    }
47
48    if let Some(endpoint_url) = endpoint_url {
49        config = config.endpoint_url(endpoint_url);
50    }
51
52    if let (Some(access_key), Some(secret_key)) = (
53        properties.get(AWS_ACCESS_KEY_ID),
54        properties.get(AWS_SECRET_ACCESS_KEY),
55    ) {
56        let session_token = properties.get(AWS_SESSION_TOKEN).cloned();
57        let credentials_provider =
58            Credentials::new(access_key, secret_key, session_token, None, "properties");
59
60        config = config.credentials_provider(credentials_provider)
61    };
62
63    if let Some(profile_name) = properties.get(AWS_PROFILE_NAME) {
64        config = config.profile_name(profile_name);
65    }
66
67    if let Some(region_name) = properties.get(AWS_REGION_NAME) {
68        let region = Region::new(region_name.clone());
69        config = config.region(region);
70    }
71
72    config.load().await
73}

75/// Create metadata location from `location` and `version`
76pub(crate) fn create_metadata_location(
77    warehouse_location: impl AsRef<str>,
78    version: i32,
79) -> Result<String> {
80    if version < 0 {
81        return Err(Error::new(
82            ErrorKind::DataInvalid,
83            format!(
84                "Table metadata version: '{}' must be a non-negative integer",
85                version
86            ),
87        ));
88    };
89
90    let version = format!("{:0>5}", version);
91    let id = Uuid::new_v4();
92    let metadata_location = format!(
93        "{}/metadata/{}-{}.metadata.json",
94        warehouse_location.as_ref(),
95        version,
96        id
97    );
98
99    Ok(metadata_location)
100}
