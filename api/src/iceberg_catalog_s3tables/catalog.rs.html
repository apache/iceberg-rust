37/// S3Tables table bucket ARN property
38pub const S3TABLES_CATALOG_PROP_TABLE_BUCKET_ARN: &str = "table_bucket_arn";
39/// S3Tables endpoint URL property
40pub const S3TABLES_CATALOG_PROP_ENDPOINT_URL: &str = "endpoint_url";

42/// S3Tables catalog configuration.
43#[derive(Debug)]
44struct S3TablesCatalogConfig {
45    /// Catalog name.
46    name: Option<String>,
47    /// Unlike other buckets, S3Tables bucket is not a physical bucket, but a virtual bucket
48    /// that is managed by s3tables. We can't directly access the bucket with path like
49    /// s3://{bucket_name}/{file_path}, all the operations are done with respect of the bucket
50    /// ARN.
51    table_bucket_arn: String,
52    /// Endpoint URL for the catalog.
53    endpoint_url: Option<String>,
54    /// Optional pre-configured AWS SDK client for S3Tables.
55    client: Option<aws_sdk_s3tables::Client>,
56    /// Properties for the catalog. The available properties are:
57    /// - `profile_name`: The name of the AWS profile to use.
58    /// - `region_name`: The AWS region to use.
59    /// - `aws_access_key_id`: The AWS access key ID to use.
60    /// - `aws_secret_access_key`: The AWS secret access key to use.
61    /// - `aws_session_token`: The AWS session token to use.
62    props: HashMap<String, String>,
63}

117impl CatalogBuilder for S3TablesCatalogBuilder {
118    type C = S3TablesCatalog;
119
120    fn load(
121        mut self,
122        name: impl Into<String>,
123        props: HashMap<String, String>,
124    ) -> impl Future<Output = Result<Self::C>> + Send {
125        let catalog_name = name.into();
126        self.0.name = Some(catalog_name.clone());
127
128        if props.contains_key(S3TABLES_CATALOG_PROP_TABLE_BUCKET_ARN) {
129            self.0.table_bucket_arn = props
130                .get(S3TABLES_CATALOG_PROP_TABLE_BUCKET_ARN)
131                .cloned()
132                .unwrap_or_default();
133        }
134
135        if props.contains_key(S3TABLES_CATALOG_PROP_ENDPOINT_URL) {
136            self.0.endpoint_url = props.get(S3TABLES_CATALOG_PROP_ENDPOINT_URL).cloned();
137        }
138
139        // Collect other remaining properties
140        self.0.props = props
141            .into_iter()
142            .filter(|(k, _)| {
143                k != S3TABLES_CATALOG_PROP_TABLE_BUCKET_ARN
144                    && k != S3TABLES_CATALOG_PROP_ENDPOINT_URL
145            })
146            .collect();
147
148        async move {
149            if catalog_name.trim().is_empty() {
150                Err(Error::new(
151                    ErrorKind::DataInvalid,
152                    "Catalog name cannot be empty",
153                ))
154            } else if self.0.table_bucket_arn.is_empty() {
155                Err(Error::new(
156                    ErrorKind::DataInvalid,
157                    "Table bucket ARN is required",
158                ))
159            } else {
160                S3TablesCatalog::new(self.0).await
161            }
162        }
163    }
164}

174impl S3TablesCatalog {
175    /// Creates a new S3Tables catalog.
176    async fn new(config: S3TablesCatalogConfig) -> Result<Self> {
177        let s3tables_client = if let Some(client) = config.client.clone() {
178            client
179        } else {
180            let aws_config = create_sdk_config(&config.props, config.endpoint_url.clone()).await;
181            aws_sdk_s3tables::Client::new(&aws_config)
182        };
183
184        let file_io = FileIOBuilder::new("s3").with_props(&config.props).build()?;
185
186        Ok(Self {
187            config,
188            s3tables_client,
189            file_io,
190        })
191    }
192}

196    /// List namespaces from s3tables catalog.
197    ///
198    /// S3Tables doesn't support nested namespaces. If parent is provided, it will
199    /// return an empty list.
200    async fn list_namespaces(
201        &self,
202        parent: Option<&NamespaceIdent>,
203    ) -> Result<Vec<NamespaceIdent>> {
204        if parent.is_some() {
205            return Ok(vec![]);
206        }
207
208        let mut result = Vec::new();
209        let mut continuation_token = None;
210        loop {
211            let mut req = self
212                .s3tables_client
213                .list_namespaces()
214                .table_bucket_arn(self.config.table_bucket_arn.clone());
215            if let Some(token) = continuation_token {
216                req = req.continuation_token(token);
217            }
218            let resp = req.send().await.map_err(from_aws_sdk_error)?;
219            for ns in resp.namespaces() {
220                result.push(NamespaceIdent::from_vec(ns.namespace().to_vec())?);
221            }
222            continuation_token = resp.continuation_token().map(|s| s.to_string());
223            if continuation_token.is_none() {
224                break;
225            }
226        }
227        Ok(result)
228    }

384    /// Creates a new table within a specified namespace.
385    ///
386    /// Attempts to create a table defined by the `creation` parameter. The metadata
387    /// location is generated by the s3tables catalog, looks like:
388    ///
389    /// s3://{RANDOM WAREHOUSE LOCATION}/metadata/{VERSION}-{UUID}.metadata.json
390    ///
391    /// We have to get this random warehouse location after the table is created.
392    ///
393    /// This function can return an error in the following situations:
394    /// - If the location of the table is set by user, identified by a specific
395    /// `DataInvalid` variant.
396    /// - Errors from the underlying database creation process, converted using
397    /// `from_aws_sdk_error`.
398    async fn create_table(
399        &self,
400        namespace: &NamespaceIdent,
401        mut creation: TableCreation,
402    ) -> Result<Table> {
403        let table_ident = TableIdent::new(namespace.clone(), creation.name.clone());
404
405        // create table
406        let create_resp: CreateTableOutput = self
407            .s3tables_client
408            .create_table()
409            .table_bucket_arn(self.config.table_bucket_arn.clone())
410            .namespace(namespace.to_url_string())
411            .format(OpenTableFormat::Iceberg)
412            .name(table_ident.name())
413            .send()
414            .await
415            .map_err(from_aws_sdk_error)?;
416
417        // prepare metadata location. the warehouse location is generated by s3tables catalog,
418        // which looks like: s3://e6c9bf20-991a-46fb-kni5xs1q2yxi3xxdyxzjzigdeop1quse2b--table-s3
419        let metadata_location = match &creation.location {
420            Some(_) => {
421                return Err(Error::new(
422                    ErrorKind::DataInvalid,
423                    "The location of the table is generated by s3tables catalog, can't be set by user.",
424                ));
425            }
426            None => {
427                let get_resp: GetTableOutput = self
428                    .s3tables_client
429                    .get_table()
430                    .table_bucket_arn(self.config.table_bucket_arn.clone())
431                    .namespace(namespace.to_url_string())
432                    .name(table_ident.name())
433                    .send()
434                    .await
435                    .map_err(from_aws_sdk_error)?;
436                let warehouse_location = get_resp.warehouse_location().to_string();
437                MetadataLocation::new_with_table_location(warehouse_location).to_string()
438            }
439        };
440
441        // write metadata to file
442        creation.location = Some(metadata_location.clone());
443        let metadata = TableMetadataBuilder::from_table_creation(creation)?
444            .build()?
445            .metadata;
446        metadata.write_to(&self.file_io, &metadata_location).await?;
447
448        // update metadata location
449        self.s3tables_client
450            .update_table_metadata_location()
451            .table_bucket_arn(self.config.table_bucket_arn.clone())
452            .namespace(namespace.to_url_string())
453            .name(table_ident.name())
454            .metadata_location(metadata_location.clone())
455            .version_token(create_resp.version_token())
456            .send()
457            .await
458            .map_err(from_aws_sdk_error)?;
459
460        let table = Table::builder()
461            .identifier(table_ident)
462            .metadata_location(metadata_location)
463            .metadata(metadata)
464            .file_io(self.file_io.clone())
465            .build()?;
466        Ok(table)
467    }

469    /// Loads an existing table from the s3tables catalog.
470    ///
471    /// Retrieves the metadata location of the specified table and constructs a
472    /// `Table` object with the retrieved metadata.
473    ///
474    /// This function can return an error in the following situations:
475    /// - If the table does not have a metadata location, identified by a specific
476    /// `Unexpected` variant.
477    /// - Errors from the underlying database query process, converted using
478    /// `from_aws_sdk_error`.
479    async fn load_table(&self, table_ident: &TableIdent) -> Result<Table> {
480        let req = self
481            .s3tables_client
482            .get_table()
483            .table_bucket_arn(self.config.table_bucket_arn.clone())
484            .namespace(table_ident.namespace().to_url_string())
485            .name(table_ident.name());
486        let resp: GetTableOutput = req.send().await.map_err(from_aws_sdk_error)?;
487
488        // when a table is created, it's possible that the metadata location is not set.
489        let metadata_location = resp.metadata_location().ok_or_else(|| {
490            Error::new(
491                ErrorKind::Unexpected,
492                format!(
493                    "Table {} does not have metadata location",
494                    table_ident.name()
495                ),
496            )
497        })?;
498        let metadata = TableMetadata::read_from(&self.file_io, metadata_location).await?;
499
500        let table = Table::builder()
501            .identifier(table_ident.clone())
502            .metadata(metadata)
503            .metadata_location(metadata_location)
504            .file_io(self.file_io.clone())
505            .build()?;
506        Ok(table)
507    }

698    async fn test_s3tables_create_delete_table() {
699        let catalog = match load_s3tables_catalog_from_env().await {
700            Ok(Some(catalog)) => catalog,
701            Ok(None) => return,
702            Err(e) => panic!("Error loading catalog: {}", e),
703        };
704
705        let creation = {
706            let schema = Schema::builder()
707                .with_schema_id(0)
708                .with_fields(vec![
709                    NestedField::required(1, "foo", Type::Primitive(PrimitiveType::Int)).into(),
710                    NestedField::required(2, "bar", Type::Primitive(PrimitiveType::String)).into(),
711                ])
712                .build()
713                .unwrap();
714            TableCreation::builder()
715                .name("test_s3tables_create_delete_table".to_string())
716                .properties(HashMap::new())
717                .schema(schema)
718                .build()
719        };
720
721        let namespace = NamespaceIdent::new("test_s3tables_create_delete_table".to_string());
722        let table_ident = TableIdent::new(
723            namespace.clone(),
724            "test_s3tables_create_delete_table".to_string(),
725        );
726        catalog.drop_namespace(&namespace).await.ok();
727        catalog.drop_table(&table_ident).await.ok();
728
729        catalog
730            .create_namespace(&namespace, HashMap::new())
731            .await
732            .unwrap();
733        catalog.create_table(&namespace, creation).await.unwrap();
734        assert!(catalog.table_exists(&table_ident).await.unwrap());
735        catalog.drop_table(&table_ident).await.unwrap();
736        assert!(!catalog.table_exists(&table_ident).await.unwrap());
737        catalog.drop_namespace(&namespace).await.unwrap();
738    }
