66impl RestCatalogConfig {
67    fn url_prefixed(&self, parts: &[&str]) -> String {
68        [&self.uri, PATH_V1]
69            .into_iter()
70            .chain(self.props.get("prefix").map(|s| &**s))
71            .chain(parts.iter().cloned())
72            .join("/")
73    }
74
75    fn config_endpoint(&self) -> String {
76        [&self.uri, PATH_V1, "config"].join("/")
77    }
78
79    pub(crate) fn get_token_endpoint(&self) -> String {
80        if let Some(oauth2_uri) = self.props.get("oauth2-server-uri") {
81            oauth2_uri.to_string()
82        } else {
83            [&self.uri, PATH_V1, "oauth", "tokens"].join("/")
84        }
85    }
86
87    fn namespaces_endpoint(&self) -> String {
88        self.url_prefixed(&["namespaces"])
89    }

91    fn namespace_endpoint(&self, ns: &NamespaceIdent) -> String {
92        self.url_prefixed(&["namespaces", &ns.to_url_string()])
93    }
94
95    fn tables_endpoint(&self, ns: &NamespaceIdent) -> String {
96        self.url_prefixed(&["namespaces", &ns.to_url_string(), "tables"])
97    }
98
99    fn rename_table_endpoint(&self) -> String {
100        self.url_prefixed(&["tables", "rename"])
101    }

103    fn table_endpoint(&self, table: &TableIdent) -> String {
104        self.url_prefixed(&[
105            "namespaces",
106            &table.namespace.to_url_string(),

124    /// Get the credentials from the config. The client can use these credentials to fetch a new
125    /// token.
126    ///
127    /// ## Output
128    ///
129    /// - `None`: No credential is set.
130    /// - `Some(None, client_secret)`: No client_id is set, use client_secret directly.
131    /// - `Some(Some(client_id), client_secret)`: Both client_id and client_secret are set.
132    pub(crate) fn credential(&self) -> Option<(Option<String>, String)> {
133        let cred = self.props.get("credential")?;
134
135        match cred.split_once(':') {
136            Some((client_id, client_secret)) => {
137                Some((Some(client_id.to_string()), client_secret.to_string()))
138            }
139            None => Some((None, cred.to_string())),
140        }
141    }

143    /// Get the extra headers from config, which includes:
144    ///
145    /// - `content-type`
146    /// - `x-client-version`
147    /// - `user-agent`
148    /// - All headers specified by `header.xxx` in props.
149    pub(crate) fn extra_headers(&self) -> Result<HeaderMap> {
150        let mut headers = HeaderMap::from_iter([
151            (
152                header::CONTENT_TYPE,
153                HeaderValue::from_static("application/json"),
154            ),
155            (
156                HeaderName::from_static("x-client-version"),
157                HeaderValue::from_static(ICEBERG_REST_SPEC_VERSION),
158            ),
159            (
160                header::USER_AGENT,
161                HeaderValue::from_str(&format!("iceberg-rs/{}", CARGO_PKG_VERSION)).unwrap(),
162            ),
163        ]);
164
165        for (key, value) in self
166            .props
167            .iter()
168            .filter_map(|(k, v)| k.strip_prefix("header.").map(|k| (k, v)))
169        {
170            headers.insert(
171                HeaderName::from_str(key).map_err(|e| {
172                    Error::new(
173                        ErrorKind::DataInvalid,
174                        format!("Invalid header name: {key}"),
175                    )
176                    .with_source(e)
177                })?,
178                HeaderValue::from_str(value).map_err(|e| {
179                    Error::new(
180                        ErrorKind::DataInvalid,
181                        format!("Invalid header value: {value}"),
182                    )
183                    .with_source(e)
184                })?,
185            );
186        }
187
188        Ok(headers)
189    }

191    /// Get the optional OAuth headers from the config.
192    pub(crate) fn extra_oauth_params(&self) -> HashMap<String, String> {
193        let mut params = HashMap::new();
194
195        if let Some(scope) = self.props.get("scope") {
196            params.insert("scope".to_string(), scope.to_string());
197        } else {
198            params.insert("scope".to_string(), "catalog".to_string());
199        }
200
201        let optional_params = ["audience", "resource"];
202        for param_name in optional_params {
203            if let Some(value) = self.props.get(param_name) {
204                params.insert(param_name.to_string(), value.to_string());
205            }
206        }
207
208        params
209    }

211    /// Merge the `RestCatalogConfig` with the a [`CatalogConfig`] (fetched from the REST server).
212    pub(crate) fn merge_with_config(mut self, mut config: CatalogConfig) -> Self {
213        if let Some(uri) = config.overrides.remove("uri") {
214            self.uri = uri;
215        }
216
217        let mut props = config.defaults;
218        props.extend(self.props);
219        props.extend(config.overrides);
220
221        self.props = props;
222        self
223    }

227struct RestContext {

231    /// It's could be different from the user config.

235/// Rest catalog implementation.
236#[derive(Debug)]
237pub struct RestCatalog {
238    /// User config is stored as-is and never be changed.
239    ///
240    /// It's could be different from the config fetched from the server and used at runtime.
241    user_config: RestCatalogConfig,
242    ctx: OnceCell<RestContext>,
243}
244
245impl RestCatalog {
246    /// Creates a `RestCatalog` from a [`RestCatalogConfig`].
247    pub fn new(config: RestCatalogConfig) -> Self {
248        Self {
249            user_config: config,
250            ctx: OnceCell::new(),
251        }
252    }
253
254    /// Gets the [`RestContext`] from the catalog.
255    async fn context(&self) -> Result<&RestContext> {
256        self.ctx
257            .get_or_try_init(|| async {
258                let client = HttpClient::new(&self.user_config)?;
259                let catalog_config = RestCatalog::load_config(&client, &self.user_config).await?;
260                let config = self.user_config.clone().merge_with_config(catalog_config);
261                let client = client.update_with(&config)?;
262
263                Ok(RestContext { config, client })
264            })
265            .await
266    }
267
268    /// Load the runtime config from the server by `user_config`.
269    ///
270    /// It's required for a REST catalog to update its config after creation.
271    async fn load_config(
272        client: &HttpClient,
273        user_config: &RestCatalogConfig,
274    ) -> Result<CatalogConfig> {
275        let mut request_builder = client.request(Method::GET, user_config.config_endpoint());
276
277        if let Some(warehouse_location) = &user_config.warehouse {
278            request_builder = request_builder.query(&[("warehouse", warehouse_location)]);
279        }
280
281        let request = request_builder.build()?;
282
283        let http_response = client.query_catalog(request).await?;
284
285        match http_response.status() {
286            StatusCode::OK => deserialize_catalog_response(http_response).await,
287            _ => Err(deserialize_unexpected_catalog_error(http_response).await),
288        }
289    }
290
291    async fn load_file_io(
292        &self,
293        metadata_location: Option<&str>,
294        extra_config: Option<HashMap<String, String>>,
295    ) -> Result<FileIO> {
296        let mut props = self.context().await?.config.props.clone();
297        if let Some(config) = extra_config {
298            props.extend(config);
299        }
300
301        // If the warehouse is a logical identifier instead of a URL we don't want
302        // to raise an exception
303        let warehouse_path = match self.context().await?.config.warehouse.as_deref() {
304            Some(url) if Url::parse(url).is_ok() => Some(url),
305            Some(_) => None,
306            None => None,
307        };
308
309        let file_io = match warehouse_path.or(metadata_location) {
310            Some(url) => FileIO::from_path(url)?.with_props(props).build()?,
311            None => {
312                return Err(Error::new(
313                    ErrorKind::Unexpected,
314                    "Unable to load file io, neither warehouse nor metadata location is set!",
315                ))?;
316            }
317        };
318
319        Ok(file_io)
320    }

327    async fn list_namespaces(
328        &self,
329        parent: Option<&NamespaceIdent>,
330    ) -> Result<Vec<NamespaceIdent>> {
331        let context = self.context().await?;
332        let endpoint = context.config.namespaces_endpoint();
333        let mut namespaces = Vec::new();
334        let mut next_token = None;
335
336        loop {
337            let mut request = context.client.request(Method::GET, endpoint.clone());
338
339            // Filter on `parent={namespace}` if a parent namespace exists.
340            if let Some(ns) = parent {
341                request = request.query(&[("parent", ns.to_url_string())]);
342            }
343
344            if let Some(token) = next_token {
345                request = request.query(&[("pageToken", token)]);
346            }
347
348            let http_response = context.client.query_catalog(request.build()?).await?;
349
350            match http_response.status() {
351                StatusCode::OK => {
352                    let response =
353                        deserialize_catalog_response::<ListNamespaceResponse>(http_response)
354                            .await?;
355
356                    let ns_identifiers = response
357                        .namespaces
358                        .into_iter()
359                        .map(NamespaceIdent::from_vec)
360                        .collect::<Result<Vec<NamespaceIdent>>>()?;
361
362                    namespaces.extend(ns_identifiers);
363
364                    match response.next_page_token {
365                        Some(token) => next_token = Some(token),
366                        None => break,
367                    }
368                }
369                StatusCode::NOT_FOUND => {
370                    return Err(Error::new(
371                        ErrorKind::Unexpected,
372                        "The parent parameter of the namespace provided does not exist",
373                    ));
374                }
375                _ => return Err(deserialize_unexpected_catalog_error(http_response).await),
376            }
377        }
378
379        Ok(namespaces)
380    }

526    /// Create a new table inside the namespace.
527    ///
528    /// In the resulting table, if there are any config properties that
529    /// are present in both the response from the REST server and the
530    /// config provided when creating this `RestCatalog` instance then
531    /// the value provided locally to the `RestCatalog` will take precedence.
532    async fn create_table(
533        &self,
534        namespace: &NamespaceIdent,
535        creation: TableCreation,
536    ) -> Result<Table> {
537        let context = self.context().await?;
538
539        let table_ident = TableIdent::new(namespace.clone(), creation.name.clone());
540
541        let request = context
542            .client
543            .request(Method::POST, context.config.tables_endpoint(namespace))
544            .json(&CreateTableRequest {
545                name: creation.name,
546                location: creation.location,
547                schema: creation.schema,
548                partition_spec: creation.partition_spec,
549                write_order: creation.sort_order,
550                stage_create: Some(false),
551                properties: if creation.properties.is_empty() {
552                    None
553                } else {
554                    Some(creation.properties)
555                },
556            })
557            .build()?;
558
559        let http_response = context.client.query_catalog(request).await?;
560
561        let response = match http_response.status() {
562            StatusCode::OK => {
563                deserialize_catalog_response::<LoadTableResponse>(http_response).await?
564            }
565            StatusCode::NOT_FOUND => {
566                return Err(Error::new(
567                    ErrorKind::Unexpected,
568                    "Tried to create a table under a namespace that does not exist",
569                ));
570            }
571            StatusCode::CONFLICT => {
572                return Err(Error::new(
573                    ErrorKind::Unexpected,
574                    "The table already exists",
575                ));
576            }
577            _ => return Err(deserialize_unexpected_catalog_error(http_response).await),
578        };
579
580        let metadata_location = response.metadata_location.as_ref().ok_or(Error::new(
581            ErrorKind::DataInvalid,
582            "Metadata location missing in `create_table` response!",
583        ))?;
584
585        let config = response
586            .config
587            .unwrap_or_default()
588            .into_iter()
589            .chain(self.user_config.props.clone().into_iter())
590            .collect();
591
592        let file_io = self
593            .load_file_io(Some(metadata_location), Some(config))
594            .await?;
595
596        let table_builder = Table::builder()
597            .identifier(table_ident.clone())
598            .file_io(file_io)
599            .metadata(response.metadata);
600
601        if let Some(metadata_location) = response.metadata_location {
602            table_builder.metadata_location(metadata_location).build()
603        } else {
604            table_builder.build()
605        }
606    }

608    /// Load table from the catalog.
609    ///
610    /// If there are any config properties that are present in both the response from the REST
611    /// server and the config provided when creating this `RestCatalog` instance, then the value

613    async fn load_table(&self, table_ident: &TableIdent) -> Result<Table> {
614        let context = self.context().await?;
615
616        let request = context
617            .client

620
621        let http_response = context.client.query_catalog(request).await?;
622
623        let response = match http_response.status() {
624            StatusCode::OK | StatusCode::NOT_MODIFIED => {
625                deserialize_catalog_response::<LoadTableResponse>(http_response).await?
626            }
627            StatusCode::NOT_FOUND => {
628                return Err(Error::new(
629                    ErrorKind::Unexpected,
630                    "Tried to load a table that does not exist",
631                ));
632            }
633            _ => return Err(deserialize_unexpected_catalog_error(http_response).await),
634        };
635
636        let config = response
637            .config
638            .unwrap_or_default()
639            .into_iter()
640            .chain(self.user_config.props.clone().into_iter())
641            .collect();
642
643        let file_io = self
644            .load_file_io(response.metadata_location.as_deref(), Some(config))
645            .await?;

727    async fn update_table(&self, mut commit: TableCommit) -> Result<Table> {
728        let context = self.context().await?;
729
730        let request = context
731            .client
732            .request(
733                Method::POST,
734                context.config.table_endpoint(commit.identifier()),
735            )
736            .json(&CommitTableRequest {
737                identifier: commit.identifier().clone(),
738                requirements: commit.take_requirements(),
739                updates: commit.take_updates(),
740            })
741            .build()?;
742
743        let http_response = context.client.query_catalog(request).await?;
744
745        let response: CommitTableResponse = match http_response.status() {
746            StatusCode::OK => deserialize_catalog_response(http_response).await?,
747            StatusCode::NOT_FOUND => {
748                return Err(Error::new(
749                    ErrorKind::Unexpected,
750                    "Tried to update a table that does not exist",
751                ));
752            }
753            StatusCode::CONFLICT => {
754                return Err(Error::new(
755                    ErrorKind::Unexpected,
756                    "CommitFailedException, one or more requirements failed. The client may retry.",
757                ));
758            }
759            StatusCode::INTERNAL_SERVER_ERROR => {
760                return Err(Error::new(
761                    ErrorKind::Unexpected,
762                    "An unknown server-side problem occurred; the commit state is unknown.",
763                ));
764            }
765            StatusCode::BAD_GATEWAY => {
766                return Err(Error::new(
767                    ErrorKind::Unexpected,
768                    "A gateway or proxy received an invalid response from the upstream server; the commit state is unknown.",
769                ));
770            }
771            StatusCode::GATEWAY_TIMEOUT => {
772                return Err(Error::new(
773                    ErrorKind::Unexpected,
774                    "A server-side gateway timeout occurred; the commit state is unknown.",
775                ));
776            }
777            _ => return Err(deserialize_unexpected_catalog_error(http_response).await),
778        };
779
780        let file_io = self
781            .load_file_io(Some(&response.metadata_location), None)
782            .await?;
783
784        Table::builder()
785            .identifier(commit.identifier().clone())
786            .file_io(file_io)
787            .metadata(response.metadata)
788            .metadata_location(response.metadata_location)
789            .build()
790    }
